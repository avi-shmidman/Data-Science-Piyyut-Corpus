״אַדֶּרֶת תִּלְבּוֹשֶׁת הוֹד וְהָדָר יַקִּיר
אֶת מִי מִלִּין לְסַפֵּר שִׁבְחָהּ?
אֶבֶן הָרֹאשָׁה יְקָרָה וְאֵין עֲרוֹךְ –
אֵין לְהַכְרִיז תְּשׁוּאוֹת שׁוּם שָׁוְויָהּ!
אֵי מְקוֹם סְגוֹר סָגוּר וְסַפִּיר
אֲשֶׁר לְפִי מְלֹאת כְּלִיל הָעֲטָרָה?
אֻומָּן הֲיִמָּצֵא, וְגַם חֲכַם חֲרוֹשֶׁת?
'אִיתִּיאֵל וְאוּכַל' מִי יְמַלֵּל?!

בְּצוּרוֹת לַהֲגִיד, נָבוֹן – מְזִימּוֹת בֶּצֶר,
בְּאֵין תַּחְבּוּלוֹת עַרְכָן וְרַב הַחוֹבֵל:
בְּפִיו חוֹטֶר-זֵיד זֵין-הַקְּרָב.
בִּין בַּעַר יְבֻולַּע וִיפֻולַּג וְאֵינֶנּוּ.
בָּלוּם אוֹצָר, מִי מִכְמָן יְשַׁעֵר?
בָּט לְהוֹסִיף - סוֹפוֹ רַב גְּרוֹעַ.
בְּמוֹתָר - לְמַחְסוֹר. מַה יִּתְרוֹן עֶצֶב?
בְּכֵן מִדַּבֵּר יָפֶה כֹּחַ שְׁתִיקָה.

גִּלָּיוֹן גּוֹדֶל קַו ממרין מֵאֶרֶץ,
גָּמְדֵי רָחְבּוֹ מִנִּי רְחַב יָדָיִם –
גְּוִילָיו גְּבוּל בַּל יוּכַל לְהָכִיל,
גַּם לְהַסְפִּיק לֹא דְיוֹ וְקֻלְמוֹס
גָּבְהֵי שָׁמַיִם, מָה לְּפוֹעַל פּוֹעַל.
גֶּשֶׁר הֲלִיכוֹת עוֹמֶק תְּהוֹם אָין:
גְּדוֹלוֹת לְהוֹדִיעַ בְּלִי סֵפֶר וָחֵקֶר –
גֶּבֶר כְּגִבּוֹר מִי חֲגוּר חֲלָצָיִם?

'דִּבְרֵי נִפְלְאוֹתָיו' - מִי יִתְהַלֵּל - 'בִּינוֹתִי,
דְּרָכָיו לְפַלֵּשׁ?' לְמִי, פְּרוֹשׁ, פֶּלֶס?
דֵּיעַ תֻּומּוֹ מִי יְחַוֶּה דָּעַת?
דּוֹר לְדוֹר יִשְׁבַּח בְּכָל יְכוֹלֶת
דּוֹהַר רַעַם גְּבוּרוֹת, חִין עֶרְכּוֹ.
דִּין דְּרוֹשׁ לְפִי מִקַּח שֶׁמֶץ
דּוֹם תְּהִלּוֹתָיו. נִיב לָא אֶכְלָא
דֵּי דִלְדּוּל הַשֵּׂג אֻומָּנוּת הַמְּלֶאכֶת.

הִגְדִּיל עֲשׂוֹת - וּמִי כְּמַעֲשָׂיו יַעַשׂ?
הִפְלִיא יְחִידִי, כְּמוֹ כֵן יְקֻולָּס.
הַתַּבְנִית, מַלְאֲכוֹת עֲלִיּוֹת וְגַנְזַכִּים וְחֶדֶר,
הַשְׂכֵּל וּמַדָּע, דִּפְתָּר וּפִנְקֵס - מִפִּיהוּ.
הִצְנִיעַ יוֹמַיִם אָמוּן אֱמוּנָה מְיֻומֶּנֶת,
הִיא לְבַנָּאִית וְצָרְכֵי בִנְיָן מְסַפֶּקֶת.
הוּא לְבַדּוֹ כֹּה וָכֹה וּבַתָּוֶךְ
הִמְתִּיחַ, הִטְפִּיחַ, וּתְפוֹסֶת-מִי אִתּוֹ?

וּמִדַּת בָּשָׂר - מַרְחִיב וּמֵצֵר וְעוֹלֶה,
וְעֶלְיוֹן נוֹרָא עִילִּית לְתַחְתִּית הִקְדִּים.
וּבָנָה וְהִטְלִיל בְּלֹא מִשְׁחָה וּמָשׁוֹחַ,
וּמַסָּר וָפֶסֶל וְכָל כְּלֵי בַּרְזֶל,
וְשִׁבֵּץ וְחִדֵּק בְּלֹא מַעְגָּל וּמַחֲלָצָיִם,
וְשִׁפָּה וְיִפָּה בְּלֹא רָהִטְנֵי וְאִזְמֵל,
וְחִזֵּק וְקָבַע בְּלֹא קֻרְנָס וּמַסְמְרִים
וְשִׁפֵּר וְתֵאֵר בְּלֹא כַרְכּוֹם וְכַרְכּוֹב.

זְרוּתִים תִּכֵּן, וְשָׂם שְׁמָם שָׁמָיִם:
זִיו-סַמִּים, שָׁמִים, וּמַבִּיט יִשּׁוֹם.
זָהֳרָם עֶצֶם בָּרוּר, לֹא מַחֲלִיד,
זְמַנֵּי זְמַנִּים תּוֹלְדוֹתָם לֹא לְיַשֵּׁן.
זִיקַת יְצִיקָתָם עָשׂ בְּרִיקּוּעַ פַּחִים,
זַכִּים, בְּרִיאִים, בְּלִי תוֹרֶף וְרִפְיוֹן.
זְקָפָם כְּטַנְדֵּס, כְּפָפָם, קִרְקְסָם בָּאוֹקְיָאנוֹס,
זְבוּלוֹ קָדְשׁוֹ עֲרָבוֹת רֶכֶב רְכוּבוֹ.

חִישֵּׁר שֶׁלֶג והקדיש קְלִיטַת מָיִם:
חוֹרֵי עָפָר דִּיבֵּק רְגָבִים וְכֵפִים.
חֶלֶד כִּיוֵּן מוּל כֵּס - שְׁרַפְרָף,
חֲנִיַּית יְצוּרִים בָּתֵּי חוֹמֶר.
חִלֵּק לְהַשְׁלִים נְצִיבֵי מְדַיְּנֵי תַחֲרוּת:
חָצַב תְּחוּמָם - לָשׂוֹר בְּלִי הַשֵּׂג.
חִמּוּשׁ עוֹלֶה חֶשְׁבּוֹנוֹ בְּחֻומְשֵׁי חֻומָּשִׁים,
חִבְּאוֹ, גְּנָזוֹ, מְנָת מֶלֶךְ לַבֵּן.

טִיף, טָעַם, מִלַּת 'יְהִי רָקִיעַ',
טִיפָּה הִגְלִיד אֲשֶׁר בְּאֶמְצַע בִּנְתָּיִם,
טַכְסִיס קִדּוּחַ-אֵשׁ מְחִיצַּת מַחְצָה.
טִיף מָסוּ לְהַעֲמִיד רִפְרוּף לִחוּי
טְעוּנֵי אַגָּן, כִּיפָּה עָפוֹת מַמְזֶגֶת
טַבּוּר לְשׁוֹקֵק, וּפֶלֶג מָלֵא מָיִם.
טוֹרַד, סֻוגַּר, חֲלַל עוֹמֶק וָרוֹחַב,
טִפְטֵף לְתוֹכוֹ נַחַל מְנָת מוֹרְדָיו.

יָעַץ לְפַנּוֹת פְּנֵי תֵבֵל מְכַסִּים,
'יִקָווּ' שָׂח, וְנָסוּ, פָּחֲדוּ מִקּוֹל.
יִקְבֵי הַמֶּלֶךְ אֻוסְּפוּ, נִגָּרִים בַּמּוֹרָד.
יָם קְלָטָם, דֻּוכּוּ דּוֹךְ נְבָכָיו.
יִשֵּׁר פּוֹעַל, אִישֵּׁר חֵיל עִזּוּזוֹ,
יִיסַּד מְקוֹמָם מְעַט מַחֲזִיק מְרֻובֶּה,
יִיסַּד כֵּן חֹפֶן, סֶלַע וְחָצֵר,
יָרַד בַּקֹּדֶשׁ, בַּדֵּי אֲרוֹן וְשָׁלֵם.

כְּנוּסִים כְּבָרְחוּ, וּפְנֵי תֵבֵל חוֹרְבוּ
כֻּוסּוּ וּמָלְּאוּ מַזְרִיעַ וְזַרְעוֹ בוֹ,
כִּלְכֵּל - סְעוֹד, עַדֵּן-זִיו וְשַׂמֵּחַ,
כָּסוּף לְמַרְאֶה, מָתוֹק וְטוֹב לְמַאֲכָל –
כַּנַּת-נֶטַע, גַּן יְרֻושַּׁת אוֹהֲבָיו:
כׇּורְתּוֹ כְּנוֹפוֹ וְעֵץ דַּעַת בְּתוֹכוֹ.
כִּוְּנוּ יַחַד צִוּוּי קַל וָחוֹמֶר,
כְּבוֹד מֶלֶךְ אָז בְּמַעֲשָׂיו שָׂשׂ.

לִבָּה מְאוֹרוֹת, מַזָּלוֹת, חֲיָילוֹת וְרָהְטוֹן,
לִגְיוֹן וְגַסְטְרָא, קַרְטוֹן, צְבָא שַׁחַק:
לַהֲקַת חֻקּוֹת, תַּפְקִיד לֹא יְשַׁנּוּ,
לְבָנָה יְחַדֵּשׁ, תִּפְאֶרֶת עֲמוּסֵי בֶטֶן.
לַמַּקְטִין עַצְמוֹ כִּי בָא בִתְחוּם
לְוָויָה סִדֵּר, כְּעֵין בּוֹלֵי וְדֵימוֹס.
לוֹ יוֹם, רְשׁוּתוֹ גַּם לָיְלָה,
לְשִׁירוֹ לְלַיְלָה וְגַם לְקִלּוּסוֹ יוֹם.

מָעוֹת יַמִּים הִשְׁרִיץ רׇוב-רֶמֶשׂ
מְפָרֵשׁ עוֹבֵר אָרְחוֹת מִשְׁבְּרֵי גַלֵּימוֹ,
מַכּוֹלֶת לחודים אֲסֵיפַת קְבוּצֵי שִׁרְיָין
מָגִינֵּי אֲפִיקִים, לַחְתּוֹם לְיוֹם חֲתוּמִים.
מֵרְקַק-מַיִם הִכְנִיף לְקַשְׂקְשֵׂי רֶגֶל
מַגְבִּיהֵי עוּף לָטוּשׂ עֲלֵי אוֹכֶל.
מְצֻויָּינִים שְׁלֹשָׁה - נִתָּרִים בְּהַכְשֵׁר אֶחָד,
מַאֲכַל שֻׁלְחָנוֹ - זִיז-שָׂדַי-עִמָּדִי.

נֶפֶשׁ חַיָּה לְפַלֵּט גַּיְא כְּעֻובֶּרֶת
נִחוּצוֹ צִוָּה, וַיַּעֲמוֹד כֵּן, וַיֶּהִי.
נִטְהַר לַהֶדְיוֹט - נִבְחַר גַּם לַגָּבוֹהַּ,
נִפְסַל מִזֶּה - נִתְעַב וְנֶאֱלַח מִזֶּה.
נִסְמַן, נִמְסַר, בְּמַרְאֶה וְלֹא בְחִידוֹת:
נָבִיא מׇרְאֶה, וְלֹא קְנֵיגִי וּפָלֶסְתָּרִי.
נָהָר יַעֲשׁוֹק וּבוּל אֶלֶף יִרְעֶה
נוֹצַר וְנָצוּר הָעוֹשׂוֹ יַגֵּשׁ חַרְבּוֹ.

סְעֻודַּת עוֹנֶג, טְבוֹחַ טֶבַח וְהָכֵן,
סֹלֶת אֲפוּיָה לְפַת-בַּג הַמֶּלֶךְ,
סוֹךְ מָזוּג רָב, כְּיַד הַמֶּלֶךְ,
סֵדֶר עָרוּךְ, וְאֵין מִי לְהָסֵב.
סֵפֶר הַיַּחַס נִמְצָא רֹאשׁ קְרוּאִים
סוֹף. הִצְבִּיר גּוּשׁ גְּבוּלַת גָּולְמוֹ, 
סִיֵּים וְקִיֵּים, הֵכִין קוֹמַת קוֹמְמִיּוּת:
סֻולָּם מֻוצָּב - וְרֹאשׁוֹ מַגִּיעַ הַשָּׁמָיְמָה.

עָרִיס הוּרַד לְפַרְדֵּס לֹא-נֶעֱבָד,
עַסְקָן וּפֻונְדְּקֵי מְלַאי מִסְּחַר חָרוּץ,
עִיטֵּר אָדוֹן כְּבוֹד שֵׁם תְּנָאָן.
עֲרֹךְ לַיְצוּרִים אֲשֶׁר נִקְּבוּ בְּשֵׁמוֹת –
עֵזֶר כְּנֶגְדּוֹ פַּרְצוּף אָחוֹר שִׁוָּה:
עָלוּ שְׁנַיִם מִיָּד יָרְדוּ אַרְבָּעָה.
עֲנָיַית-שָׁוְא עָן עָלֵי יוֹצֵר
עָרוּם לְהָרַע וּכְסִיל לְהוֹצִיא דִבָּה.

פָּשְׁטָה לְקַבֵּל חִוּוּי חֵיו חַוָּה,
פִּיהָ פָּתְחָה לִגְלוּג גְּנַאי גַּרְגְּרָנִית.
פָּרְשָׂה מְצוֹדֶיהָ לְהַשְׁמִיד וּלְהַחֲרִים רַבָּה,
פִּקְפְּקָה בְּדָם, בְּנֵר וְחִלּוּל חַלָּה.
פָּרִיצֵי פֶרֶץ שְׁלֹשָׁה עוֹלֵי גַרְדּוֹן –
פְּסַק הַדִּין גַּם מֵנֶקֶת מְאָרֵר.
פְּקֻודַּת מָוֶת אָז לַכֹּל כְּנִקְנְסָה,
פּוֹרַשׁ עוֹפֶר לְתַעֲנִית וְעִנּוּי כִּפּוּר.

צִיצֵי פִטְרָם הֻופְרְדוּ: אִיכָּר וְנוֹקֵד,
צוּר עוֹלָמִים אִישׁ מִנְחָתוֹ הִדְרִינוּ.
צָעִיר כְּהוּשַׁע - קִנֵּא רַב, וְאָרָב.
צָדַק כְּשָׁב, גְּזַר דִּינוֹ לְבַצֵּעַ.
צוּרַת תַּעְתּוּעִים רֵאשִׁית חַטָּאת שְׁלִישִׁי –
צָפוּ מַיִם צַפְצֵף עוֹלָם שְׁלִישׁוֹ.
צִיקַת מַיִם גָּאוּ, שָׂחוּ: 'סוּר',
צְרוּרִים פִּתַּח, אֲצוּרִים פָּתַח וְאִבְּדָם.

קִימָם הִכְחִיד וְיִתְרָם אָכְלָה אֵשׁ;
קָשׁוּ מִדַּי וַתִּשָּׁבֵר כָּעֵץ עַוְלָה.
קְלוּטֵי מַסְגֵּר, נְפָשׁוֹת עַם לוֹקֵחַ
קַיִץ וָחוֹרֶף בְּרִית מֶלַח קָמָה.
קֶלַח מְשׁוֹאָר עָשׂ ובַדִּים וּפֹארוֹת,
קִמְּשׂוֹנֵי בִקְעָה סֻובְּכוּ סִירִים סְבוּאִים:
קִיר חֲרֶשֶׂת, זִימַּת אִוֶּלֶת בָּנוּ,
קִינָּם שִׂים בֵּין כּוֹכְבֵי שָׁחַק.

רָאשֵׁי תַנִּינִים, כְּבָר מַכְתֶּשֶׁת קְבוּעָה
רָאוּ מְחָרְפִים - וּמוּסָר לֹא לָקָחוּ.
רוֹעַ מְגוּרָתָם וְקַו מִדָּתָם בָּא:
רֻוסַס וּבֻוקַּעַ שֻׁוקַּע שׁוּר שְׁרִירוּתָם.
רֵיעָה וְאִיחָה יָחִיד יְחִיד עוֹלָם,
רְחוֹקִים קָרֵב קִרְבַת לְקֹרְאָיו קָרוֹב.
רָץ וְהִרְצָה תּוֹעִים אַחַר מִנְהָג,
רֵיחַ הִפְרִיחַ רִאשׁוֹן בָּעֲשִׂירִי כְּנֻוסָּה.

שָׁמַע ושבח מוֹלָדוֹ וְחֶדֶר הוֹרָיו.
שְׁמִינִי כְּקִבֵּל - שֻׁלַּם, טֹהַר וְקֻדַּשׁ.
שָׁלוּחַ כְּפוֹרַחַת, פְּרִי קֹדֶשׁ הַלּוּלִים.
שָׁקוּל כְּמִבְחָר הַסֵּךְ נֶסֶךְ שֵׁכָר,
שְׁאֵלוֹ לַשֵּׁם לְהַר הַמּוֹר לְהַעֲלוֹ.
שָׁקֵד זָקֵן פִּי מֶלֶךְ שְׁמוֹר,
שׁוּרַת בָּחוּר בְּיוֹם חֲתֻונַּת חֻופָּתוֹ:
שְׁנֵיהֶם שָׁוִים עֲשׂוֹת רְצוֹן קוֹנָם.

תּוֹכֵן כְּחָז צַדִּיק נִבְחָן וְנֶאֱמָן –
תִּרְגֵּל וְהֵרִיץ שֶׂה זֻומַּן חֲלִיפָתוֹ.
תָּמִיד בַּעֲדֵנוּ אֵפֶר צִבּוּר יֵרָאֶה
תְּמוּר עוֹלָה, אֱמוּרֵי מִקְטָר, קְטֹרֶת.
תּוֹלְדוֹת חֲלָצָיו, עָקוּד עָקוֹד כְּהוֹלִיד,
תּוֹמִים מִבֶּטֶן גְּדֵלִים הרס, חָרוּל.
תִּפְאֶרֶת עֲטֶרֶת חֵלֶק חוּט מְשֻׁולָּשׁ,
תָּם, יוֹשֵׁב תַּרְבִּיץ אָהֳלֵי דָת.

תַּבְנִית דְּיוֹקְנוֹ נֶחְתַּם בְּכֵס מָעַל,
תַּחַת כֵּן עוֹלִים וְיוֹרְדִים חָלָם.
תִּתּוֹ מָקוֹם עַד עֲבוֹר חֲצוּפָה
תְּנָיִים הִדִּיר וְהוּשַׁב 'כֵּן', וְנֶאֱמָנוּ.
תָּמוּךְ בְּתֻומּוֹ בִּימִין צֶדֶק וָחָיִל
תָּקְפּוֹ נִתַּן, הַמְשֵׁל וְעֵזֶר פָּחַד;
תְּמוּרוֹ יְשֻׁונַּן: 'אַשְׁרֵי יְלוּד אִשָּׁה
תּוֹבֵעַ צְרָכָיו וּבוֹרֵא מַבְטִיחַ וְשׁוֹמֵר!'

שִׁבּוֹלֶת עָבַר בְּמַקְלוֹ וּקְלִי צִקְלוֹנוֹ,
שָׁב - וּבְיָדוֹ מַטּוֹת וְעוֹשֶׁר רָב.
שְׁתִילֵי זַיִת, סֵדֶר הֶיקֵּף שֻׁולְחָנוֹ,
שְׁנֵים עָשָׂר, כְּמִנְיָן מַזְּלֵי שָׁחַק:
שְׁלֵימָה מִטָּה, כֻּלָּהּ בְּלִי מוּם,
שִׁבְטֵי יָהּ, עֵדוּת לֹא תִשָּׁכֵחַ.
שְׁלִישִׁי עָל, לְאָחוֹר עֲשִׂירִי, קֹדֶשׁ,
שׁוֹעֵר, שׁוֹמֵר, מְשׁוֹרֵר, מְשָׁרֵת, מְבָרֵךְ.

רֻוגַּל מִצּוֹעַן מָסוּר בְּקִדּוּשׁ שֵׁם,
רִיב מְרִיבָה וְעוֹל מַסָּה מְקַבֵּל:
רַק לְבַדּוֹ מִסִּין עומר בְּמֵימָיו
רָאוּי וְהָגוּן עֲמוֹד בְּהֵיכַל מֶלֶךְ.
רוֹעֶה מִקִּרְבּוֹ אֲשֶׁר לְאֶבֶן קָם,
רַב רְבָבָה, אָב שֻׁולַּשׁ וְשָׁלִישׁ,
רָם עֲנָנוֹ עִילּוֹ, כִּיסּוֹ וְקִדְּשׁוֹ,
רַשַּׁאי וְרָצוּי בֹּא בְּמַתַּן שְׁכִינָה.

קֹדֶשׁ קָדָשִׁים נִבְדַּל קְדוֹשׁ יְיָ,
קֶרֶן כְּהֻונָּה לוֹ לָעַד לְכָבוֹד,
קִנְיַן אֲחֻוזָּה, תְּקוּעָה וְלֹא תָמוּשׁ:
קוֹשְׁרָיו לְקַלַּחַת קָרַחַת וזוֹרַחַת פֹּרַחַת.
קָרוֹב וּבֻוחַר, אֻושַּׁר שְׁכוֹן חֲצֵרִים,
קָבוּעַ וְשָׁתוּל וְרַעֲנָן בְּהֵיכְלֵי מֶלֶךְ:
קְשׁוּרִים יַחַד, בּוּר וּמֵימָיו חֲבִיבִין,
קֻויַּם לַצַּדִּיק: אַשְׁרֵי בָנָיו אַחֲרָיו.

צָלַח בַּעֲנָוָה לְרִיבּוּי וּמְשִׁיחַת שְׂרָרָה
צָבָא בְּפִתּוּי וּפֵירַשׁ וְשִׁמֵּשׁ וְחֻונַּךְ,
צִוּוּי 'כָּכָה' וְעִנְיַן 'כִּי כֵן'
צֻויַּין לְצֶאֱצָאָיו לְצוֹרֶךְ קְדֻושָּׁה וְטָהֳרָה.

צְרוּפַת שִׁבְעָתַיִם לִשְׁכַּת צִיר, צְתִיָּיתָהּ
צְבִי קֹדֶשׁ, צִיּוֹן, יַרְכְּתֵי צָפוֹן,
צַד מְכֻוָּון עַל פְּנֵי בִירָה,
צֵלָע צָפוֹנָה, מִקְצוֹעַ פְּאַת מִזְרָחָה.

פְּקֻודַּת עָשׂוֹר: פְּקִיד כֹּהֵן הָרֹאשׁ
פּוֹרַשׁ מִבֵּיתוֹ, חָרֵד, חָשׁ לַסָּפֵק.
פִּנַּת פַּרְהֶדְרִין חֶדֶר מִטַּת אַפִּרְיוֹנוֹ,
פַּתְשֶׁגֶן שעירים כְּתָב-הַדָּת לְלִשְׁכָּתוֹ.
פּוֹעַל עֲבוֹדוֹת עָלָיו כָּל שִׁבְעָה;
פְּנוֹת הַיּוֹם יֻזֶּה וְאֵין מַפְלִיג.
פְּרוּשִׁים מַשְׂכִּילָיו קוֹרִין וּמַקְרִין סֵדֶר,
פֶּן שָׁכַח, וְעוֹד לֹא לָמַד.

עוֹד יַכִּיר בֹּקֶר וְעֶרֶב הַחַרְסָה
עוֹלוֹת וְחַטָּאוֹת וּמוּסָפִין, שְׂעִירֵי עָם.
עָרֵב לוֹ כָּל שִׁבְעָה לְשָׂבְעָה,
עָצוּר לוֹ כָּל טֻומְאָה מַרְגִּיל.
עִם חֲשֵׁיכָה, תְּשִׁיעִי, יֶחְסַר לַחְמוֹ
עֲבוּר מִלְּהַמְתִּיק רׇוב-אוֹכֶל שְׁנָתוֹ.
עֲמִיתָיו בַּמֶּרְקָחָה יְאַמְּנוּ חֳופָנָיו וְיַשְׁבִּיעוּהוּ
עַל דַּעְתָּם וְעַל דַּעַת בֵּית דִּין,

'סוּר מֵרָע וְטוֹב לֹא תְשַׁנֶּה'
סַמִּים מִלְּקַלְקֵל וְכַף עֵגֶל תְּדִיקֶנּוּ.
סַב וּבוֹכֶה כְּנִדּוֹן לְכַף רֶשַׁע,
סָבִיב כְּנֶגְדּוֹ, כִּי נָתְנוּ דוֹפִי.
סִיעַת לְוָיָתוֹ בְּנוֹעַם עֲרֵבִים יְסֻובּוּהוּ
סֶרֶךְ הַגּוֹרֶן, סַדֵּר יַחַד כְּשַׁקְדָנִים.
סַחַר פֶּלֶס כִּסּוֹ אִם יֶהִי
סוֹדֵר וְקוֹרֵא. וְרֵיק - קוֹרִין לְפָנָיו.

נִצָּנֵי לְוִיָּה, מְנַצְּחֵי עַל שׁוֹשַׁנִּים,
נִמְנוּמוֹ לְהַעֲבִיר, קוֹל צְרֵדָתָם יַשְׁפִּירוּ.
נוֹאֲמִים לוֹ: קוּם קִדָּה לְהַרְאוֹת,
נְעִיצַת גֻּודָּלִים, שְׁחוֹחַ, נָשׁוֹק וְזָקוֹף.
נֵבֶל וְכִנּוֹר לְהַעֲסִיקוֹ לֹא יְנַעְנְעוּ,
נוֹצֵר וּבַנַּאי אָדוֹן שִׁיר יְנַגְּנוּ.
נְדִיבֵי שָׁלֵם גַּם שְׁנָתָם נִגְזֶלֶת,
נֶחְבַּר קוֹל, וְיָדִיד לֹא נֶחְטָף.

מַשְׁכִּימֵי קוּם, מֵאַחֲרֵי שֶׁבֶת הַמּוֹעֵד,
מְהִירֵי מִצְוֹת, מַקְדִּימִים לַבֹּקֶר בֹּקֶר.
מִבֹּא בְדָמִים מְחַבְּבֵי מִצְוֹת חֲשׁוּבוֹת
מְטִילִין חֲלָשִׁים, וְאֵין פֶּגַע וָשֶׁבֶר.
מְפַיְּיסִין וְחוֹזְרִין עַד אַרְבָּעָה פְיָיסוֹת.
מַמְתִּיקִין סוֹד בְּבֵית אֱלֹהִים בְּרָגֶשׁ.
מְנוֹרָה וּמִזְבְּחוֹת פַּיִיס שְׁנֵי כְּסֵדֶר,
מְעַשֶּׁרֶת שְׁלִישִׁית, חֲדָשִׁים, לֹא שׁוֹנִים.

לַמִּזְבֵּחַ מִכֶּבֶשׁ - חֲדָשִׁים גַּם יְשָׁנִים,
לַעֲבוֹדָה וַעֲבוֹדָה, וְזָכָה מִי שֶׁזָּכָה.
לִבְלוּב לְבָנָה דִּיקְרָנֵי אַיֶּלֶת כְּהִקְרִין
לַגָּג לְחִזָּיוֹן עֲמוֹד מְצַפֶּה הִנְהִיגוּ.
לְעֵת יְבַשֵּׂר רוֹאֶה 'בּוּרְקַאי' בֶּעָשׂוֹר,
לְמִקְוֵה חוֹל טְבִילוֹתָיו חָמֵשׁ רָד.
לֶכֶת הַצְנֵעַ בְּדֶרֶךְ כָּבוֹד נִכְבָּד
לחבר עֲבוֹדָה מְחִיצַּת בּוּץ הִסְדִּינוּ.

כִּיוֵּן וּפָשַׁט וְטָבַל וְעָלָה וְסִפֵּג,
כְּלִי חֶמְדָּה, הִתְפָּאֶר-בָּם, הִקְרִיבוּ.
כְּמִין פִּמְלְיִינָא שנום מִכְנְסֵי שְׁנָצִים,
כִּסּוּי נַבְלוּת סִיג סוּגַת שׁוֹשַׁנִּים.
כְּתוֹנֶת פַּסִּים מֵפֶסֶת פַּס דָּם,
כִּלְאַיִם מְחַפָּה, נָלוֹז וּמַלִּיז מְיַשֶּׁרֶת.
כִּשְׁרוֹן כֹּחַ אַבְנֵט שְׁלֹשִׁים וּשְׁתַּיִם,
כְּנֶגֶד עַקְמָנִים - פָּנִים וְאָחוֹר עוֹקֵם.

כּוֹתֶרֶת עַמּוּד גֻּולַּת רֹאשׁ מִצְנֶפֶת,
כִּפּוּר צוֹנֶפֶת בְּעַד רָמֵי רוּחַ.
כְּבֻודַּת אֵלֶּה תַּכְרִיךְ בּוּץ פְּנִימָה
כַּגָּדוֹל כְּהֶדְיוֹט לְשָׁם. אַרְבַּעְתָּם חוּצָה:
כְּלִיל תְּכֵלֶת מַעֲטֶה מְעִיל צֶדֶק,
כְּרוּם יָם, גּוֹאֵן גֵּוו גְּבוֹהִים,
כָּפוּל לְתוֹכוֹ בֵּית צַוָּארוֹ וּמִמֶּנּוּ,
כְּשִׁרְיָין שְׂפָתוֹ מַעַשׂ אֻומָּן גַּרְדִּי.
כַּנַּת שׁוּלָיו רִמּוֹן וְזָג זָהָב,
כְּלָלָם בְּמִנְיַן מֵאָה וְאַרְבָּעָה וְאַרְבָּעִים.
כִּעְכּוּעַ בָּשָׂר רָם לֶעָנָו דִּיגֵּם,
כִּשְׁכּוּשׁ רַע בּוֹ בְּקוֹל יְכֻופָּר.

יָאָה מְלֶאכֶת אֵפוֹד מַעֲשֵׂה חוֹשֵׁב,
יֳופִי חֲמִשָּׁה מַטְוֶה, וּקְצִיצַת פְּתִילִים.
יִקְרַת שֹׁהַם מִפֹּה וּמִפֹּה לְחוֹבְרוֹתָיו,
יְלָדִים כְּתוֹלְדוֹתָם פִּתּוּחֵי חוֹתָם מְפֻותָּחוֹת.
יְחוּסִים בְּאַחַת שִׁשָּׁה, בְּעֶשְׂרִים וַחֲמִשָּׂה,
יִתְרָם בַּשְּׁנִיָּה, מְלֵאִים עוֹלִים חֲמִשִּׁים.
יִשְׂרָאֵל לְזִכָּרוֹן אֲפֻודָּה זֵר זֵרוּי,
יַעֲקֹב יְנַקֶּה הַפְצַר אָוֶן וְעָמָל.

יְקֻובַּע בּוֹ חֹשֶׁן יָחִישׁ חֲשָׁאִים,
יָדוֹת וּמׇואֳחָזִים מִזֶּה וּמִזֶּה לְמַחְבֶּרֶת.
יְרִיַּית אֲבָנָיו מוּל שְׁבָטִים לְפִתּוּחַ,
יָרוֹק וְאָדוֹם וּשְׁאָר מִינֵי צְבָעִים
יָשָׁר מְחֻוקֶּה, שִׁבְטֵי יְשֻׁורוּן וְאָבוֹת.
יָאִיר יַשְׁלִים, כּוֹרֵת וּמֻופְלָא וּבָחוּן –
יוֹשֵׁב שֶׁבֶת וִישִׁיבַת גַּנִּים נִשְׁאָלִים:
יִטּוּ וְיַטּוּ - עוֹל קִלְקוּלָם מְתַקֵּן.
יַחַד עִמָּם צְפִירַת צִיץ, חוֹתָמוֹ:
יוֹד הֵא קוֹף דָּלֶת שִׁין,
יְהִירִים מְשַׁקֵּעַ, זֵדִים מוֹסֵר לְפֻורְעָנוּת,
יְחִידִים לְרָצוֹן, לְכַפָּרַת עַזּוּת מֵצַח.

טַלֵּת נָאָה, תַּכְשִׁיט שְׁמוֹנָה אֵלֶּה,
טמום חֲמִשָּׁה, זַךְ זָכָה, וְזִכָּה
טֶפֶשׁ טוֹפְשָׁה. עוֹטֵף וְעוֹטֵר וְהוֹלַמְתּוֹ;
טָעוּן בְּלׇובְשָׁהּ קִדּוּשׁ יָדַיִם וְרַגְלִים.
טוֹרַח עֲבוֹדוֹת הֶכְשֵׁרָן לְבַד בּוֹ.
טָלֶה קוֹרֵץ וּמִצְוָה מְרִיקָה. בְּאַחַת,
טָס וְהִקְטִיר בֵּין דָּם לַנֵּרוֹת,
טִיֵּיב חֲמִשָּׁה, הִפְסִיק, וְהִקְרִיב אַרְבָּעָה.

טִירַת רִבּוּעַ עָשׂוֹר בְּקִדּוּשׁ בָּא,
טוֹהַר-קָדוֹשׁ שָׁנָה פְּשׁוֹט פָּז,
טָבַל וְנִסְתַּפַּג, לָבַשׁ לָבָן וְקִדַּשׁ,
טִיֵּיל לְפָרוֹ, לְכַפָּרַת דְּבָרִים בַּעֲדֵהוּ.

טִירוֹנֵי כְהֻונָּה, כמכין עִם תַּלְמִיד,
טֻוכְּסוּ בְּשֵׁם, חֶבֶל חוּט מְשֻׁולָּשׁ –
טֻופְּלוּ לַשְּׂעִירִים, וְקַלְפֵּי חוֹל וְחִיסָּכוֹן
טְמוּמַת כִּוּוּן, מַחֲזֶקֶת אַךְ שְׁתָּיִם.
טָרַף בָּהּ וְהֶעֱלָה שְׁנֵי גוֹרָלוֹת
טוֹב טַעַם; שֶׁל זַיִת, אֱגוֹז וְאֶשְׁכְּרוֹעַ.
טַסֵּי זָהָב עֲשָׂאָן בֶּן גַּמְלָא,
טָבוּעַ גְּמוּלוֹ לְזֵכֶר בְּרָכָה וָשֶׁבַח.

חֲרוּטֵי מִכְתָּב: אֶחָד - כְּתָבוֹ: לַשֵּׁם.
חֻקַּת שֵׁנִי: כָּתוּב עָלָיו לַעֲזָאזֵל.
חֶבֶל שֶׁפֶר וְשִׂימַת אוֹת לְטוֹבָה,
חֵן בְּהַשְׂכֵּל - וְגוֹרַל שֵׁם מְיֻומָּן.
חָשׁוּב הַמַּיְמִין יַצְהִיל: 'בִּגְאוֹן-הַשֵּׁם,
חַזֵּק יְמִינְךָ אִישִׁי כֹּהֵן גָּדוֹל!'
חָלַף - 'שְׂמֹאל' מִשְׁנֶה הַמַּשְׂמְאִיל יַגִּיד.
חִוָּה 'לַיְיָ!' וְזוֹ אַחַת מֵעֶשֶׂר.
חֲבִילַת זְהוֹרִית קָשַׁר בְּרֹאשׁ הַמִּשְׁתַּלֵּחַ,
חֲבֵרוֹ, מִלְּהִתְעָרֵב, סִימַּן בֵּית הַשְּׁחִיטָה.
חַף חָב לְזַכּוֹת בְּמִדַּת הַדִּין,
חוֹזֵר לְפָרוֹ, עָלָיו וְכָל אֶחָיו:

חִטּוּי חַטָּאת שְׁחָטוֹ וְהִמְרִיס דָּם
חוּץ לַהֵיכָל, עַל הָרוֹבֶד הָרְבִיעִי.
חָטַף מַחְתָּה אֲדֻומַּת פַּרְוַיִם, חֲמִשִּׁים,
חָשׁ לַחִיצוֹן, לְצַד סָמוּךְ לִפְנִים,
חָתָה וְיָרַד וְהִנִּיחַ בְּרוֹבֶד עֲזָרָה.
חֲבֶרְתָּהּ הוֹצִיא גְּדוּשָׁה, וְכַף רֵקָנִית,
חָפַן וְנָתַן מְלֹא חָפְנָיו בַּכָּף –
חוֹבַת הַיּוֹם: דַּקָּה מִן הַדַּקָּה.

זִמֵּן מַחְתָּה לְיָמִין, וּשְׂמֹאל - בָּזֶךְ,
זֵרֵד מָתְנָיו כְּגִבּוֹר וְאִישׁ חָיִל:
זָע בַּהֵיכָל בֵּין שֻׁלְחָן לַכּוֹתֶל,
זוֹחֵל לַפָּרוֹכֶת, מְקוֹם פְּרִיפַת חִצוֹנָה,
זוֹ מִזֶּה וּמִזֶּה זוֹ מִתִּוּוּכוֹתָיו.
זָז וּמַצְפִּין, פּוֹרֵף עוֹד פְּנִימִית,
זזוג שְׂמֹאלוֹ עִם הַפָּרוֹכֶת וְהִדְרִים.
זַח וְהִגִּיעַ עַד מְקוֹם אֲרוֹן
זִיקַּת מַחְתָּה: בְּעַד אֲרוֹן דַּדָּיו,
זֵירוֹ כְּנִגְנַז - שֶׁן סֶלַע כַּבַּדִים.
זָכוּר לִצְבּוֹר קְטֹרֶת פְּנִימָה - חוּצָה,
זָהִיר הִשָּׁמֵר, הִכָּווֹת בְּמוֹ אֵשׁ.
זֵרוּז מַקֵּל עֲשַׁן עָנָן מִתַּמֵּר,
זוֹקֵף וְעוֹלֶה, מְמַשְׁמֵשׁ כְּתָלִים וְיוֹרֵד.
זוּן עֵינָיו כּוֹבֵשׁ, יָרֵא מֵהַבִּיט:
זְכִיּוֹת וּמִצְוֹת, חֲבִילַת תַּחְבּוּלוֹת עִמּוֹ.
זוֹרַז וְיָצָא דֶּרֶךְ בֵּית כְּנֵיסָתוֹ,
זוֹעֵק קְצָרָה עֵת מַרְפֵּא מִלְּהַבְעִית:
זוֹל וְשָׂבׇע מִשָּׁם חָפַר אוֹכֶל
זֻמַּן לַכֹּל, לְמֵרָחוֹק עֵינָיו יַבִּיטוּ.

וְדַם פַּר מְמָרֵס הִמְצִיא לוֹ,
וְשָׁנָה לַמָּקוֹם אֲשֶׁר עָמַד שָׁם,
וְעוֹמֵד וּמַזֶּה: וְלֹא מַטִּיף וְזוֹרֵק.
וְהִזָּה בְּאֶצְבַּע יְמָנִית מְיֻומֶּנֶת יָמִין,
וְאַחַת לְמַעְלָה וְשֶׁבַע לְמַטָּה סוֹפֵר,
וּמַצְדִּיד כְּמַצְלִיף וְלֹא כְּמִתְכַּוֵּן לְכַוֵּון,
וְאַחַת מוֹנֶה בְּכָל אַחַת וְאֶחָת.
וְגָמַר וְיָצָא וְהִנִּיחַ בְּכַן הַזָּהָב,
וְכֹה יוֹסִיף שְׂעִיר חַטַּאת עָם,
וּשְׁחָטוֹ וְחִטְּאוֹ כְּמִשְׁפָּט וּמִסְפַּר עָם.
וְהִשְׁלִים חוֹק וְלֹא הִפִּיל דָּבָר,
וְהִנִּיחַ בַּשֵּׁנִי. וְיַז דַּם פָּר,
וּמִחוּץ לַפָּרוֹכֶת כְּנֶגֶד אֲרוֹן קָם,
וּבְמִדָּה פְנִימִית - מִדָּה חִצוֹנָה לַפָּרוֹכֶת.
וְהִנִּיחַ, וְנָטַל, וְסִיֵּים שְׁנֵיהֶם שָׁוִים,
וְדָם בְּדָם עֵירָה, עֵירַב כְּאַחַת,
וְקִדֵּשׁ וְטִהַר בְּרָבוּעַ וְרוּם שָׁלֹשׁ,
וְהַנִּשְׁאָר בַּדָּם - לִיסוֹד מַעֲרָבִי הַחִצוּן.
וְכִלָּה מִכַּפֵּר, וּבָא בְּמַעֲמַד חָי,
וִידּוּי דְּבֵקִים לְאֵל חַי הִרְצָה.

הִטְעִינוֹ מַשּׂוֹי בוכר, קַלּוֹת וַחֲמוּרוֹת,
הַמִּדְבָּרָה שִׁלְּחוֹ בְּיַד עִתִּי לְצוּק,
הַר קָשֶׁה, גְּזֵירָה, לֹא נוֹשֶׁבֶת,
הַמֶּלֶךְ כְּדָתוֹ, כִּי כֵן יִסַּד.
הוּא אָמַר - וְשִׁלְטוֹן דְּבַר מֶלֶךְ.
הוּא צִוָּה, יָעַץ - וּמִי יָפֵר?
הָאָדוֹן יְיָ גְּזֵירָתוֹ מִי יְשִׁיבֶנָּה?
הִרְהוּר לְיַעֲקוֹב - רְשׁוּת לֹא נִתְּנָה.
הַכֹּל כְּשֵׁרִין - זָר, בְּטֻומְאָה וּבְשַׁבָּת;
הִידּוּר וּמַעֲלָה מַנְהִיגֵי סִלְסוּל קָבְעוּ.
הַבָּא, בְּיָדוֹ מְזֻומָּן וּמָסוּר לוֹ,
הוֹלֵךְ לְשַׁלְּחוֹ, וְיַקִּירֵי מְדִינָה יְלַוּוּהוּ,
הִדּוּרוֹ: מִדְבָּר, עַד בֵּית שִׁלּוּחוֹ,
הֲפוּגוֹת לוֹ מִשָּׁלֵם בְּסַךְ סֻכּוֹת.
הֻווְעַד לוֹ מִסֻּכָּהּ לְסֻכָּה מִסְעָדָיו:
הַפַּת בְּסַלּוֹ, נִסְמַךְ וְלֹא נִצְרַךְ.
הִגִּיעַ לוֹ מְקוֹם מִשְׁלָח, מִשְׁמַעְתּוֹ:
הֶחֱצָה זְהוֹרִית וַחֲצִי קָשַׁר בַּסֶּלַע,
הַחֲצִי עוֹד קָשַׁר בֵּין קַרְנָיו
הַבּוֹשׁ לְהָסִיר מִלְּהוֹסִיף כְּהִתְקִינוּ.

דּוֹחֵף שָׂעִיר, זוֹקְרוֹ בְּבַת רֹאשׁ,
דֶּרֶךְ יְרִידָה מִתְגַּלְגֵּל, מִתְגַּזֵּר וּמִתְגַּמֵּר.
דּוֹחוֹ כֵּן צְפוּנָה עִם צָרוּר!
דֻּויַּיר לָאַחֲרוֹנָה, שָׁם שָׁבַת תַּחְתָּיו.
דְּבִיר וְחוּצָה - פּוֹעַל הַיּוֹם מְלַאכְתּוֹ.
דָּרֹשׁ דָּרַשׁ פַּר וְשָׂעִיר וּקְרָעָם:
דָּמָם כְּהוּבָא - חֵלֶב הִמְגִּיס וְהִקְטִיר;
דֻּגְמַת קְלִיעָה הוֹצִיאָן לְבֵית שְׂרֵיפָה.
דִּירְכָאוֹת כְּהוֹדִיעוּ: 'הִגִּיעַ שָׂעִיר לַמִּדְבָּר!'-
דִּלֵּג וְקָרָא סֵדֶר יוֹם וְעִנְיָנָיו,
דִּבֶּר וְשִׁנֵּן כִּכְתָב וְעַל פֶּה,
דֵּי מַדָּיו בְּרָכוֹת - כְּסִלֵּק - הוֹדָה.
דָּהַר וְקִדַּשׁ וּפָשַׁט כְּלֵי בָד,
דִּלֵּג וְטָבַל, זָהָב עָט וְקִדַּשׁ,
דָּת צוֹרֶךְ עֲבוֹדָה שְׁלִישִׁית נָתַן:
דִּבֵּק כַּחוֹק עֲבוֹדַת יוֹם וּמוּסָף.

גֶּשֶׁת פְּנִימָה קִדַּשׁ פְּשׁוֹט זָהָב,
גֶּוְויוֹ טָבַל, וְלָבָן בְּלָבַשׁ קִדַּשׁ,
גָּבַר וְנִכְנַס וְהוֹצִיא כַּף וּמַחְתָּה.
גַּם קִדַּשׁ, וּפָשַׁט בָּד, וְהִצְנִיעַ.
גְּלַל תָּמִיד, קְטֹרֶת, וַהֲטָבַת נֵרוֹת
גְּלוֹם פָּז טָבַל עש וְקִדַּשׁ.
גִּבּוֹרֵי מַטָּה זִיֵּין, עַם קְדוֹשִׁים,
גָּמַר חָמֵשׁ וְקִדַּשׁ וּפָשַׁט קֹדֶשׁ.

בִּגְדֵי עַצְמוֹ הֵבִיאוּ לוֹ וְלָבָשׁ,
בְּכָבוֹד וּבְגוֹדֶל עַד בֵּיתוֹ יְלַוּוּהוּ,
בְּשָׁלוֹם כְּהִשְׁלִים מְלֶאכֶת מֶלֶךְ שָׁלֵם,
בְּלִי פֶגַע, פְּסוּל, וְאִנּוּי אָוֶן.
בְּיוֹם טוֹבָתוֹ - טוֹב-לְכָל הֵטִיב –
בְּטוֹב לֵב מֵטִיב יוֹם לְאוֹהֲבָיו.
'בָּרוּךְ הַמָּקוֹם בָּרוּךְ הוּא' יְדֻובָּר,
'בְּזֶרַע אַהֲרֹן בְּלֹא נִמְצָא פְּסוּל!'

אִישׁ אֱמוּנִים שֻׁולַּח לַסֶּלַע, מִדְבָּרָה,
אוֹמֶר יְשׁוּעוֹת לְהַר צִיּוֹן יְבַשֵּׂר:
אֲרוּכַת תְּעָלָה וּמְרַפֵּא רְטִיַּית שׁוֹלְחָיו,
אֶת נֶפֶשׁ יָשִׁיב, כְּקָרִים עַיֵפָה.
'אֱכוֹל בְּשִׂמְחָה' בַּת קוֹל תְּפוֹצֵץ
'אֱלֹהִים כְּבָר קִבֵּל וְרָצָה מַעֲשֶׂיךָ!'
אָז יְרַנְּנוּ וְיִשְׂמְחוּ וְיָשִׂישׂוּ חוֹסִים,
אוֹהֲבֵי שֵׁם יַעְלְצוּ, יַעְלְזוּ בְּכָבוֹד.״

״אוֹר יֶשַׁע מְאֻושָּׁרִים
שִׁמּוּר זֶה מֻוכְשָׁרִים
אֲהוֹדֶנּוּ בִּידִידָיו כְּשָׁרִים
שִׁיר הַשִּׁירִים.

אַיֶּילֶת אִוּוּי תְּשׁוּקוֹת,
לִיחְלוּחַ עִיּוּף שׁוֹקֵקוֹת,
אֲסָמַי שׂבַע לְהַשְׁקוֹת –
יִשָּׁקֵנִי מִנְּשִׁיקוֹת.

בְּרוּכֵי מֵעֲלָמוֹת מִשְׁמַנֶּיךָ!
מוֹשְׁלֵי גַנְזַכֵּי מִכְמַנֶּיךָ
בִּשַּׂמְתָּם תַּמְרוּק סַמְמָנֶיךָ –
לְרֵיחַ שְׁמָנֶיךָ.

בְּנֵי בֵיתְךָ וְחוֹרֶיךָ
הִנָּם! חַזֵּר סְחָרֶיךָ!
בֵּית מִדִּין צְחֹרֶיךָ
מָשְׁכֵנִי אַחֲרֶיךָ.

גֵּאֶה ורוב עֲנָוָה –
הוֹדִי מִבָּנוֹת נִוָּה:
גַּם כִּי דָּוָה
שְׁחוֹרָה אֲנִי וְנָאוָה!

גַּחַל כִּי מְחֻורְחֶרֶת
קֶרֶב רַגָּז, סְחַרְחֶרֶת,
גְּחֹן קְדוֹרַנִּית כַּחֶרֶת –
אַל תִּרְאֻנִי שֶׁאֲנִי שְׁחַרְחֹרֶת.

דּוֹלִי וּמִכְּמִדְּלִי בְּדָלִי,
טוֹטֶפֶת כְּלִילוֹ גְּדָלִי;
דִּגְלִי, עוֹז-מִגְדָּלִי,
הַגִּידָה לִי.

דָּרְבוֹנִי, מַלְמַד-מַרְדְּעִי,
נָתוּן רוֹעֶה דַּרְדְּעִי,
דָּן וְהוֹכִיחַ פָּדְעִי –
אִם לֹא תֵדְעִי.

הִדְרִיךְ סוּס לִמְנוּסָתִי
בַּיָּם בִּקַּע לִבְסִיסָתִי,
הֱמִירוֹ, וּמִלֵּא בְּפִיסָּתִי –
לְסֻסָתִי.

הִיתְרַנִי 'מִילַּיי חַיַּיִךְ,
רְטִיַּית נוֹמֵי מָחֲיַיִךְ,
הַזֶּה וְהַבָּא לִתְחִיַּיִךְ,
נָאווּ לְחָיַיִךְ.'

וְחַנַּנִי חֵן מֻוצְהָב
בְּנִיצּוּל צוּל רָהַב,
וְכַבִּיר הִצְבִּיר יָהָב,
תּוֹרֵי זָהָב.

וְלִבִּי גַּס בּוֹ.
יִידַּע סִיג סֻובּוֹ,
וְסָר בְּאוֹכֶל עֶשְׂבּוֹ
עַד שֶׁהַמֶּלֶךְ בִּמְסִבּוֹ.

זִימֵּן זְמַן מִזְמוֹר
ייחוד עָרִיצִים לִזְמוֹר:
זִיבְּלַנִי בַּצַּר לִכְמוֹר,
צְרוֹר הַמֹּר.

זִילְזוּל כֵּירוּם כֶּפֶר,
הִזִּיל כַּרְכּוֹם חֵפֶר.
זִיֵּיף חוֹב מִסֵּפֶר –
אֶשְׁכֹּל הַכֹּפֶר.

חָשַׁק חִיבַּת רֵיעוּתִי
וְחָפֵץ חֶלְבֵי רַעְוָותִי.
חִיוָּה לְחֻופַּת יְרִיעָתִי:
הִנָּךְ יָפָה רַעְיָתִי!

חָשׁ חַפְּשִׁי מִדּוּדִי,
דִּיץ בְּרִיוּוּי דַּדִּי
חִינַּנְתִּיו: חֲתָנִי, יְדִידִי,
הִנְּךָ יָפֶה דוֹדִי!

טִילְטֵל פְּנִיקְטֵי נִתְרָזִים,
הָרִים עוֹקֵר, וּתְרָזִים,
טָס מַטַּע מְזוֹרָזִים:
קֹרוֹת בָּתֵּינוּ אֲרָזִים.

טִירַת בֵּיתוֹ נֶאֱצֶלֶת
יִתְּנֵנִי רוֹן מַצְהֶלֶת
טַכְסִיס חֲתֻונָּה מְצַלְצֶלֶת –
אֲנִי חֲבַצֶּלֶת.

יַעֲלוּ רוֹם מְתוּחִים
חֲרוּשִׁים, דּוּשִׁים, תִּיחוֹחִים,
יָאִים, עֲרֵיבֵי נִיחוֹחִים –
כְּשׁוֹשַׁנָּה בֵּין הַחוֹחִים.

יָצִיץ - וְלֹא לְמִצְעָר –
יֶתֶר שְׁאָר מִזְעָר!
יִפְרַח ליווי מַעַר,
כְּתַפּוּחַ בַּעֲצֵי הַיַּעַר.

כִּתְרִי מֵהַכְשֵׁר זָיִין,
הִרְבַּנִי מְלֹא עָיִן!
כְּתָב רְאָיָיה - וּמִנָּיִין?
הֱבִיאַנִי אֶל בֵּית הַיָּיִן!

כִּלְכֵּל עֶדְנִי מִתְּשִׁישׁוֹת;
יִסְּרַנִי קָשׁוֹת מְקֻושָּׁשׁוֹת,
כִּיוַּנְתִּי חוֹלַת חֲשָׁשׁוֹת –
סַמְּכוּנִי בָּאֲשִׁישׁוֹת.

לֹא-דֶרֶךְ כְּהִפְלִישִׁי,
גַּם צִיָּה כְּהִפְרִישִׁי –
לְמָסָךְ וּפִינּוּי טַרְשִׁי
שְׂמֹאלוֹ תַּחַת לְרֹאשִׁי.

לֵב אֶחָד וּשְׁכֶם
דָּאִים כָּעָב לְמׇושְׁכֶם!
לַאֲסִירִים יַחְפֹּץ כְּסֶכֶם
הִשְׁבַּעְתִּי אֶתְכֶם!

מִבּוֹר תַּחְתִּית רִידּוּדִי
לָחַץ עַקְרַבֵּי גְּידּוּדִי,
מִהַר, קִידָּר מְדַדִּי:
קוֹל דּוֹדִי!

מִבֵּית חוֹמֶר קִצְּבִי
בֵּית חַיִּים, נְצִיבִי.
מְצַפִּי, מְצִיפִי, מַצְבִּי,
דּוֹמֶה דוֹדִי לִצְבִי!

נַחֲלָה שָׁפְרָה לִּי
תִּקְוַת שִׁילּוּשׁ גּוֹרָלִי.
נְעִימוֹת בְּמַר לִי
עָנָה דוֹדִי וְאָמַר לִי.

נָם: תַּתִּי סְבָר,
וְרַק אֵין דָּבָר,
נוּסוּ, נוּדוּ כְּבָר,
כִּי הִנֵּה הַסְּתָו עָבַר.

סַרְסוּר זֵרוּז מֶרֶץ
רֹאשׁ פֵּירוּז הֵירֶץ,
סְלוֹל שָׁלוֹם וָתֶרֶץ
הַנִּצָּנִים נִרְאוּ בָאָרֶץ!

סַפֵּק עַתִּיק מְפִיגֶיהָ,
הָפֵק מַמְתִּיק פִּיגֶיהָ,
סַלֵּת סוֹלֶת סְפוּגֶיהָ –
הַתְּאֵנָה חָנְטָה פַגֶּיהָ.

עֲקַלָּתוֹן נוֹשֵׁף קֻולַּע
וּפִיתוֹס, מְכַשֵּׁף, נִצְלַע,
עָרֵב שִׁילְשׁוּל בְּהַקְלַע,
יוֹנָתִי בְּחַגְוֵי הַסֵּלַע.

עֲרוֹדִים עִם עִילִּים
מָדוּ, זָדוּ שְׁעוּלִים.
עִנְיָין לו מַעֲלִים
אֶחֱזוּ לָנוּ שׁוּעָלִים.

פְּתִיגִיל וְעָגִיל בְּדָלִי
עִינּוּק, פִּינּוּק קְדָלִי;
פָּנִים בְּפָנִים הִשְׁדִּילִי:
דּוֹדִי לִי.

פּוֹעַל בָּם תָּפוּחַ,
שָׂכָר לֹא קָפוּחַ
פקוד דָּר טָפוּחַ –
עַד שֶׁיָּפוּחַ.

צִימְצֵם שִׁכְנוֹ מֶרְכָּבִי
יוֹם חִנּוּךְ כַּרְכֻּובִּי,
צִפִּיתִי: פֹּה עִכּוּבִי,
עַל מִשְׁכָּבִי.

צְבִי קֹדֶשׁ מִינָּה
מְקוֹם מִבְחָר מָנָה:
צִיּוֹן, רֹאשׁ אֲמָנָה,
אָקוּמָה נָּא.

קָרַצְתִּי, דַּצְתִּי בְּאוֹמְרִים:
טַפְתִּי, נַפְתִּי בְּמוֹרִים,
קִילְקַלְתִּי שְׁמָנִים שְׁמָרִים –
מְצָאוּנִי הַשֹּׁמְרִים.

קִינְיָין כִּי שָׁבַרְתִּי
וְטוֹב לֹא סִבַּרְתִּי –
קׇודְשׁוֹ לוּלֵא חֻובַּרְתִּי,
כִּמְעַט שֶׁעָבַרְתִּי!

רָשַׁם אִסָּר: בְּצִבְאֹתֵיכֶם,
בְּזִיעֲזוּעַ חִיל אַיָּלוֹתֵכֶם,
רִיחוּק, דִּחוּק אוֹתְכֶם
הִשְׁבַּעְתִּי אֶתְכֶם!

רֵישׁ גְּלֵי כְּהֶעֱלָה,
יוֹנָה תַמָּה עוֹלָה –
רָגְשׁוּ בְּנֵי עַוְולָה:
מִי זֹאת עֹלָה?

שֵׁירוּת שַׁפְרִיר שִׁטָּתוֹ,
מַטּוֹת, וְרִיבּוֹ חִיטָּתוֹ,
שַׁעַר 'צֵא' וְשִׁטָּתוֹ.
הִנֵּה מִטָּתוֹ!

שִׁירַת עִרְבּוּב מִלְּעָרֵב –
אֻמָּנוּת אַבּוּב עָרֵב,
שִׁירַת נִבּוּב הֶרֶב,
כֻּלָּם אֲחֻזֵי חֶרֶב.

תָּאַב, פָּסַק לוֹ,
מְלוֹת כַּנֵּי טְרִסְקְלוֹ;
תַּטְלִיל חָסָה לוֹ
אַפִּרְיוֹן עָשָׂה לוֹ.

תֵּיאַר פָּנִים מֵהִכָּסֵף,
נוֹעֳמוֹ יְדִידוּת כּוֹסֶף –
תַּבְנִית אוֹת יוֹסֵף,
עַמּוּדָיו עָשָׂה כֶסֶף.״

״צְאֶינָה וּרְאֶינָה
מַשְׂכִּיל שִׁיר יְדִידִים,
רִנַּת חֲתֻונַּת דּוֹדִים.

הִנָּךְ יָפָה
דָּת, דִּין וּפִקּוּדִים
כָּפַל יָחִיד לִפְקוּדִים.

שִׁנַּיִךְ
הַרְחֵק כִּיאוּר חִמּוּדִים
קִשּׁוּט לְשׁוֹן לִמּוּדִים.

כְּחוּט הַשָּׁנִי
טַעַם זִיֵּן יְלָדִים,
נֶגְדּוֹ שֻׁולַּם לַיְלָדִים.

כְּמִגְדַּל דָּוִיד
יִקְרַת פִּנַּת יְסוֹדִים
גְּבוּל הוֹרָיוֹת יְסוֹדִים.

שְׁנֵי שָׁדַיִךְ
דִּיבּוּק אַחִים חֲסִידִים
לְרוֹעִים, תַּיָּירִים חֲסוּדִים.

עַד שֶׁיָּפוּחַ הַיּוֹם
יוֹם צְדָקָה לַחֲרֵדִים,
חִבּוּל זֵדִים זְרָדִים.

כֻּלָּךְ יָפָה
בְּלִי מוּם, מְכֻובָּדִים
תָּם וְחֶלְקוֹ, מְזֻובָּדִים.

אִתִּי מִלְּבָנוֹן כַּלָּה
וִיעוּד שְׁכִינָה בְּשִׁיעְבּוּדִים
רָצוּף בְּבוֹא מֵאִיבְּוּדִים.

לִבַּבְתִּנִי אֲחֹתִי כַלָּה
הַבַּת בַּת נְגִידִים,
אַהֲבַת, חִבַּת אֲגוּדִים.

מַה יָּפוּ דֹדַיִךְ אֲחֹתִי כַלָּה
מְנוּחָה וּשְׁאָר וִיעוּדִים,
וּשְׁמָנַיךְ - קִרְיַת מוֹעֲדִים.

נֹפֶת תִּטֹפְנָה שִׂפְתוֹתַיִךְ כַּלָּה
אַהֲבַת מְשִׁיחִים עוֹמְדִים
מִבְּלִי קִנְאָה מְצֻומָּדִים.

גַּן נָעוּל
נָקָם - וְאֵין מַגִּידִים,
סָתוּם לְתִּלְבּוֹשֶׁת הַבְּגָדִים.

שְׁלָחַיִךְ פַּרְדֵּס
שְׁפַר, נָאִים וַהֲדוּרִים
לְטַכְסִיס מְלָכִים אֲדוּרִים.

נֵרֵדּ וְכַרְכֹּם
מִזֶּה וּמִזֶּה סְדוּרִים,
הַחוּט הַמְשֻׁלָּשׁ, גְּדוּרִים.

מַעְיַן גַּנִּים
הֲדָחַת קַלִּים וַחֲמוּרִים,
טְבִילַת נְקִיּוֹת שְׁמוּרִים.

עוּרִי צָפוֹן וּבוֹאִי תֵימָן
יַחַד לְכַנֵּס נִסְעָרִים
גָּלוּת כָּל שְׁעָרִים.

בָּאתִי לְגַנִּי אֲחֹתִי כַלָּה
דֻּוגְמַת מִלּוּאִים גְּמוּרִים,
לְקַבֵּל אֵיבָרִים וֶאֱימוּרִים.

אֲנִי יְשֵׁנָה וְלִבִּי עֵר
בִּיאַת עִיר מִשְּׂעִירִים,
תּוֹחֶלֶת יִסְחָבוּם צְעִירִים.

פָּשַׁטְתִּי אֶת כֻּתׇּנְתִּי
רַבָּתִי וּשְׁנֵי כְתָרִים,
הָאוּרִים וְתֻומֵּי סְתָרִים.

דּוֹדִי שָׁלַח יָדוֹ
אָז לְסִעוֹד הוֹרִים,
מְזֻומָּן עוֹד לַנִּמְהָרִים.

קַמְתִּי
נִצַּבְתִּי בְּחֶמֶד הָרִים
וְדַצְתִּי, אַצְתִּי בַּזְּהָרִים.

פָּתַחְתִּי
מֵאֵלַי, בְּנִטּוּף מוֹרִים:
נַעֲשֶׂה וְנִשְׁמַע אֲמָרִים.

מְצָאֻנִי
לִיגְיוֹנוֹת סְבָבוּנִי בַּנְּהָרִים
עוֹמֵד בַּהֲדַסִּים בֶּהָרִים.

הִשְׁבַּעְתִּי
דְּרוֹשׁ דּוֹד מֵישָׁרִים,
סוּר מֵרֻוכְסֵי קְשָׁרִים.

מַה דּוֹדֵךְ מִדּוֹד
לֶעָתִיד שְׁאוֹל נִפְשָׁרִים
הֵן בָּאֻומּוֹת הַכְּשֵׁרִים.

מַה דּוֹדֵךְ מִדּוֹד
נוֹאֲמִים בְּזֶה הַקּוֹשְׁרִים
נֶגֶד נֶהֱרָגִים כַּשְּׁוָורִים,
נֵזֶר יִיחוּדוֹ קוֹשְׁרִים,
נֶצַח מַלְכוּתוֹ שָׁרִים,

דּוֹדִי
נׇועֳמוֹ עַמּוֹ מְיַשְּׁרִים,
נִכְתָּב וְנִקְרָא בַּשִּׁירִים,
צִדְקוֹתָיו לָרֵיקִים מְחַזְּרִים,
צִיבְיוֹנוֹ עֲשׂוֹת מִתְאַזְּרִים,
צְפוּתֵינוּ צְפוֹת מְשַׁחֲרִים.
צִיּוּן עַמּוּד שְׁחָרִים
צוּר קוֹנֵינוּ מַכְתִּירִים,
צׇורְכֵּנוּ בְּפֶלֶל מַעְתִּירִים.
חָזֹה נֶחֱזֶה בְּבֵיאוּרִים
חׇשְׁכֵּינוּ לָעַד לְאוֹרִים.
חַי, זַכֵּינוּ כְּמִתְפָּאֲרִים,
קַיֵּים לְדוֹר דּוֹרִים –
הַמֵּאִיר לָאָרֶץ וְלַדָּרִים!״

״רֹאשׁוֹ כֶּתֶם פָּז
תָּג, עִיטּוּר, כֶּתֶר,
תַּחַן יוּחַן, עֶתֶר,
שִׁכְנוֹ בְּיוֹשֵׁב סֵתֶר –
שָׁלִיחַ מְפוֹרָשׁ פּוֹתֵר;
רָם וְנִישָּׂא בְּיוֹתֵר,
רַב-בַּנַּאי וְסוֹתֵר,
קוֹרְאָיו פֶּתַח חוֹתֵר,
קַבֵּל כְּנִיתּוּחַ בֶּתֶר.

עֵינָיו כְּיוֹנִים
צַד רִיבּוּעַ דְּפָנִים
צוֹפוֹת צֹפִים צְפוּנִים,
פְּתוּחוֹת שִׂיחַ סְפוּנִים,
פּוֹנוֹת לְאֵלָיו פּוֹנִים,
עָתִיד וְגָנוּז פְּנִינִים
עָרוּךְ נֶגְדּוֹ מִלְּפָנִים,
סִכְיוֹנוֹ אֵין לְפָנִים,
שְׂרָפִים וְחַיּוֹת וְאוֹפַנִּים.

לְחָיָו
נִיטּוּף מוֹר לְקָחִים,
נְעִימִים אֲמָרִים מַלְקוֹחִים,
מִגְדְּלוֹת מֶרְקַח רִיקּוּחִים,
מַתָּן לְנֶפֶשׁ פִּיקּוּחִים,
לוּחוֹת חֵירוּת מְשִׂיחִים,
לֵב עָלֵב מְשַׂמְּחִים,
כְּאֵב מַמְרִיחִים וּמוֹשְׁחִים
כַּעֲרוּגַת עֲרֵיבִים שִׂיחִים.

יָדָיו
יְדֵי אָדָם פְּשׁוּטוֹת,
יְצוּרָיו קַבֵּל מִלִּסְטוֹת,
טוֹעִים בַּדֶּרֶךְ קוֹשְׁטוֹת,
טֶרֶף לַכֹּל מוֹשִׁיטוֹת,
חֲזָקוֹת - וּמַחֲזִיקוֹת מָטוֹת,
חִישׁ מִדֶּחִי מְלַקְּטוֹת,
זְכוּת מְגַלְגְּלוֹת וּמַטּוֹת –
זַעַם הַדִּין מִלִּנְטוֹת.

שׁוֹקָיו
וַועַד כְּתָלֵינוּ מְצָרֵף
וְתַחַת רַגְלָיו שְׁרַפְרֵף,
הֲדוֹמוֹ מַרְעִיד וּמְרַפְרֵף,
הַיָּם מַלְחִיךְ וּמְשָׂרֵף.

חִכּוֹ מַמְתַקִּים
דָּת מְשָׁלִים עַתִּיקִים,
גּוֹזֵר אוֹמֶר וּמֵקִים,
בִּדְבָרוֹ שִׁיפֵּר שְׁחָקִים,
אָמַר - וַיִּקְרָא אֲרָקִים
שִׁבְחוֹ לִגְיוֹנָיו מְפִיקִים,
הֲמֻולַּת-הָמוֹן, כָּאֲפִיקִים,
חָלִים, זָעִים, קְרִיאַת שִׁילּוּשׁ-קְדֻושָּׁה מַסְפִּיקִים.״

״אֲהֵבוּךָ-נֶפֶשׁ לְהָדֵךְ,
מוֹנִים בּוֹטֵחַ וְשׁוֹדֵךְ:
אָנָא הָלַךְ דּוֹדֵךְ?

בְּשׁוּב נוּחֲךָ גְּנוּנוֹ –
רְבָבָה תְּרַנֵּן בְּגִינּוֹ:
דּוֹדִי יָרַד לְגַנּוֹ!

גָּשִׁים: 'נִבְנֶה!' כְּהַחְדִּילִי
דּוֹבְבִים 'נֵלְכָה!' לְהַבְדִּילִי
- אֲנִי לְדוֹדִי - וְדוֹדִי לִי.

דִּימַּנִי נָאוָה וּתְרוּצָה,
כִּינַּנִי: בְּלֹא פִירְצָה,
יָפָה אַתְּ רַעְיָתִי כְּתִרְצָה.

הִזְהִירַנִי: 'מִמַּעַשׂ מְעַנַּיִיךָ
הַתְעֵב גֵּיא כְּנַעֲנַיִיךְ
הָסֵבִּי עֵינַיךְ!'

וְעַמֵּךְ כֻּולָם הִילּוּלִים,
קוֹדֶשׁ וְלֹא חֻולִּים,
שִׁנַּיִךְ כְּעֵדֶר הָרְחֵלִים.

זְבָדַנִי זְבוּל אַרְמוֹן,
טַיֵּיב יוֹשְׁבֵי אָמוֹן
כְּפֶלַח הָרִמּוֹן.

חַכְמֵי מַלְאֲכוֹת מְלָאכוֹת
נֶצַח עוֹלָם הֲלִיכוֹת –
שִׁשִּׁים הֵמָּה מְלָכוֹת.

טוֹעֲנִים אֲסֻופּוֹת גַּנָּתִי,
יַחַד נוֹטְעִים כַּנָּתִי,
אַחַת הִיא יוֹנָתִי.

יׇושְׁרָם יָלִיץ הִיתָּקְפָה
גַּבְהוּת קֶרֶן זְקוּפָה –
מִי זֹאת הַנִּשְׁקָפָה.

כַּעַס וּשְׂחוֹק יָגוֹז,
דַּלַּת רֹאשׁ מִלֵּיגוֹז
אֶל גִּנַּת אֱגוֹז.

לְגוֹלֵל אֶבֶן צִימְּתַתְנִי,
בְּאֵר חַיַּי צִיְּימַתְנִי –
לֹא יָדַעְתִּי נַפְשִׁי שָׂמַתְנִי.

מָתַי לְחָצִיר נַעֲמִית
תּוֹעָה וּמַתְעָה לְהַעֲמִית
שׁוּבִי שׁוּבִי הַשּׁוּלַמִּית?!

נָאִים הִתְנַדֵּב, מְעֻולִּים,
רְאוֹת וּלְקַלֵּס מוֹעֲלִים –
מַה יָּפוּ פְעָמַיִךְ בַּנְּעָלִים.

סֵדֶר עֲבוֹדָה מְהֻוגָּן
הָשֵׁב בְּשִׁיתוֹ נֻוגַּן
שָׁרְרֵךְ אַגַּן.

עֲרוֹךְ אֵלִים וּפָרִים
כְּאָז, בְּיַד סוֹפְרִים,
שְׁנֵי שָׁדַיִךְ כִּשְׁנֵי עֳפָרִים.

פֶּתַח שִׁיר מִלִּידַּל
הֲדַר זוֹקֶן מְגֻודָּל –
צַוָּארֵךְ כְּמִגְדַּל.

צַו חֶסֶד כְּגוֹמֵל
וְנִישָּׂא כָּאָמוּר, חוֹמֵל,
רֹאשֵׁךְ עָלַיִךְ כַּכַּרְמֶל.

קֶלֶס - תְּמוּר: 'אָשַׁמְתְּ
גְּנַאי זִכְרוֹנֵךְ שַׂמְתְּ' –
מַה יָּפִית וּמַה נָּעַמְתְּ.

רִיב לְטוֹב יוּמָר,
וּלְשִׁפְלוּת עֵמֶק יְזֻומָּר:
זֹאת קוֹמָתֵךְ דָּמְתָה לְתָמָר.

שַׁתֵּף בְּכָל מִשְׁמָר
כֻּולָּם צַדִּיקִים בְּמַאֲמָר
אָמַרְתִּי אֶעֱלֶה בְתָמָר.

תַּעַן דָּבָר טוֹב
שׁוּרָה אַחַת לַחְטוֹב –
וְחִכֵּךְ כְּיֵין הַטּוֹב.

שְׁחוֹרָה וְנָאוָה תְּשׁוּקָתוֹ,
לֹא מֵרוֹב חֲשׁוּקָתוֹ –
אֲנִי לְדוֹדִי וְעָלַי תְּשׁוּקָתוֹ.

מֵחֲפוּרִים שׁוּר סְפוּרִים,
הִסְתַּכֵּל בָּךְ כּוֹפְרִים,
לְכָה דוֹדִי נֵצֵא הַשָּׂדֶה, נָלִינָה בַּכְּפָרִים.

בִּינָה רִיצּוּי תּוֹרְמִים
בִּסּוּס עוֹלָם גּוֹרְמִים,
נַשְׁכִּימָה לַכְּרָמִים.

יֻותַּן קוֹל צוֹרֵחַ
הַטּוֹבוֹת וְהָרָעוֹת הַפְרֵחַ –
הַדּוּדָאִים נָתְנוּ רֵיחַ.

דּוֹדִי נוֹחַ לִי,
הַמְנַחֲמִי וּמוֹחֵל לִי,
מִי יִתֶּנְךָ כְּאָח לִי.

חֲזוֹת כְּאָז צְבָאֲךָ,
זוֹהַר שְׁכִינַת מַרְאֲךָ,
אֶנְהָגְךָ אֲבִיאֲךָ.

קְרוֹבְךָ עַם דַּבְּקֵנִי,
וֶוסֶת שִׁבְחֲךָ סַפְּקֵנִי,
שְׂמֹאלוֹ תַּחַת רֹאשִׁי וִימִינוֹ תְּחַבְּקֵנִי.

אַחֲרִית, לְטַהֵר שׁוּלָיִם,
מַהֵר לְשַׁלֵּשׁ בְּכִפְלָיִם –
הִשְׁבַּעְתִּי אֶתְכֶם בְּנוֹת יְרוּשָׁלִָם.

צִיּוּן רְשָׁפִים מְלֻובֶּיךָ
'מִי זֹאת' כְּלַבְּבֶיךָ,
שִׂימֵנִי כַחוֹתָם עַל לִבֶּךָ.

אוֹתוֹת רְאוֹת מְרֻובִּים,
דֶּרֶךְ לַעֲבוֹר מְסֻובִּים
מַיִם רַבִּים.

בִּיטָה, עֲנִיָּיה לְנַחֵמָה,
תַּעֲרַת אָחוֹת אִם חוֹמָה,
אֲנִי חוֹמָה.

וִיהוּדָה כֶּרֶם גְּפָנָיי,
רַגֵּל הַהֵיכָל לִפְנָיי,
כַּרְמִי שֶׁלִּי לְפָנָי.

אֶלֶף הַמָּגֵן מָגִינִּים,
אֶדֶר שָׂרִים וּסְגָנִים –
הַיּוֹשֶׁבֶת בַּגַּנִּים.

מִסִּרְיוּת מַסְרִיחַ רְדִידִי,
נֵרְדְּ וּוֶרֶד יְדִידִי
בְּרַח דּוֹדִי,

וּדְמֵה לְךָ – בְּטִיּוּלִים
אַמֵּץ, כְּאָז, חֲיָילִים –
לִצְבִי אוֹ לְעֹפֶר הָאַיָּלִים.

מִשְׁקָל בְּמִשְׁקָל - סַמִּים,
נִיחוֹחַ מוֹר - שָׂמִים,
עַל הָרֵי בְשָׂמִים.״

״עַל הָרֵי בְשָׂמִים סוֹב, דְּמֵה לְךָ דוֹדִי,
מַשְׁגִּיחַ, מֵצִיץ מֵחֲרַכֵּי בֵית וַועֲדִי,
'קוּמִי לָךְ רַעְיָיתִי' דְּפוֹק לְעוֹדְדִי,
כִּי דַלּוֹתִי מְאֹד מַמְתֶּנֶת מוֹעֲדִי
יוֹשֶׁבֶת עֲגוּמָה וַעֲגוּנָה וְאֵין מְוַועֲדִי –
וְאַתָּה יְיָ מָגֵן בַּעֲדִי.

אֶל גִּבְעַת הָעֲרָלוֹת, מוֹר וּלְבוֹנָה וַאֲהָלוֹת,
הַבֵּט מִמַּעֲלוֹת, כִּי מַכּוֹת נַחְלוֹת,
קוּמָה, קַדְּמָה חוֹפֵשׂ עוֹלוֹת, עֲלִילוֹת,
טֶנֶף וּגְעִילוֹת יְחַלֵּל קוֹדֶשׁ, מְעִילוֹת,
נָא, שְׁבוֹר זְרוֹעוֹ מִלְּהַעֲלוֹת תְּעָלוֹת,
וְעַם קְהִילּוֹת, בְּמַקְהֵלוֹת, לְהוֹדוֹת הַלֵּלוֹת.

עַל הָרֵי בָתֶר, עַל אַחַד הֶהָרִים,
יֵרָאֶה לִבְחִירִים גְּמוּל פּוֹעַל הוֹרִים,
דֻּובִּים וּנְמֵרִים, אֲרָיוֹת וַחֲזִירִים,
פָּרִים אַבִּירִים - לְפַסֵּג גִּזְרֵי גְזָרִים,
כִּיקַר כָּרִים, כָּלִים, גְּמוּרִים,
תּוֹר וְגוֹזָל שְׁלֵימִים וְלֹא חֲסֵירִים,
שְׁמוּרִים בְּלֵיל שִׁימּוּרִים הַלֵּל גּוֹמְרִים,
כְּעוֹבְרִים שָׁרִים, זַמָּרִים, אָשִׁירָה לַייָ אוֹמְרִים!״

״בְּרַח דּוֹדִי עַד שֶׁתֶּחְפָּץ אַהֲבַת כְּלוּלֵינוּ,
שׁוּב לְרַחֵם, כִּי כִילּוּנוּ
מַלְכֵי רִשְׁעָה, שׁוֹבֵינוּ, תּוֹלָלֵינוּ.
הֲרוֹס וְקַעֲקַע בֵּיצָתָם מִתִּילֵּינוּ,
הָקֵם טוּרְךָ, נַגֵּן שְׁתִילֵינוּ:
הִנֵּה זֶה עוֹמֵד אַחַר כָּתְלֵנוּ!

בְּרַח דּוֹדִי עַד שֶׁיָּפוּחַ קֵץ מַחֲזֶה,
חִישׁ, וְנַסּוּ הַצְּלָלִים מִזֶּה,
יָרוּם וְנִישָּׂא וְגָבַהּ נִבְזֶה,
יַשְׂכִּיל וְיוֹכִיחַ וְגוֹיִם רַבִּים יַזֶּה,
חֲשׂוֹף זְרוֹעֲךָ קְרוֹא כָזֶה:
קוֹל דּוֹדִי הִנֵּה זֶה!

בְּרַח דּוֹדִי וּדְמֵה לְךָ לִצְבִי
יִגָּל יִגָּשׁ קֵץ קִצְבִּי
דַּלּוֹתִי מִשְּׁבִי לַעֲטֶרֶת צְבִי
תּוֹבְעִים תְּאֵיבִים הַר צְבִי
וְאֵין מֵבִיא וְנָבִיא וְלֹא תִשְׁבִּי,
מְשַׁוִּי מְשִׁיבִי ורִיבָה רִיבִי
הָסֵר חוֹבִי וּכְאֵיבֵי
וְיֵרֶא וְיֵבוֹשׁ אוֹיְבִי
וְאָשִׁיבָה חוֹרְפִי בְּנִיבִי
זֶה דוֹדִי גּוֹאֲלִי וּקְרוֹבִי
רֵיעִי וַאֲהוּבִי אֵל אֱלֹהֵי אָבִי.״

״אוֹמֶץ דָּר חֲזָקִים
אֲבָאֵר לְפִי סִפּוּקִים
בְּתוֹךְ וַועַד קְהַל שׁוֹמְרֵי חֻוקִּים.
בְּחִיל וָזִיעַ אַבִּיעַ מַעֲשָׂיו הַחֲקוּקִים.
גְּבוֹהִים רִיקַּע כְּטַס, וּמַרְאָם תָּמִיד נִרְתָּקִים,
גַּם יָדוֹ יָסְדָה יְסוֹדוֹת אֲרָקִים.
דִּילֵּק וְזָרַח אוֹרָה לִישָׁרִים וְצַדִּיקִים.
שְׁפַר אוֹמֶר יֻובְטָא
לָעֶלְיוֹנִים כַּדּוֹק נָטָה,
מֵכִין בְּחָכְמָה מָטָּה,
הָאוֹר כְּשַׂלְמָה עָטָה, קָדוֹשׁ.

הַמַּיִם עוֹלָם מִלִּמְחוֹת
הֵכִין רָקִיעַ מְתִיחוֹת
וּבַתָּוֶוךְ שָׂמָהוּ, חֶצְיָים בְּמֵי-רוּם, לְהַשְׁחוֹת.
רָגְנוּ, וְנִשְׁבַּע-לָמוֹ לְקַבֵּל מֵהֶם שְׁבָחוֹת.
זִימֵּן וְעָשָׂה מֵרוּחוֹת בְּלֶכְתָּם בַּמִּשְׁלָחוֹת
זֶרֶם מַיִם מִתְחַלְּפִים בְּשָׁרְתָם לֵאלֹהֵי הָרוּחוֹת.
חֲשַׁשׁ וָקֶדַח עָרַךְ לְמַכְעִיסָיו בִּפְלָחוֹת.
שִׁכֵּן בָּם אַכְזָר
לְיַסֵּר עוֹבֵד זָר,
מְיַחֲדָיו שָׁאַל - וְהֻוחְזָר,
הֱיוֹת חֶלְקוֹ גָזָר, קָדוֹשׁ.

טוֹב, בִּרְאוֹתוֹ נָסִים
טְבוּעַת אֶדֶן מְכַסִּים –
יִשְּׁוָום בְּגַעַר בַּמַעֲמַקִּים וְגִילְגְּלָם בִּמְכֻונָּסִים.
יָהֲרוּ עוֹלָם לְהָצִיפָה - וּגְבוּלָם חוֹל הֵשִׂים.
כִּילֵּל פְּנֵי הָאָרֶץ אִילָנוֹת וּפַרְדֵּסִים,
כִּינֵּן גַּן בְּעֵדֶן וּמַאֲכַל עֲסִיסִים
לְהַנְחִיל אוֹהֲבָיו יֵשׁ, לְתַלְמִידָיו, עֲמוּסִים.
שְׁאוֹנוֹ בְּחוֹל נִגְרָרִים,
לַנֶּאֱמָן עֵדוּת מְאַדְּרִים,
מִשְׁבְּרֵי יָם מַאְדִּירִים
הַמַּשְׁבִּיחַ שְׁאוֹן אַדִּירִים, קָדוֹשׁ.

מְאוֹרוֹת וְכוֹכְבֵי רִינּוּנִים,
מַזָּרוֹת וְנֶחָמַת בָּנִים
נָפַח בְּרוּחַ פִּיו וְהִכְנִיסָם בָּעֶלְיוֹנִים.
נִסְתָּר אֵין מֵחַמָּתוֹ וּמְחֻותָּל עֲנָנִים
סִידְּרָם בָּרָקִיעַ לְאוֹתוֹת וּלְמוֹעֲדִים וְשָׁנִים
סֶלָה לִמְשׁוֹל וּלְהָאִיר יָמִים וְאִישׁוֹנִים,
עֶלְיוֹן מַשְׁקִיעָם עֲבוּר עוֹשֵׂה כַּוָּונִים.
שְׁמוֹ לְהַלֵּל בְּרָאָם,
לְכֻלָּם בְּשֵׁם קְרָאָם,
מְסַדֵּר מוֹצָאָם וּמוֹבָאָם
הַמּוֹצִיא בְּמִסְפָּר צְבָאָם, קָדוֹשׁ.

פִּיקֵד מַיִם לְהַשְׁרִיצָה
פּוֹרְשֵׂי אֶבְרָה וְנוֹצָה,
צָפֵי אׇורְחוֹת יַמִּים לְהַרְבּוֹת וְהַפְרִיצָה;
צוּר צָר מֵהֶם - וּבָהֶם לֶעָתִיד רִיצָה,
קָרוּי זִיז בַּמְעוֹפְפִים לְעוֹנֶג מרצה,
קַשׁ חוֹשֵׁב תּוֹתָח מִתַּנִּינֵי שְׁחִיצָה
רֵיעִים יִכְרוּ עָלָיו וּבְנֵיהֶם לְהִתְיַיחֲצָה.
שְׁאוֹן גּוֹדֶל תַּנִּינִים
לְתוֹךְ פִּיו נְתוּנִים,
מְדוֹרוֹ בְּמַיִם תַּחְתּוֹנִים –
הֱכִינוֹ לְיוֹם חִיתּוּנִים, קָדוֹשׁ.

שַׁדַּי לְנֶפֶשׁ תַּאֲוָה
שׁוֹכֵן שְׁחָקִים בְּגַאֲוָה
תֵּבֵל הֵכִין בְּכוֹחוֹ לְהוֹצִיא צִיוָּה
שֶׁפַע נֶפֶשׁ חַיָּיה. מוּכָן לְיוֹם חֶדְוָה –
לְעֵת שׁוּב לְבִיצָּרוֹן אֲסִירֵי הַתִּקְוָה –
מְטִיל בַּרְזֶל גְּרָמָיו דִּימָּה וְהִתְוָוה
הוּא רֵאשִׁית דַּרְכֵי אֵל דְּרָכָיו הֻשְׁוָוה.
שְׁמָרוֹ לֶעָתִיד לָבֹא
לְיִשְׂרָאֵל עַם קְרוֹבוֹ,
מִרְעֵהוּ אֶלֶף יַרְבּוֹ –
הָעוֹשׂוֹ יַגֵּישׁ חַרְבּוֹ, קָדוֹשׁ.

הֵיכִין קָרוּץ מֵחוֹמֶר
קָדִימָה הִכְנִיסוֹ לְשַׁמֵּר,
טִיכּוּס חֻופַּת עֶשֶׂר טִילְּלוֹ מְגַלֵּה טֶמֶר,
נְאוּם צִיוּוּי הֵיפַר וְנִתְחַיַּיב לְהִיגָּמֵר.
חִינֵּן בַּעֲדוֹ שְׁבִיעִי לְמֵקִים אוֹמֶר,
זִימֵּן מִיָּד לָאֵל שִׁיר וָזֶמֶר,
קִדֵּשׁ לָכֵן שַׁבָּת, לְהִזָּכֵר וּלְהִשָּׁמֵר.
שׁוֹמֵר שַׁבָּת מֵחַלְּלוֹ
לַעֲוֹנוֹתָיו מָחוּל לוֹ.
מוּטָב וְנוֹחַ לוֹ
הָעוֹלָם הַבָּא לְהַנְחִילוֹ, קָדוֹשׁ.״

״אֵל נִישָּׂא
אֲרַנֵּן בְּהִתְעַלְּסָה.
אוֹמֶץ עָשָׂה –
אַגִּיד בִּכְנֵיסָה.

בְּסוֹד בָּרָה –
בְּיוֹשֶׁר אֶתְחַבְּרָה,
בְּרֵאשִׁית בָּרָא
בָּאֵר וּלְדַבְּרָה.

שִׁיתוּ לַעֲבֹד
מֶלֶךְ הַכָּבוֹד - קָדוֹשׁ,

גְּדׇול-פֶּלֶל
גְּבוּרוֹתָיו נְמַלֵּל;
גִּיהַּ וָלֵיל
גׇּודְלוֹ נְסַלֵּל.

דְּרוֹשׁ כְּיֵצַר
דֵּי מַעֲצָר –
דַּל יִנְצַר
דֵּיעַ מֵהִתְקַצַּר.

הָאֵל כְּחָשְׁבוֹ
הָעוֹלָם לְיַשְּׁבוֹ,
הִתְיָעֵץ בְּמַחֲשָׁבוֹ
הַמְשַׂחֶקֶת בְּמוֹשָׁבוֹ.

וּרְצוֹנָהּ נִישְׁוָה,
וְהִסְכִּימָה בְּמַחֲשָׁבָה,
וְלִבְרֹאות הֵשִׁיבָה
וְרוּחוֹ נָשְׁבָה.

זְמָמוֹ הִירְהַר –
זֵירוּז מִיהַר:
זִימֵּן מִזּוֹהַר
זְבוּלֵי טוֹהַר.

חֶלֶד לְהַאֲרִיק –
חִיוָּר הִזְרִיק.
חֵץ הִימְרִיק
חֵלֶף מַעֲרִיק.

טִילּוּל הִקְרִישׁ,
טִיעָתוֹ הִשְׁרִישׁ,
טְמוּנִים לְהַגְרִישׁ,
טְרוֹד וּלְהַפְרִישׁ.

יָצַר פּוֹצְחִים
יִחוּדוֹ לִנְצָחִים,
יִימֵּן קִידּוּחִים
יסוד נִדָּחִים.

כִּינֵּס בַּאֲגוּדוֹ
כַּבִּירֵי מִדּוּדוֹ,
כַּהֲזִידָם נֶגְדּוֹ –
כִּיבְּשָׁם בְּפַחְדּוֹ.

לְרֹאשׁ הַנֶּשִׁי
לַח, לְהַדְשִׁיא,
לְהַרְבּוֹת וּלְהַפְשִׁיא
לְמַזְרִיעַ וּמַדְשִׁיא.

מַעֲשִׂים זִימְּנָם –
מֵאוֹמֶר יִימְּנָם;
מְאוֹרוֹת וּמוֹנָם –
מֵרוּחוֹ מִינָּם.

נֶעֱלָץ צוֹעֵד,
נֶאֱמָן עֵד,
נוֹטְרִים מוֹעֵד,
נִתְקָפִים מֵהִימָּעֵד.

סִילֵּק מֵחֲבוּאִים
שָׂחִים וְדָאִים,
סִימֵּן לְנִבְרָאִים
סוּר מִטְּמֵאִים.

עָרוֹךְ בָּחַר
עֵינָיו כַּשַּׁחַר,
עָתִיד לִסְחַר
עֲמוּסֵי מִשְׁחָר.

פּוֹעֵל בְּחָכְמָה
פִּיקֵּד אֲדָמָה
פְּלוֹט בְּיִחוּמָהּ
פְּרָאִים וּבְהֵמָה.

צֶאֱלִים יְסֻכּוּהוּ
צָפָה לְעָרְכֵהוּ,
צָדוֹ לְנָכְסֵהוּ,
צְנוּעִים לְכוֹסֵיהוּ.

קָדוֹשׁ, מֵאַרְצוֹ
קִיצַּע קְרוּצוֹ,
קוֹמְמִיּוּת הִפְרִיצוֹ,
קִישְּׁטוֹ מִמַּעֲרָצוֹ,

רֵיק בְּשִׁיחַת –
רוּחוֹ נִפְחַת,
רֶשֶׁת שָׁחַת,
רֶדֶת לְתָחַת.

שָׁלֵם פָּעֳולוֹ,
שָׁרֵת בִּזְבוּלוֹ,
שְׁבִיעִי עִילּוֹ
שָׁמוֹר מֵחַלְּלוֹ.

תִּינּוּי שֶׁיפֶר
תְּמוּר שִׁיפֵּר –
תֵּת כּוֹפֶר
תַּעַל עוֹפֶר

שְׁמָרוֹ לְהַנְחִילוֹ
לְעַם נִיהֲלוֹ;
מָן בְּהַשְׁחִילוֹ
הַשְׁמִיעֵהוּ הִתְחִילוֹ.

הִתְעַנֵּג, לְהִתְנָאוֹת,
קְלוֹט יְצִיאוֹת,
טוֹעֵן מַשָּׂאוֹת
נוֹחַ מִלִּילְאוֹת.

שִׁיתוּ לַעֲבוֹד
מֶלֶךְ הַכָּבוֹד - קָדוֹשׁ.״

״אֲחַשְּׁבָה לָדַעַת עָמָל וּדְבָרִים יְגֵיעִים
לָמָּה כְּאֵיבִי נֶצַח, אָנוּשׁ מִמַּרְגּוֹעִים.
בְּרִית בָּאִים יִסּוּרִין לִנְגָעִים נוֹגְעִים,
וּרְשׁוּת רוֹפְאִים נִתְּנָה לִפְגָעִים פּוֹגְעִים.

גָּזוּ וְגַם גֻּונְּזוּ בּוֹשְׁלֵי הַסַּמְמָנִים,
סָרוּ אַף הוּסָרוּ סוֹרְחוּ הָאֻומָּנִים.
דַּרְכִּי כְּנִסְתְּרָה אֵלֵךְ אֵצֶל מְמֻונִּים
מְחַזֶּרֶת וְשׁוֹאֶלֶת מְצוֹא תְּשׁוּבַת מוֹנִים.

הַשַּׁאֲנַנִּים בֶּטַח וּבַעֲמַל אֱנוֹשׁ אֵינֵמוֹ
וְעִם אָדָם לֹא יְנֻוגְּעוּ אוֹנֵימוֹ,
וּבְרִיאִים כִּמְרִיאִים יֵצֵא מֵחֵלֶב עֵינֵימוֹ,
יַבִּיעוּ יְדַבְּרוּ רָע וּמַר עִנְיָינֵימוּ:

זֹאת הֵעֵיזָּה וּמִצְחָהּ לֹא בוֹיְשָׁן!
לָהּ לָדַעַת - רַק חַדַּת לִישָּׁן;
חִיתּוּל סַם הֲיוּשָׂם וְנֶגַע יְדֻושָּׁן –
זֶה כַּמָּה שָׁנִים יָשָׁן נוֹשָׁן!

טְעוֹתִי אַחֲרֵיהֶם חַיָּתָם לֹא פוֹסֶקֶת,
כְּפֶיחָם לַגֶּחָלִים וְעֵצִים לָאֵשׁ נִיסֶּקֶת:
'יְרִיעוֹתַיִךְ שַׁכְּנִי בִּשְׁכֻונָּה לֹא נִיזּוֹקֶת,
וְסִיפְרֵי רְפוּאָתֵךְ מַה תְּיַגְּעִי עוֹסֶקֶת?!

כַּלְּלִי יׇופַיֵיךְ בְּמַכָּה לֹא שׁוֹלֶטֶת!
בּוֹחַן אִם תְּבַקְשִׁי - הַשַּׁלְוָה מׇוחֳלֶטֶת!'
לַבַּת לִבִּי אַף כִּי קוֹלֶטֶת –
'קִרְבָתְךָ לִי טוֹב' פּוֹלֶטֶת, בּוֹלֶטֶת.

'מַה לָּכֶם - וְלִי מֵחֵן הַדְּגֵל,
אַף הַטִּנּוֹפֶת פַּגֵּל, לֹא הִתְגַּלְגֵּל!
נָעִים וָטוֹב מַה מָּתוֹק מִתְרַגֵּל
נִיצּוּי לְרִיצּוּי, אָב לְבֵן יְפַרְגֵּל!'

סְפִירַת הַחֶשְׁבּוֹנוֹת אַף לְפִי כָלְתָה
וַאֲרֻכַת בַּת-עַמִּי לֹא עָלָתָה –
עִינְיַין 'חַכֵּה לוֹ' מַצֶּבֶת גֻּולָּתָהּ
תִּקְוַת נֶפֶשׁ חִיבַּת דּוֹד כָּלְתָה.

פִּתְגָּם זֶה, תַּרְתִּי בְּלִבִּי, הִמְצִיעַ
הַמַּכּוֹת אֲשֶׁר הֻוכֵּיתִי, רַב הַמַּרְצֵעַ.
צוּר - מִידּוֹת שׁוֹנוֹת: טֶרֶם הִקְצִיעַ
כִּיוֵּן מַטְלֵית, מַקְצִיעַ, וְאִיסְפְּלָנִית הִצִּיעַ.

קֶשֶׁר קָמִיעַ מֻומְחֶה וּלְחִישׁוֹת
מְעֻולּוֹת דְּבָרוֹ, בָּדוּק וּמְהֵימָן כַּמָּה מַעֲלוֹת.
רֹפֵא אֻומָּן, לוֹ נִתְכְּנוּ עֲלִילוֹת,
אֲשֶׁר בְּיָדוֹ נַרְתֵּק רַב תְּעָלוֹת.

שִׁיקַּצְתִּי חַיַּי מְשֻׁובֶּשֶׁת מֵרׇוב-בּוֹשֶׁת,
טְעוֹנָה כְּלִימּוֹת וָרוֹק מִנִּי דַבֶּשֶׁת.
תְּבוֹעַ אָשִׁית, עַד יְבוֹשֶׁת תַּחְבּוֹשֶׁת –
אַחַר הַמַּכָּה - צְדָקָה לְךָ תִּלְבּוֹשֶׁת!

שְׁפוֹט לִטְרוֹף טֶרֶף עיניי תְּצַפֶּינָה,
לֹא לְשַׁחֵת בְּרִיּוֹתָיו דְּרָכָיו תְּרַפֶּינָה.
מִתּוֹךְ דִּין - לְרַחֵם, שׁוּרוֹתָיו תְּסֻובֶּינָה:
הִכְאִיב - וְיֶחְבָּשׁ, יִמְחַץ - וְיָדָיו תִּרְפֶּינָה.

הַמַּפְלִיא לַעֲשׂוֹת עִם בָּרִיא וְאָמֵל,
קְרוֹבִים רַחֲמָיו וְלֹא כְּחָצִיר קָמֵל;
טְרִיָּיה לִרְטִיָּיה יְגַלְגֵּל בּוֹ בְּאִיזְמֵל,
נֵס בְּתוֹךְ נֵס יְנוֹסֵס חוֹמֵל!

יְצוּרָיו חַיָּיבִים לְהוֹדוֹת חֶסֶד לָמוֹ כְּגוֹמֵל,
גַּלֵּי יָם יוֹרְדִים - וּמַכְנִיסָם לְנָמֵל,
דֶּרֶךְ תּוֹעִים יְיַשֵּׁר, יַעַר וְכַרְמֶל,
לְבוּטִים מְרַפֵּא, וּמַתִּיר חָבוּשׁ וְעָמֵל.

בִּין וּשְׁמׇור-אֵלֶּה, עַם חָכָם,
תָּר בְּנֹף מַכְאוֹבָם וּבְכוֹשֶׁר מְשָׁכָם,
וַיַּנְחֵם לָבֶטַח, וְיָם כִּיסָּה פּוֹרְכָם,
רִיגְּלָם הַמִּדְבָּרָה וְאַרְבָּעִים סִיפֵּק צׇורְכָּם.

חִזְקוּ וְיַאֲמֵץ לְבַבְכֶם מְיַחֲלִים לוֹ הוֹוִים,
סוֹלּוּ לָרוֹכֵב בָּעֲרָבוֹת, וּמְכַלְכֵּל יְהָבִים!
הוֹדוּ לְזֵכֶר קׇודְשׁוֹ, מוֹחֵץ רחבים,
וְעַל זֹאת דָּצוּ וְשִׁבְּחוּ אֲהוּבִים.״

״אֵין צוּר חֵלֶף
תַּבְנִית כֹּל גּוֹלֵף!
לְכוֹשֵׁל חִיּיוּל חוֹלֵף,
לִמְכַשֵּׁל - חִית שׁוֹלֵף.
בַּעַר שָׁקַד לְסַלֵּף
עַם דָּת אִילֵּף –
מְעַט פִּרְחֵי תֶלֶף
גִּיבַּר מֵרׇוב אֶלֶף.

גֻּומְּצֵי שִׁיחוֹת כָּר,
שְׁלֹשׁ עֶשְׂרֵה כִּירְכַּר,
רֹאוֹת-בְּרֵיכוֹת סָקַר –
בְּעִינּוּי חֹשֶׁךְ נָקַר.
דִּיבֶּר: שֵׁם יָקָר
כָּל מַזְכִּיר יְדֻוקָּר.
וְאָז, לָרוֹב הֻופְקָר,
מְרוֹד וּכְפוֹר בְּעִיקָּר.

הֶעֱרִים סוֹד בְּלִיַּעַל,
לְהוֹגֵי חֶן-יַעַל
מְסוֹךְ סַף רַעַל,
חַיֵּימוֹ הַמְרֵר וְהַרְעַל.
וְעוֹד הוֹסִיף מַעַל:
לָאֵל מוֹדֵד בְּשַׁעַל
בִּיטֵּל עוֹד מֵהַעַל
קָרְבַּן תָּמִיד מוּעָל.

זֶרַע קֹדֶשׁ הַחֲרֵץ,
חָכְמוּ קַלֵּי מֶרֶץ,
גְּזור אֶצְבַּע קֶרֶץ
כּוֹתֵב כְתֻובַּת פֶּרֶץ,
חִיבּוּל טֻומְאַת שֶׁרֶץ
גַּן נָעוּל הַפְרֵץ,
שִׁיתּוּק, אִיסּוּף הַפְרֵץ,
זִימָּה סַפֵּק אֶרֶץ.

טָסוּ חַרְגֻּולֵּי יֶלֶק,
זְמוֹרַי הַבְקֵק וְהַמְלֵק,
כְּנֵץ, צִיפּוֹר דּוֹלֵק
גּוֹזָלַיי בַּתֵּר וְהַמְלֵק.
יָקְדוּ כְּהַבְהֲבֵי עֶלֶק
חַרְחַר רִיב וָחֵלֶק,
עֲקוֹר מַתַּן חֵלֶק 
חַג, וּמְנוֹרַת דולק.

כִּינּוּיָם שְׁפַנֵּי סֶלַע –
הוּמְרוּ לְתַנִּינֵי בֶלַע,
לְבָאֵי שֵׁן וּמַתְלַעַ,
תְּבוּאַת קֹדֶשׁ לִילַע.
לְאֵיד הֵכִינוּ צֶלַע,
נְטוֹשׁ מְדָנִי גֶלַע,
בְּשַׂר קוֹדֶשׁ קָלַע
מֵעַם מָשׁוּל תּוֹלָע.

מִידַּת חָמָס גָּמְלוּ –
כֵּן גְּמוּל הׇוגְמְלוּ:
כְּסוּף יְאוֹר קָמֵלוּ,
וּכְרֹאשׁ שִׁיבּוֹלֶת נָמָלוּ.
נִיטְיוֹן רָעוֹת נִיכְּלוּ,
בַּל יוּכְלוּ, שֻׁוכְּלוּ:
כְּקַשׁ יָבֵשׁ אֻוכְּלוּ,
כָּלוּ בֶּעָשָׁן, כָּלוּ.

סוֹד שֵׁם לְחַלֵּל
תְּמוּר רָשָׁע מִתְחוֹלֵל
לְיוֹם הָפַךְ לֵיל,
לָכֵן יֻוגְמַר הַלֵּל,
עוֹז בְּכֵן יִתְמַלֵּל,
מִפִּי יוֹנֵק וְעוֹלֵל –
לַמַּשְׁבִּית צַר וּמִתְגּוֹלֵל
כֹּל הַנְּשָׁמָה תְּהַלֵּל!

פֻּולְּגוּ לְשִׁבְעָה חֲלוּצִים
וְקֻובְּעוּ שְׁמוֹנָה מְלִיצִים,
נֶגֶד שֶׁקּוֹדַת נִיצִּים
וּתְקוּפַת שְׁנַת קִיצִּים.
צִיעֲרוּ כְּעַקְרַבֵּי עֳקָצִים –
צוֹעֲרוּ כְּנִמְלֵי שְׁקָצִים
לוֹהֲטוּ כַּעֲצֵי קְצָצִים –
דּוֹעֲכוּ כְּאֵשׁ קוֹצִים.

קְהָלִים נוֹסְדוּ מְדָנַי
לִשְׁנֵים עָשָׂר עִידָּנַי –
רָב רִיבָם אֲדֹנָי,
עַל חוֹשְׁבֵי אׇובְדָנַי,
שְׁפוֹט כֵּן דִּינַי
עוֹד לְזוּלָתְךָ אֲדוֹנַי,
תְּאַבֵּד כֵּן זֵידוֹנַי,
כָּל אוֹיְבֶיךָ אֲדֹנָי!

שְׁמוֹר לְפִי מִלִּים
הַט, הַקְשֵׁב קוֹבְלִים,
טֵירוּד נוֹף בִּיהוּלִים
יָחִיל רוֹמִי בְעָלִים

יְצִיאַת יוֹם הִילּוּלִים
וְקֶרַע דִּין הוֹלְלִים
חֲזוֹת, קְרוֹא כִּמְהַלְלִים –
מִי כָמוֹכָה בָּאֵלִים!״

״אָב לְרַחֵם וְרַב-סְלוֹחַ, חוֹלַלְתָּנוּ
בְּנֵי בְרִית קוֹדֶשׁ, וְלֹא חִילַּלְתָּנוּ.
גּוֹלִים וְאַתָּה בְּקִרְבֵּינוּ כַּאֲשֶׁר יִיחַלְתָּנוּ –
וְסָלַחְתָּ לַעֲוֹנֵנוּ וּלְחַטָּאֵינוּ וּנְחַלְתָּנוּ!

דָּם וְגֵיאוּל תָּדִיחַ, דִּין תָּאִיר,
הַחֵטְא יׇועֳלַם - וּבַל יִמָּצֵא לְהַכְאִיר;
וְעָוֹן יְבֻוקַּשׁ וְאֵינֶינּוּ, כְּנַמְתָּה לְהַבְאִיר:
כִּי אֶסְלַח לַאֲשֶׁר אַשְׁאִיר.

זַהַם מְרֻובֶּה - וּפְסַק נִיחוֹחַ מְמֻולָּח.
חֲצִיצַת חֲפִיפַת לִיכְלוּךְ שֶׁרֶץ הַלָּח
טְבִילַת טָהֳרַת צִינּוֹר מֵימֶיךָ יְקַלַּח –
כִּי אַתָּה יְיָ טוֹב וְסַלָּח.

יַעַן סָרַחְנוּ - נִידַּחְנו, מִשְׁמָן רָזֶה,
כִּי רַבַּת הַטֻּומְאָה וְאֵין מַזֶּה.
לְמַעַנְךָ יְיָ כְּאָז, לְפָנִים מִזֶּה –
סְלַח נָא לַעֲוֹן הָעָם הַזֶּה.

מִסַּפֵּר עָצְמוּ חַסְדֵי רוֹב גְּמוּלָךְ!
נִיהַגְתָּ לְעַם בֵּית מֶרִי לְמוּלָךְ;
סִיבְלָם נָא תִשָּׂא, כְּמִקֶּדֶם נְטוּלָךְ –
וְסָלַחְתָּ לְעַמְּךָ אֲשֶׁר חָטְאוּ לָךְ.

עָלֵינוּ מִמָּרוֹם רוּחַ קׇודְשׁוֹ יֵעָרֶה
פָּרוֹשׁ וְהִתְקַדֵּשׁ, וְחָדָשׁ לָנוּ יִבָּרֵא.
צֵירוּף וּמֵירוּק בְּלֹא יִסּוּרִין תּוֹרֶה,
כִּי עִמְּךָ הַסְּלִיחָה לְמַעַן תִּוָּרֵא.

קוֹמֵם נְפוּלָה מִלְּהוֹסִיף עוֹד דַּאֲבוֹנֵינוּ
רוּחַ חֵן רַוֵּה, שַׂבַּע רַעֲבוֹנֵינוּ:
שָׁלְמוּ, תַּמּוּ, עֵת רִיעוּעַ, בּוֹנֵינוּ,
לְמַעַן שִׁמְךָ יְיָ וְסָלַחְתָּ לַעֲוֹנֵנוּ.

שִׁירַיִים הִנְנוּ - וּמִי כָמוֹךָ נוֹשֵׂא?
לֹא עַל צִדְקוֹתֵינוּ אוֹת נְנַסֶּה:
מִידַּת הֲמוֹן רַחֲמֶיךָ נַפִּיל מַחֲסֶה,
יְיָ שְׁמָעָה, יְיָ סְלָחָה, יְיָ הַקְשִׁיבָה וַעֲשֵׂה!

בְּיִרְאָתְךָ וּבְתוֹרָתְךָ הָכֵן פְעָמֵינוּ לִצְעוֹד,
בְּיַד עֲוֹנֵינוּ כִּי כָשַׁלְנוּ לִמְעוֹד,
נָא, יֵאָמֵן דְּבָרְךָ הַטּוֹב לִסְעוֹד:
כִּי אֶסְלַח לַעֲוֹנָם, וּלְחַטָּאתָם לֹא אֶזְכָּר עוֹד.

יוֹדֵעַ תַּעֲלוּם כָּל דָּם וּלְחוּם!
חַטֹּאותֵינוּ יְדַעֲנוּם, וְאִתָּנוּ בְּכָל תְּחוּם;
הִנְנוּ, אָתָאנוּ לָךְ, מְצוֹא כוֹפֶר וְתַנְחוּם,
וְאַתָּה אֱלוֹהַּ סְלִיחוֹת, חַנּוּן וְרַחוּם!״

״אוֹמְרָה לְאֵל: סַלְעִי,
לָמָה שְׁכַחְתָּנִי בְּצַלְעִי,
בְּלַחַץ אוֹיֵב לְבַלְּעִי,
מְחָרְפִי וּמַכֶּה לוֹעִי?

גְּבוּל רִשְׁעָה מֻצָּב –
נְוֵה צֶדֶק חֻוצָּב;
דְּבֵיקֶיךָ בְּרוֹגֶז וּמַעֲצָב,
סוּרֶיךָ - בָּרֶכֶב וּבַצָּב.

הוֹדַע כִּי הִנֶּךָ
עוֹמֵד בְּקֶרֶב מַחֲנֶיךָ,
וּבָנֶיךָ, בְּנֵי בְחוּנֶיךָ
פְּקוֹד, וְחוֹן מְחַנְּנֶיךָ.

זְעַם זְדוֹן הַמַּטֶּה
צְפִירַת דָּמִים וּמֻוטֶּה.
חֲרָבוֹת כְּמַדְקֵרוֹת בּוֹטֶה –
קִיצּוֹ וְחֵילוֹ יַטֶּה.

טָעוּן עוֹל מוֹטוֹ,
רוֹדֶה בָּאַף לַהֲמִיטוֹ –
יִשְׁפַּל שְׁחוֹחַ הַחוֹבְטוֹ:
שְׁבוֹר זְרוֹעוֹ וְשִׁבְטוֹ,

כְּשֵׁבֶר נֵבֶל יוֹצְרִים
כָּתוּת וּמְדֻוכָּא בַּחֲצֵרִים:
תָּחִישׁ שְׁלוֹחַ צִירִים,
תָּחִיל כִּמְצֵירָה מִצִּירִים.

שׁוֹרֶשׁ יִשַׁי נֵצֶר
לְפִתּוֹחַ זִיק הָאוֹצֵר,
מְמֻונֶּה, בְּעַמְּךָ עֹצֵר,
הָקֵם, אֱמוּנִים נֹצֵר.

הָרִאשׁוֹנִים נָא נְצוֹר,
קַדְמוֹנִים בְּרִית מִלִּבְצוֹר;
טִירָתְךָ מַלֵּא, וְצוֹר,
נָתוֹן תִּתֵּן כְּחָצוֹר.

יַגִּיעַ 'תַּם' לְצִיּוֹן!
גַּלֵה, פְּקֹד בְּעֶצְיוֹן
דִּבְרֵי עֲוֹונוֹת רִצְייוֹן:
לָקְתָ כִּפְלַיִם בני ציון.

בּוֹטְחֶיךָ אַל תַּחְפִּירֵם.
תּוֹרַת חֶסֶד תַּשְׁפִּירֵם.
וּגְמוּלוֹת שִׁיקַּרְנוּ וַנִכְפְּרֵם
רַחוּם, אַתָּה תְּכַפְּרֵם.

הֵן 'חָטָאנוּ צוּרֵינוּ'
נוֹאֲמִים נֶעֱצָרֵנוּ,
סְלַח לָנוּ יוֹצְרֵינוּ,
תֵּיעָתֶר-נָא מִצָּרָתֵינוּ!״

״אוֹרַח צְדָקָה וְעַלְמוּת נְחִיַּית דַּבָּרֶיךָ,
אוֹמֶץ לְהַעֲצִים עֲשׂוֹת רְצוֹנְךָ נִדְבָּרֶיךָ,
'בְּרוֹגֶז רַחֵם תִּזְכּוֹר' יִחַלְתָּנוּ סִבְרֶיךָ –
בְּמִשְׁפָּט אַל תָּבֹא, וְתֶאֱחוֹז דּוֹבְרֶיךָ.

גֶּזֶר גָּמוּר וּבֵינוֹנִי נֶגְדְּךָ מָתוּחַ,
גּוֹל אֵלֶיךָ כִּיוַּנְתָּה שַׁעַר פָּתוּחַ,
דּוֹרְשֵׁי קִרְבָתְךָ בְּלֵב קָרוּעַ וְרָתוּחַ –
דִּינָם לְהַצְהִיר - צֶדֶק קַבֵּל כְּנִיתּוּחַ.

הֵן כַּבִּיר וְשַׂגִּיב וְשַׂגִּיא כֹּחֶךָ!
הִנְנוּ אָתָאנוּ בְּדַלּוּת קַדֵּם נׇוכְחֶיךָ,
וְאַתָּה רַחוּם, מַסְבִּיר, מְקַבֵּל בּוֹרְחֶיךָ –
וּמִי כָמוֹךָ מוֹרֶה חֲטָאִים אׇורְחֶיךָ.

'זִכִּיתִי לִבִּי' ומִי בַעַל דְּבָרִים זָךְ,
זוֹכֶה וּמְזַכֶּה לְחַטֵּא גְּבָרִים?
חוֹבוֹת, חֲבִילוֹת וּגְדִישִׁים מִכָּל עֲבָרִים,
חֶסֶד יִגְדַּל לְרַפְּאוֹת מִיטַּת שְׁבָרִים.

טִיף הָעִיקָּר סָרוּחַ, וּמִתּוֹכוֹ מַתְלִיעַ,
טֶנֶף תְּבָלִין שְׂאוֹר בָּעִיסָּה מַבְלִיעַ,
יוֹקֶשׁ לְמִישׁוֹר, חֵטְא וְעָוֹן לְהַקְלִיעַ,
יָקָר לְזַלְזֵל בְּכָל תּוּשִׁיָּה לְהַגְלִיעַ.

כָּשַׁל הַנִּדּוֹן, זֻונְּחוּ פְּרַקְלִיטֵי יוֹשֶׁר,
כִּיבּוּשׂ, כִּיבּוּס, גֶּזֶר מִלִּשְׁכַּת הָעוֹשֶׁר:
לַמּוֹרָא אֵיךְ אֵין לְהָבִיא תֶּשֶׁר,
לָכֵן רַב הַפֶּגֶר וְאַמִּיץ הַקֶּשֶׁר.

מֵעֵת הוּסַר הַתָּמִיד וְעוֹלוֹת מֵחֶיךָ –
מכיחי לֵב מִנְחָה וְנִיחוֹחִים מְשַׂמְּחֶיךָ.
נְזוּפִים בָּרִיא אוּלָם מִשִּׁימּוּשׁ שְׁבָחֶיךָ –
נִשְׁבְּרוּ, טֹהֲרוּ, הׇוכְשְׁרוּ כְּקָדְשֵׁי זְבָחֶיךָ.

סֻלְּקוּ סְנֵיגוֹרֵי עַמְּךָ, דּוֹרְשָׁן וקרובן,
סְמוּכִים לְעֵד קְרוֹבֶיךָ בְּךָ, קְרוֹבָם,
עֲשׁוּקִים כִּי הֵמָּה, גּוּף וּמַעֲרָבָם,
עֲרֻובַּת טוֹב וְהַצָּלָה הֱיוֹת עֲרֵבָם.

פֶּגַע, שִׂיחַ וּמַעֲמָד - אֻומָּנוּת קַדְמוֹנִית –
פִּי תוֹלַעַת יַעֲקֹב תֵּחַת אַדְמוֹנִית!
צוֹעֶקֶת שְׁלֹשׁ עֶשְׂרֵה, מִשּׁוּב רֵיקָנִית,
צוֹוַחַת קוֹל עָרֵב יוֹנָה שַׁתְקָנִית.

קוֹמְמִיּוּת קֶדֶם הִילַּכְנוּ - וְעַתָּה קְדוֹרַנִּית,
קֶרֶן נְבוּאָה וּמַלְכוּת, לְוִיָּה וְאַהֲרֹנִית,
רוֹמֵם צִדְקָם, וּזְכוֹר בְּרִית אַחֲרוֹנִית –
רִאשׁוֹנִית גִּידַּלְתָּ, וְגִידּוּלֶיהָ מִנְיָין אֲחוֹרַנִּית.

'שׁוּבָה וַאֲנִי עֵד!' - פִּיַּיסְתָּ תְּשׁוּבָתֵינוּ,
שַׁכֵּךְ כַּעַסְךָ וַהֲשִׁיבֵנוּ, וְשׁוּב שְׁבִיתֵנוּ.
תָּשׁוּב, תְּרַחֲמֵינוּ, תִּכְבּוֹשׁ וְתַשְׁלִיךְ חוֹבוֹתֵינוּ –
תִּתֵּן אֱמֶת לְיַעֲקֹב כְּנִשְׁבַּעְתָּ לַאֲבוֹתֵינוּ!

שָׁמְרָה דַּל שְׂפָתֵינוּ וּבְקִרְבֵּינוּ תִּשְׁכּוֹן,
לְמִקְדָּשׁ מְעַט חֲשׁוֹב כִּבְחוֹבֵנוּ מַשְׁכּוֹן:
מֻוקְטָר וּמֻוגָּשׁ בְּאֵין קְטוֹרֶת מַתְכּוֹן –
הַעֲרֶב-נָא תְּפִילָּתֵינוּ - וּלְפָנֶיךָ תִּיכּוֹן!״

״אָז, קַשְׁתִּי וְחַרְבִּי
אִוָּנִי, וְלֹא מֵרֻבִּי.
אֵיךְ חָר בִּי!
חַם לִבִּי בְּקִרְבִּי.

בְּקִרְבִּי יֵשׁ בְּדִלְדּוּלִים
בּוֹעֲרִים פִּיהֶם מַגְדִּילִים:
בֹּקֵעַ אַיֵּה גַלִּים,
דֶּרֶךְ לַעֲבֹר גְּאוּלִים?

גְּאוּלִים קֶדֶם כְּבָר
גְּאוֹנָם מְאֹד גָּבַר;
גָּזוּ בְּכָל מַעְבָּר,
וְאֵין מַכְלִים דָּבָר.

דָּבָר שָׁלַח וְצִוָּה
דֹּחֵיתִי בְּהֵנַע וְעִוָּה.
דַּנְתִּי בְּלֹא שִׁוָּה,
לְמַעַן סְפוֹת הָרָוָה.

הָרָוָה, נָוָה וְעִדַּן –
הֵסִיר הַמִּצְנֶפֶת וְכִדַּן:
הָעִדָּן שִׁנָּה וְדָן,
חֶרֶב וְהֶרֶג וְאַבְדָן.

וְאַבְדָן אַשּׁוּר וְגוֹזָן
וּדְחִי זִרְמַת מְיֻזָּן;
וְצָרוֹת מְפִיקוֹת רָגְזָ
מִזַּן אֶל זַן.

זַן וְשָׁת כְּגָבְרוּ –
זָנַח וְחִיֵּב כְּעָבְרוּ.
זְאֵבִים וָמַיִם דֻּובְּרוּ –
כָּל אֵלֶּה חָבְרוּ.

חָבְרוּ לְעֵמֶק כְּרִיָּה
חֲרוּשָׁה בְּמַחֲרֶשֶׁת נָכְרִיָּה,
חִבּוּל וָפֶצַע פּוֹרִיָּה,
חַבּוּרָה וּמַכָּה טְרִיָּה.

טְרִיָּה וְלוֹ אֵין פְּגוּעָה,
טַרְחוּת וְלֹא רְגוּעָה,
טְרוּפָה פָּרָה וְגוֹעָה,
עַד כִּי יָגְעָה.

יָגְעָה יָד מִנּוֹשֵׁף,
יוֹנָה לְפַח וּלְשַׁפְשֵׁף,
יִסַּבּוּ בְּנֵי רֶשֶׁף –
כָּשַׁלְנוּ בַצָּהֳרַיִם כַּנֶשֶׁף.

כַּנֶשֶׁף אִישׁוֹן וַעֲלָטָה –
כִּוֵּן עֵת הִמָּלְטָהּ
כַּחֲשֵׁכָה אוֹרָהּ קְלוּטָה –
הִנֵּה הִיא לוּטָה.

לוּטָה אַדֶּרֶת עַזֵּק
לְשַׁלֵּם הַמַּבְעִיר הַחֶזֵּק.
לֻוהֲקוּ קוֹצִים לְרַזֵּק –
וְאֵין אֶחָד מִתְחַזֵּק.

מִתְחַזֵּק וּמַחֲזִיק טֻמָּנוּ,
מְחָרֵף וּמְגַדֵּף זֻמָּנוּ;
מְלִיצִים נִחוּמִים עִמָּנוּ –
מַיִם לֹא נֶאֱמָנוּ.

נֶאֱמָנוּ רָעִים מִלְּחַסְּרָהּ,
נָכוֹנוּ שְׁפָטִים לְיַסְּרָה,
נָבָל בְּנָדִיב, הִמָּסְרָה,
כִּי דִבֶּר סָרָה.

סָרָה עֲנִיּוֹת נִכְזָבוֹת:
'סֻלְּקוּ תִּקְרָה וּמַעֲזֵבוֹת,
סוּרוּ!' - הִזִּיל כְּזָבוֹת,
וְכֶאֱסֹף בֵּיצִים עֲזֻבוֹת.

עֲזֻבוֹת עָרִים וּפִנּוֹת,
עָרוּ עָרוּ כְּנָפוֹת,
עַרְעָר בְּכָל דְּפָנוֹת,
מֵאֵין עוֹד פְּנוֹת.

פְּנוֹת אֶל הַמִּנְחָה,
פָּעַלְתִּי עָוֶן וְנִזְנָחָה.
פֶּרֶץ וְיוֹצֵא וּצְוָחָה,
סוֹרֲרִים שָׁכְנוּ צְחִיחָה.

צְחִיחָה, צָמָא וּמַדְקִיר,
צָעַק וָשׁוֹד כְּהָקִיר –
צָרוֹת מָתַי לְהַעֲקִיר
נְגַשְׁשָׁה כַעִוְרִים קִיר.

קִיר עֵירָה וַיַּהֲרוֹשׁ
קָנֶה קִיַּמְתָּ וַיִּשְׁרוֹשׁ.
קָמֶיךָ עוֹלִים כִּבְרוֹשׁ,
וּמְשַׂנְאֶיךָ נָשְׂאוּ רֹאשׁ.

רֹאשׁ זִנַּבְתָּ לְהַחֲלִישׁ,
רוֹאֵימוֹ יְרֵאֵימוֹ לְהַפְלִישׁ.
רִעַלְתָּ כּוֹסָם לְהַגְלִישׁ
וַתַּשְׁקֵמוֹ בִּדְמָעוֹת שָׁלִישׁ.

שָׁלִישׁ בְּמוֹעֵצוֹת מִבְחָר
שָׁבַת מִכָּל מִסְחָר.
שִׁמְעָה: גְּרוֹנֵינוּ נִיחַר,
וַעֲשֵׂה, אַל תְּאַחַר.

תְּאַחַר שִׁלּוּמַת מְאֻוחָר
תּוּחַשׁ כָּעֵת מָחָר:
תְּנַקֶּה כְּצֶמֶר צָחַר
הַנִּשְׁקָפָה כְּמוֹ שָׁחַר.

שַׁחַר, מוֹצִיא מַזָּרוֹת,
לָאוֹר תּוֹצִיא גְזֵירוֹת.
מַהֵר יְשׁוּעוֹת נְצוּרוֹת
כִּי נִשְׁכְּחוּ הַצָּרוֹת.

הַצָּרוֹת, חֵטְא הִטָּנוּ,
הִרְבָּה כּוֹשֵׁל וּמִעֲטָנוּ.
קַח טוֹב אֲמָרֵינוּ,
נוֹאַלְנוּ אֲשֶׁר חָטָאנוּ!״

״אֵין כְּמִידַּת בָּשָׂר מִידּוֹתֶיךָ!
אַיֵּה קִנְאָתְךָ וַעֲצַת עֶמְדָּתֶיךָ?
בַּת בֵּירַרְתָּ לְבֵית חֶמְדָּתֶיךְ –
בְּעָלוּהָ אֲדוֹנִים, וְאֵין לְצִמְדָּתֶיךְ.

גּוֹעָה: לְדוֹדִי אֲנִי מְאוֹרֶסֶת!
גּוֹעַל רוֹדִי - וְאֵיךְ נִדְרֶסֶת?
דִּין הֶרֶג עַצְמָהּ מְקַנֶּסֶת,
דְּחוּיָה, גַּם כִּי נֶאֱנֶסֶת.

הָאֵם - בְּלִי סֵפֶר לְשִׁילּוּחַ,
הַבֵּן - כַּיָּתוֹם אוֹבֵד שָׁלוּחַ.
גַּן נָעוּל הֶפְקֵר כְּמַלּוּחַ,
וּמַעְיַן חָתוּם - נִרְפָּס, דָּלוּחַ.

זַעַם כְּרֶגַע - וְעַתָּה לְהִיפּוּךְ.
זֵירֵי קוֹדֶשׁ חוּצָה לְשִׁיפּוּךְ.
חֲבִיבַת רֵיעַ כְּקֶרֶן הַפּוּךְ –
חֲשׁוּבָה וַעֲזוּבָה כְּקוֹרַעַת בַּפּוּךְ.

טוֹב הַתּוֹאַר קְדוֹרַנִּית מִפֶּרֶךְ,
טְרִיָּיה וּמַכָּה מִבְּלִי אֶרֶךְ:
יוֹם נָקָם - נִסְתַּם דֶּרֶךְ,
יוֹם שִׁילּוּם - נֶחְתַּם חֶרֶךְ.

כֹּחַ הַסַּבָּל הִכְשִׁיל נֵטֶל,
כָּסוּל, שָׁחוּל הַנִּשְׁאָר מִקֶּטֶל,
לֹא לְמַרְגּוֹעַ נָד וּמִטַּלְטֵל,
לְעָמָל וְיָגָע - וְאַחֵר נוֹטֵל.

מַטַּע קֶרֶן שֶׁמֶן הַמְדֻוֹשָּׁן –
מִרְמָס מַדּוּעַ כַּחֲרֻולֵּי כִבְשָׁן?
נוֹצֵר לְמַעְלָה לֹא יִישָּׁן,
נִטְעוֹ לְמִשְׁלַח פָּרוֹת הַבָּשָׁן.

שׂוֹרֵקָה מַה בֶּצַע נִקְטֶפֶת?
סוֹרַחַת מַה יִּתְרוֹן נֶחֱטֶפֶת?
עַל יַד חֲקוּקָה וְטוֹטֶפֶת
עַל מָה אַהֲבָתָהּ נִשְׁטֶפֶת?

פְּעֻולַּת הָרוֹבִים בְּמִיעוּט מְלֶאכֶת,
פְּרִיעַת שָׂכָר רַבָּה וְהוֹלֶכֶת,
צוֹפָה הִיפָּנוֹת - עִתָּהּ נִמְשֶׁכֶת,
צָרָה פְּקוּדָה וְאַחֶרֶת נִסְמֶכֶת.

קוֹדֶשׁ רֵאשִׁית עֲרֵימַת שְׁעָרִים
קַצְוֵי אֶרֶץ זְרוּיִם כִּשְׂעוֹרִים,
רוּחַ מְנַשֶּׁבֶת בַּעֲלֵי יְעָרִים
רִידּוּף נִשְׁמֶטֶת נָסִים וְנִסְעָרִים.

שְׁמוֹנֶה וְתִשְׁעִים הָאָלוֹת הָאֵלֶּה,
שָׁלְמוּ אֵלֶּה, וּמֵאֵלֶּה, וְכָאֵלֶּה,
תְּרַחֲמֵינוּ, וּתְקַבְּצֵינוּ מִקְּרָנוֹת אֵלֶּה,
אָז יֹאמְרוּ בַגּוֹיִם: הִגְדִּיל יְיָ לַעֲשׂוֹת עִם אֵלֶּה!״

״אֵין מִי יִקְרָא בְצֶדֶק!
אִישׁ טוֹב - נִמְשַׁל כְּחֵדֶק.
בַּקֵּשׁ בְּעַד שְׁחוּקֵי הָדֵק –
בְּשׁוּם פָּנִים אֵין בֶּדֶק.

גֶּבֶר תָּמִים וְנָבָר אָפַס.
גָּמַר חָסִיד - וְצַדִּיק נִרְפַּס.
דּוֹר עָנִי בַּעֲוֹנוֹ נִתְפַּס –
דְּבָרָיו לְהַגִּיד מִי יְחֻופַּשׂ?

הוֹסַפְנוּ בַּעֲווֹנֵינוּ חֵימָה; לְהַגְעִיר
הַמִּתְנַדְּבִים כִּבְנֵי בַיִת לְהַפְעִיר.
וּמַה יַּעֲצׇור-כֹּחַ צָעִיר
וִוידּוּי וָפֶלֶל לְקַדִּישׁ וָעִיר?

זָחַלְתִּי וָאִירָא בְּעַד מַחַן
זַעַק לְחַוּוֹת לַחוֹקֵר בְּבֹחַן;
חֲסַר חֶסֶד וְיִתּוּר צַחַן –
חֵן אֵיךְ אֶמְצָא בְּתַחַן?

טוֹב לְקוֹרְאֶיךָ בְּנֶפֶשׁ רַהַב –
טׇורְחָם שׂוֹא וּלְכַלְכֵּל יַהַב,
יְקַר חַסְדְּךָ עָלַי יֻורְהַב:
יַעַן קוֹלִי לְהַאֲזִין בְּאַהַב,

כַּהֲגוּן-מִידּוֹת וּבִתְפִילָּה שָׁלֵם,
כְּזָקֵן רָגִיל וְלֹא כְגוֹלֶם,
לְהֵחָשֵׁב נֶגְדְּךָ דִּכְאִי מִלְּהִכָּלֵם –
לְרַחֲמֶיךָ זׇוכְרָה, לְרַוְוחָתִי מֵהִתְעַלֵּם.

מְרֻבִּים צְורָכֵינוּ, אֵין לְהֵאָמֵר
מִקּוֹצֶר דֵּיעַ וּמֵרׇוב-מֶימֶר,
נֶגְדְּךָ הַכֹּל, יוֹצֵר חוֹמֶר,
נוֹהֵג וְרוֹעֶה, צֵל וְשׁוֹמֵר.

שׂוֹרַדְנוּ כְּתֹרֶן-הָר בְּדוּדֵנוּ,
סְחִי וּמָאוֹס הוּשַׂם כְּבוֹדֵינוּ,
עֲנֵינוּ, וּתְנֵינוּ מִחְיָה בְּשִׁיעְבּוּדֵינוּ,
עוֹד לְמִנְיָינְךָ בַּקֵּשׁ אֲבוּדֵינוּ.

פְּקֻודַּת נִגְעֵי תוֹכְחוֹתֶיךָ שְׁבוּטִים,
פְּזוּרִים, פְּרוּדִים, וּבַגּוֹיִם עֲבוּטִים,
צָפְנֵם בְּסֻכְּךָ מֵרִיב וּשְׁפָטִים
צְפִיַּית תִּפְאַרְתְּךָ לָמוֹ מַבָּטִים.

קוֹל כֹּחֲךָ לַהַב חוֹצֵב,
קֶצֶב טוֹב וְחִילּוּפוֹ קוֹצֵב:
רֵיעֶיךָ דוֹפְקִים בְּקוֹל עָצֵב,
רְצוֹת נִדְבָתָם וּבְקִרְבָּם הִתְיַצֵּב.

שׁוֹקְדִים בְּצוֹם לִבָּם לְהַכְנִיעַ,
שְׁאוֹנָם מִזַּעַם בַּחֲדָרֶיךָ תַצְנִיעַ,
תּוֹבְעִים בְּלַחַשׁ שָׂפָה לְהָנִיעַ –
תַּאֲוָתָם אַל נָא תַמְנִיעַ.

שִׁמְךָ אֱלֹהִים חַיִּים מִתְפָּאֵר,
לְחַיִּים טוֹבִים מִמְּךָ נִשָּׁאֵר.
מְקוֹר חַיִּים עִמְּךָ מִתְבָּאֵר –
הִבִּיטָה וַעֲנֵינוּ, וְעֵינֵינוּ הָאֵר!״

״אַךְ בְּךָ מִקְוֵה יִשְׂרָאֵל, יְיָ,
אָחוֹר וָקֶדֶם עַם נוֹשַׁע בַּיְיָ,
וּמַדּוּעַ בּוֹשֵׁשׁ יוֹם זֶה לָבוֹא כַּמָּה עִידָּנָי,
לִישׁוּעָתְךָ קִוִּיתִי יְיָ.

גַּרְתִּי וָאֵחַר מֶשֶׁךְ יְמֵי יְגוֹנַיי
גֵּיא גּוֹי נָבָל וְעַם גְּנַאי,
וּמַדּוּעַ דַּל כָּכָה כָּבוֹד וְלִי דִּינַיי
קַוֵּה אֶל יְיָ חֲזַק וְיַאֲמֵץ לִבֶּךָ וְקַוֵּה אֶל יְיָ.

הֲלֹא אַתָּה מִקֶּדֶם חֶרֶב גַּאֲוָתֵינוּ
הִקְוִינוּךָ אַף אוֹרַח מִשְׁפָּטֶיךָ תִּקְוָתֵינוּ,
וּמַדּוּעַ וְאֵין אִישׁ מְכַפֵּר עַל חוֹבוֹתֵינוּ,
וּמִדַּאֲבוֹנֵינוּ אוֹמְרִים: יָבְשׁוּ עַצְמוֹתֵינוּ, וְאָבְדָה תִּקְוָתֵנוּ.

זָכַרְתִּי חֶשְׁבּוֹנוֹתָם - וּמִדָּתָם קָצְרָה,
זֶה פַּעֲמַיִם קָצִיר, וְלֹא עָצְרָה;
וּמַדּוּעַ חֻושַּׁבְנוּ כַבְּהֵמָה וְנִטְמִינוּ מִשִּׁיטַּת הַשּׁוּרָה,
מִקְוֵה יִשְׂרָאֵל, מוֹשִׁיעוֹ בְּעֵת צָרָה.

טוֹעֲנִים וּפוֹרְקִים בָּנִים הַכָּתוּב סִיבְּלָם,
טְרוּדֵי חַבְלָם וְחָרֵב בֵּית זְבוּלָם,
וּמַדּוּעַ יַסְתִּיר הָאָב שְׁבִילָם הָאָמוּר בִּגְלָלָם:
וְיֵשׁ תִּקְוָה לְאַחֲרִיתֵךְ נְאֻם יְיָ וְשָׁבוּ בָנִים לִגְבוּלָם.

כִּמְעַט נָטָיוּ רַגְלָי וְשֻׁופְּכוּ אֲשׁוּרִים,
כִּי קִנֵּאתִי בַּהוֹלְלִים, מְעַרְבְּבֵי מֵישָׁרִים,
וּמַדּוּעַ לֹא תִקְצַר וְתִשָּׁבֵר יַד רָמַת הַקּוֹשְׁרִים,
וְקוֹיֵ יְיָ יַחְלִיפוּ כֹחַ, יַעֲלוּ אֵבֶר כַּנְשָׁרִים.

מָלֵא רֻודַּפְתִּי וַחֲסֵרָה שְׁנַת גְּאֻלַי,
מְסוּרִים לְמַכִּים וּמוֹרְטִים וּלְתִקְוָה אוּלַי,
וּמַדּוּעַ נִסְחֲפוּ אַבִּירַי וּמְתֵי שִׁיר איילי,
קַוֹּה קִוִּיתִי יְיָ וַיֵּט אֵלַי.

שָׂרִיגֶיךָ, סוּגֶיךָ, בְּקָקוּם בּוֹקְקִים וְהִבְאִישׁוּ,
סִירֵי סוּרֶיךָ גָּבְשׁוּ וְלֹא עָבְשׁוּ,
וּמַדּוּעַ עָתְקוּ וְגָבְרוּ וְכִמְדֻובָּר לֹא יֵבוֹשׁוּ,
מִקְוֵה יִשְׂרָאֵל יְיָ כָּל עֹזְבֶיךָ יֵבֹשׁוּ.

פּוֹעֲלֵי אָוֶן יִתְאַמְּרוּ: אִיוִּינוּ, רִיוִּינוּ,
פָּצוּ אוֹתֹתָם: אוֹתוֹת שַׂמְנוּ, הִתְוִינוּ,
וּמַדּוּעַ צָלְחָה דַּרְכָּם וְחַסְדְּךָ נדמינו וְנִדְוִינוּ?
יְיָ חָנֵּינוּ, לְךָ קִוִּינוּ!

קִימַת סֻוכַּת שְׁלֹשׁ עֶשְׂרֵה פֶּרֶץ
קִיצָהּ רָחַק וּלְמִקְרֶה עָלָה הַפּוֹרֵץ,
וּמַדּוּעַ רְשָׁעִים יִחְיוּ וְלֹא יִדְּמוּ קֶרֶץ?
וְקֹוֵי יְיָ הֵמָּה יִירְשׁוּ אָרֶץ.

שׁוֹסֵי חֶבֶל נַחֲלָתְךָ, מְחַלְּקֵךְ לְגָרְשֵׁינוּ,
'שׁוּבִי שׁוּבִי הַשּׁוּלַמִית' בְּפִיהֶם, לְהַפְרִישֵׁנוּ.
'וּמַדּוּעַ תִּתְנַשְּׂאוּ?' - זֶה נָשִׁיב, וְשִׁבְחֲךָ נְפָרְשֶׁינּוּ,
טוֹב יְיָ לְקוֹיָו, לְנֶפֶשׁ תִּדְרְשֶׁנּוּ.

שַׁדַּי, תְּשׁוּבוֹת מַדּוּעֵינוּ פִּינוּ נְמַלֵּא:
לַיְיָ אֱלֹהֵינוּ חָטָאנוּ וְתִקְרֶאנָה אוֹתָנוּ כָּאֵלֶּה,
מִשְׁפָּטֶיךָ אֱמֶת וְאַתָּה מָרוֹם וּמִתְעַלֶּה,
הַצְּדָקָה לְךָ, וְלָנוּ הַבּוֹשֶׁת נְגַלֶּה.

הֵעַדְתָּה רַבּוֹת וַנֹּאמֶר: לֹא נַקְשִׁיב.
קוֹל טִיף נְבִיאֶיךָ בָּזִינוּ לְהַקְשִׁיב,
יִגְדַּל כֹּחַ, סְלַח, יְחִידִים מוֹשִׁיב,
בִּתְשׁוּבָה שְׁלֵמָה אוֹתָנוּ יָשִׁיב.

'מַדּוּעַ אָדוֹם לִלְבוּשֶׁיךָ' תְּקָרֵב עוֹנָתוֹ,
יֵז נִצְחָם עַל בִּגְדֵי עֲטִיָּיתוֹ,
נוֹדֶה סֶלָה יוֹם הַבָּא בְּשַׁעְתּוֹ,
זֶה יְיָ קִוִּינוּ לוֹ נָגִילָה וְנִשְׂמְחָה בִּישׁוּעָתוֹ.״

״אַלּוּף נְעוּרַיי נוֹעַד,
אִימְּצִי, גַּם סָעַד,
אַט בְּצַו וָועַד –
הַטּוֹב יְכַפֵּר בְּעַד.

בְּעַד שְׁלוֹם הָעֵדֶר
בְּלִי מִכְלָה וָדִיר,
בְּפֶרֶץ אֵין כְּסֵדֶר
אִישׁ גֹּדֵר גָּדֵר.

גָּדֵר לְקֵץ הַמֻּובְדָּל,
גָּבַר עָוֹן וַיִּגְדַּל,
גַּלְגֵּל לֹא יֶחְדַּל
עַל עַם דָּל.

דָּל מִמַּשְׂכִּיל וּמוֹרֶה,
דּוֹרֵשׁ, קוֹרֵא וּמַקְרֵא,
דַּעַת כִּי לַמִּקְרֵא –
בַּעֲצַלְתַּיִם יִמַּךְ הַמְּקָרֶה.

הַמְקָרֶה וכונה מַעֲלוֹת,
הַעְתֵּר פֶּלֶג תְּעָלוֹת
הַחוּצָה, כָּכָה עוֹלוֹת
זִיף וָטֶלֶם וּבְעָלוֹת.

וּבַעֲלוֹת גּוֹעוֹת הֶאֱזִינָה,
בְּשִׁיטָּתְךָ זוּן נָא,
וְעוֹרֶף לֹא יְזִיזֶנָּה –
וּמֵצַח אִשָּׁה זוֹנָה.

זוֹנָה - לְהַעֲצִיר גְּזֵירָה –
זְרֵה, וְחִנָּם מְזוֹרָה,
זַעַם אִם הֶחֱרָה
וְאַף יְיָ חָרָה.

חָרָה וְהִנָּחֵם וְהֶרֶף
חַיָּיתֵינוּ הַעַל מִטֶּרֶף,
חִישׁ פְּתַח עֶרֶף
הָאוֹצָר, וִיהִי טֶרֶף.

טֶרֶף לִשְׁכַּת הַמַּצָּע
טֶנֶף בְּרָכוֹת מַצַּע
טוֹרְדוּ חוֹק לְבַצַּע
חָנֵף כִּי יִבְצַע.

יִבְצַע, יַעַשׂ גַּפַּיִים,
יִקַּח נֶפֶשׁ אֲגַפַּיִים,
וְכֻוסֶּה אוֹר בְּמָגַפַּיִים
עַל הָרַע כַּפָּיִם.

כַּפַּיִים וְשַׂרְעַפִּים שׁוֹבָב
כטוות, מִיָּד יְחוֹבָב,
כִּי כֵן יְדוֹבָב
וַיְיָ יִרְאֶה לַלֵּבָב.

לַלֵּבָב מָסוּר סְתוֹם,
לִיבַּבְנוּ בְּלִי תּוֹם,
לָכֵן וִוכְחֵנוּ חֲתוֹם –
אֵין בּוֹ מְתֹם.

מְתוֹם לֹא מוֹצְאִים,
מִכֶּלֶא לֹא יוֹצְאִים,
מַשָּׂא דּוּמָה נוֹשְׂאִים
שְׁנֵים עָשָׂר נְשִׂיאִם.

נְשִׂיאִים בְּלִי כֶתֶר
נִדְבַת שָׁוְא פֹּתֵר,
נבעלים שְׁמֵי עֶתֶר
בין עמים לְשׁוֹן סֵתֶר.

סֵתֶר, גָּלוּי, תַּרְעֵלוֹת
סְחוֹר, בְּלִי תְעָלוֹת,
סַפֵּר רַב פְּעֻולּוֹת,
וְנִשְׁאַר בּוֹ עוֹלֵלֹת.

עוֹלְלוֹת עַם נוֹשַׁע
עַדֵּן, עַנֵּג, תִּישַׁע,
עַל כְּבֵן הִשְׁתַּעְשַׁע –
עֲבֹר עַל פָּשַׁע.

פֶּשַׁע נַקֵּה מִלְּצוּדוֹ,
פֻּרְעָנוּת פַּנֵה מִצִּידּוֹ,
פִּירְנוּס זַמֵּן רִצּוּדוֹ,
יָכִין לָעֹרֵב צֵידוֹ.

צֵידוֹ לְשׂוֹבַע מַנֵּה,
צִימְצוּם אִם יִקְנֶה,
צֹאנְךָ חֲמוֹל, קַנֵּא,
אֵל עֶלְיוֹן קֹנֶה.

קונה עַמְּךָ וּקְרַב
קְרוֹא כְּעָבָר רַב;
קַבֵּל מַרְעִיף וְיֶעֱרַב
עֲלֵי אָדָם רָב.

רַב רִידּוּי נִדְחָקִים,
רְאוֹת וּבוֹשׁ שׂוֹחֲקִים
רְצֵינוּ, צֵאת מִשְּׁחָקִים,
אֲשֶׁר יִזְּלוּ שְׁחָקִים.

שְׁחָקִים, צוּר, לְכַלְכְּלָה,
שִׁיקּוּי לְפֵירוֹת כַּלְכְּלָה
שׁוֹאֲלִים, כִּי לְכָלָה
רַחֲמֶיךָ לֹא תִכְלָא.

תִּיכְלָא רוֹעַ נִישּׁוּב,
תָּנִיף נְדָבוֹת חָשׁוּב,
תַּעְשְׁרֶנָּה, לִפְקוֹד שׁוּב,
אֱלֹהִים צְבָאוֹת שׁוּב.

שׁוּב שֶׁבַע יֻושְׁקַל
לֶקַח מְהֲלְלֶיךָ הֵקַל,
הַבַּרְזֶל אִם יִתָּקֵל
וְהוּא לֹא פָנִים קִלְקַל.

קִלְקַל הַדּוֹר מִלְּחַשֵּׁר.
טוֹב וְיָשָׁר אֲשֶׁר
נִיב חֲיָילִים –לְיַישֵּׁר –
יְגַבֵּר, וְיִתְרוֹן הַכְשֵׁר.

הַכְשֵׁר וְכִיפּוּר יִסָּפֵר
בְּתוֹרָה תּוֹדָה יִשָּׁפֵר,
הֶנֶג, פִּינוּ יְסַפֵּר,
הַיּוֹם הַזֶּה יְכַפֵּר.

יְכַפֵּר חֵטְא הִישְׂטָנוּ
גַּם עָוֹן היססנו,
דְּבַר פֶּשַׁע הִבְעִיטָנוּ –
לַייָ אֱלֹהֵינוּ חָטָאנוּ.״

״אִם אָמְרִי אֶשְׁכְּחָה מְרִי שִׂיחִי –
תּוּגוֹת, דְּאָגוֹת, תּוֹגְיוּן נַפְשִׁי לַהֲשִׂיחִי,
בְּלִי נוֹחַ עֶצֶב לְמוֹתָר מַטְרִיחִי.
שַׁחוֹתִי עַד מְאֹד, וַתִּקְצַר רוּחִי.

גִּישׁוֹת צָרוֹת סֵירוּג, שְׁתֵּים-שְׁתַּיִם,
רֶיוַח אֵין בִּנְתַּיִם, מְעִיקוֹת כִּנְחֻשְׁתַּיִם.
דּוֹלְקוֹת, דּוֹבְקוֹת, בְּלִי עֲצַלְתַּיִם,
קַץ הַמֵּצֵר, סוּר שְׂחוֹק וּמְצִלְתַּיִם.

הִרְכַּבְתָּ לְרֹאשׁ עַמְּךָ חוֹרֵשׁ גַּבָּם;
צֵאת לָרְוָיָה צוֹעֲקִים לְךָ מִשְׂגַּבָּם.
וַתִּרְבֶּינָה סַרְעַפּוֹתָיו, וְכָל הַיּוֹם בְּקִרְבָּם 
פֹּאורוֹתָיו תֶּאֱרַכְנָה, וְיָשׁוּב, יַהֲפָוךְ-בָּם.

זָעוּ, נָעוּ, דַּלּוּ מֵאֱנוֹשׁ מוֹשִׁיעִים,
עָתְקוּ, גַּם גָּבְרוּ חַיִל רְשָׁעִים.
חָדְלוּ אֳרָחוֹת, וְלֹא נִכָּרִים שׁוֹעִים –
סֵפֶר תַּנְחוּמֶיךָ מָרַת נֶפֶשׁ שַׁעֲשׁוּעִים.

טוֹפְלֵי שֶׁקֶר וְתַרְמִית, לִבָּם לְהַסְבִּיר
נוֹשְׂאֵי הֶבֶל פִּיצַת פִּימוֹ לְהַגְבִּיר;
יוֹצְרֵי עָמָל עֲלֵי חוֹק לְהַחְבִּיר,
מְצִיקִים צוֹרְרִים, עַמְּךָ מִמְּךָ לְהַעֲבִיר.

כּוֹנְנוּ לְהַשְׁחִית לְךָ בָּנִים, אֲבִיהֶם
לֹא תֶחֱשֶׁה לְעַנּוֹתָם וּלְכַלּוֹתָם אֹיְבֵיהֶם!
שַׁלָּמָּה הֱיוֹת קֹוֶיךָ טְרוּדֵי נְוֵיהֶם?
יֵעוֹרוּ גְאוּלִים דְּאוֹת לְעִיר מִקְוֵיהֶם!

אַדְמַת קֹדֶשׁ, מְקוֹם דְּבִירְךָ נַהֲגֵינוּ,
בְּקוֹל תּוֹדָה וְרִנָּה, הֲמוֹן חוֹגְגֵינוּ!
זִבְחֵי צֶדֶק קָרֵב לְךָ, מָגִנֵּינוּ,
אָמְרֵינוּ הַאֲזִינָה, יְיָ, בִּינָה הֲגִיגֵינוּ!״

״אִם עֲוֹנֵינוּ עָנוּ בָנוּ,
אָתָאנוּ לָךְ - וּבְשִׁמְךָ בָאנוּ;
כְּרַחֵם אָב תָּשׁוּב, תְּרַחֲמֵינוּ,
כְּאִישׁ אֲשֶׁר אִמּוֹ תְּנַחֲמֶנּוּ.
אַל בְּאַפְּךָ, פֶּן תַּמְעִיטֵנוּ.

אַתָּה יְיָ - טוֹב וְסַלָּח.
אוֹתָנוּ - רוֹעַ יֵצֶר אִילַּח.
בָּנֶיךָ בְּיַד פִּשְׁעָם מִלְּשַׁלַּח –
בַּיִר מִסִּרְיוּת לִבֵּינוּ הַמְלַח.

גְּדָל-חֶסֶד, נוֹצֵר לָאֲלָפִים –
גַּלְגֵּל מִידּוֹתֶיךָ, וְלֹא חִילּוּפִים!
דְּבָרְךָ לֹא רֵיקָם וְסִילּוּפִים,
דִּימִּינוּ חַסְדֶּךָ כַּעֲיֵיפִים וַעֲלוּפִים.

הַסְכֵּת צֵת צִיר, הִתְבּוֹנֵן,
הָאֵל, דְּבָרְךָ אֵם לְגוֹנֵן;
וִיתַּקְתּוֹ דְרָכֶיךָ, חַנּוּן, שַׁנֵּן –
וּבָהּ בַּמִּידָּה אוֹתָנוּ חַנֵּן.

זוֹכֵר נִשְׁכָּחוֹת - לְךָ הִילּוּל!
זַעַם תַּקְטִין כְּרֶגַע מִילּוּל.
חֲתוֹךְ חַיִּים דִּין פִּילּוּל,
חוֹן חִנָּם חׇוקְרָךְ חָלוּל.

טִיט נִגְרָשׁ - תּוֹכוֹ תַּצְלִיל,
טוֹעַ זָדוֹן, טָמוּן וַעֲלִיל.
יוֹדֵעַ יֵצֶר מַה מַּעֲלִיל
יָד לְטוֹבָה עָלֵינוּ תַּגְלִיל.

כְּלָיוֹת בּוֹחֵן - וּשְׁבִילְךָ לֵב,
כַּשְׁרוּת, יַשְׁרוּת, וּשְׁרִירוּת הַלֵּב,
לֵב עָלוּב עִם עוֹלֵב,
לָעוּן, מָתוֹק, שָׁב וּמְכַלֵּב.

מַצִּיל מָטֵי דִין הֶרֶג,
מְקוֹר קוֹרוֹת לְךָ עוֹרֵג,
נַקֵּל שַׁקֵּל עָוֹן מֵהִשְׂתָּרֵג,
נָשׂוּאי, נָשׁוּי, בְּבוֹא מְקַטְרֵג.

סִירְחוֹן כְּתָב יָד הַנֶּחְתָּם,
סִפְּרוֹ זַיֵּף, טוֹעֵן הִסְתַּתָּם,
עֵין עוֹז עָוֹן הַנִּכְתָּם –
עַל יְדֵי רַחֲמֶיךָ יִתָּם.

פַּנֵה אֶבֶן מִדַּרְכְּךָ, הִיתָּקֵן,
פַּלְגֵי מַיִם פָּנוּי מִלְּהִתְרוֹקֵן.
צוּר מִכְשׁוֹל נוֹעַר וָזוֹקֶן
צַמֵּת, וּכְרַחֲמָן - רַחֵף הַקֵּן!

קָרוֹב לְרַצּוֹת קִילּוּסְךָ אָמוּר:
קֳושִׁי לִיגְיוֹנְךָ - שָׁנוּי וְתָמוּר.
רִיבָךְ - לֹא לָנֶצַח שָׁמוּר;
רַחֵם בְּרוֹגֶז דִּין גָּמוּר.

שׁוֹבָבִים שָׁבִים - שַׁכַּח שַׁחֲרוּת!
שְׁתוּלִים יַישֵּׁר, יַלְדוּת, בַּחֲרוּת.
תָּגֵן חֵירוּת - מִיַד תַּחֲרוּת,
תָּחוֹן תַּיו חַיִּים חָרוּת!

שָׁלוֹם, שָׁלוֹם בַּשֵּׂר! מִתְקָרֵב
מְעַט - סַיַּיע, קַדֵּשׁ הֶרֶב.
הַבַּכּוּרוֹת וְחִילּוּפֵיהֶן כְּאַחַת עָרֵב,
הָקֵשׁ טֶנֶא. יַחַד הַעֲרֵב!
אַל בְּאַפְּךָ פֶּן תַּמְעִיטֵנוּ!״

״אִם עֲוֹנֵינוּ, רַבּוּ לְהַגְדִּיל,
בָּנוּ עָנוּ, עָבוֹת כְּגָדִיל,
גָּרְמוּ לָנוּ בִּנְתַּיִם לְהַבְדִּיל –
דַּרְכֵי רַחֲמֶיךָ לֹא תַּחְדִּיל.

הִתְנַהֵג בְּמִידּוֹתֶיךָ-חֶסֶד הִתְנֵיתָה,
וְאַתָּה הוּא, לֹא שָׁנִיתָ,
זְכוֹר עֲדָתְךָ קֶדֶם קָנִיתָ,
חוֹן שִׁירֵי בְּכוֹר כִּינִּיתָה.

טִיעַנְתָּנוּ גַּפֵּי קֶרֶת נְתוּנִים,
יִשַּׁבְתָּנוּ שֶׁן סֶלַע אֵיתָנִים:
כְּאַחַת דִּיכִּיתָנוּ בִּמְקוֹם תַּנִּים,
לָרְוָיָה צֵאת כַּמָּה מְתוּנִים.

מֵרוֹב פְּקֻודָּה וּבֶהָלָה מְחַלְחֶלֶת,
נַפְשֵׁינוּ נָקְטָה, שָׁחָה לֶעָפָר בּוֹחֶלֶת,
סָמְכָה לָאָרֶץ בֶּטֶן נִשְׁחֶלֶת –
עוּרָה, לָמָּה תִישַׁן, תּוֹחֶלֶת?!

פְּקַח קוֹחַ אֲסִירֶיךָ חֲפוֹץ,
צוֹק הָעִיתִּים - חֶשְׁבּוֹנָם קְפוֹץ.
קַבֵּץ קֹוֶויךָ, עֵדֶר הַנָּפוֹץ,
רְאוֹת עוֹלָתָה וּפִיהָ תִקְפּוֹץ.

שְׁמוֹר שְׁבוּעַת חֶסֶד, וּתְנַאי
תָּם, וְצָבוּר, בַּנַּי, אֵיתָנַי,
שְׁלוֹמוֹ יְצַוֶּה לִבְלִי גְנַאי,
הֲפוֹךְ וְשַׁנּוֹת לְטוֹבָה הַפְּנָאי.

קָטָן כִּי יַעֲקֹב וָדָל,
יְדוּעַ חוֹלִי, וְנִבְזֶה, וַחֲדָל,
חַיִּים וָחֶסֶד, מָעוֹז וּמִגְדָּל,
כַּאֲשֶׁר עַתָּה כֹּחֲךָ יִגְדָּל!״

״אָמְנָם הֲרֵעֹנוּ מַעֲשֵׂינוּ
בְּשׇׁוחְכֵינוּ יְיָ עוֹשֵׂנוּ,
גִּלּוּלִים שַׂמְנוּ מַחֲסֵינוּ,
וְהָרַע בְּעֵינֶיךָ עָשִׂינוּ.

עָשִׂינוּ דַּרְכֵי תּוֹעַ יִצְרֵינוּ,
הַחִלֹּנוּ שֵׁם יוֹצְרֵינוּ.
וְעַתָּה, בִּשְׁנֵי יְצָרֵינוּ –
מִתְוַדִּים: חָטָאנוּ צוּרֵינוּ.

צוּרֵינוּ, זִהַרְתָּנוּ דָּת וְחֻקִּים
חֵשֶׁק הוֹרִים לְהָקִים,
טָעִינוּ, וּבְקָדְשְׁךָ מְפַקְפְּקִים,
וְקַמְנוּ בְּלֹא יָקִים.

יָקִים, יָקוּם, בְּצֶלֶם נִשְׁלַח
כְּשִׁיחַתְנוּ, נָוֶה מְשֻׁלָּח.
לְמַעַנְךָ, טוֹב וְסַלָּח,
לַחֲטָאֵינוּ סְלָח.

סְלַח מַעַל, תַּכְלִית מֶמֶר
נֶשֶׁף אָרְכֵּנוּ לְהִכָּמֵר.
סִלַּפְנוּ חֹק לְהִתְעַמֵּר,
יְיָ יִתֵּן אוֹמֶר.

אוֹמֶר עֲשׂוֹת לֹא חִלְחַלְנוּ,
פָּעַלְנוּ שֶׁקֶר וְנֹחַלְנוּ.
צִדְקוֹתֶיךָ וְחַסְדְּךָ יִחַלְנוּ –
צוּרֵינוּ סְלַח לָנוּ.

לָנוּ קְדוּמָה חִיוִּיתָ לִלְמוֹד,
רְכוּשׁ רֵיעַ מִלַּחֲמוֹד,
שְׁקוֹד דָּתְךָ לִתְמוֹד –
פִּקּוּדֶיךָ לִשְׁמוֹר מְאֹד.

מְאֹד תִּעְתַּעְנוּ בְּחוֹזֵי צִירֵינוּ
שַׁחַר וָנֶשֶׁף מַפְצִירֵינוּ.
לֹא תִטּוֹר לַהֲצִירֵינוּ –
סְלַח לָנוּ יוֹצְרֵינוּ!

יוֹצְרֵינוּ, מְנָת חֶלְקֵינוּ מִלְּפָנִים,
הוֹדֵינוּ גְלוּיִּם וּצְפוּנִים;
הַצְּדָקָה לְךָ! - סֹפְנִים,
וְלָנוּ בֹּשֶׁת הַפָּנִים.

הַפָּנִים קַדֵּם, בְּתוֹרַת מִבְטָאֵינוּ
טָעַנְנוּ: חָטָאנוּ בְּשִׁטָּתֵינוּ!
נָא, כְּשִׁיטָּתְךָ לְהַטּוֹתֵינוּ –
וְסָלַחְתָּ לַעֲוֹנֵנוּ וּלְחַטָּאתֵנוּ!״

״אָמַרְנוּ: נִגְזַרְנוּ לָנוּ, וְאֵין דּוֹרֵשׁ.
אֲבָל אֲשֵׁמִים אָנִינוּ, וְקָרוֹב פּוֹרֵשׁ.
בַּעֲוֹן בִּצְעֵינוּ - צוּר לַצָּר יַחֲרֵשׁ,
בּוֹלֵעַ וּמֵדִיק וְרָד וְרוֹפֵס הַחוֹרֵשׁ.

גַּל אַחַר גָּל - וְאֵין דּוֹבֵר –
גּוֹלֵל וָצָף עֲלֵי רֹאשׁ גֶּבֶר.
דֶּבֶק שֶׁבֶר עַל שֶׁבֶר חוֹבֵר –
דַּאֲבוֹן הַבָּא קָשֶׁה יִשָּׁכַח הָעוֹבֵר.

הִמֵּס יִמַּס הַלֵּב מֵרוֹב הַמִּכְשׁוֹל
הוֹלֵךְ וְסוֹעֵר כִּי רַב הַנַּחְשׁוֹל,
וּמוֹשִׁיעַ וָרַב אֵין עוֹד לִמְשׁוֹל,
וְאָבַד הַמַּנְהִיג וְעָיֵף שׁוֹל נִישּׁוֹל.

זָנַחְנוּ טוֹב - עַל כָּכָה הִגִּיעַ,
זוֹכְרֶיךָ זִיעַמְתָּ בְּיַד עֲוֹנֵינוּ לְהַפְגִּיעַ.
חַיִּים לִירֶץ וֶוסְתְּךָ, אַף לְהַרְגִּיעַ,
חוּשָׁה לַעֲזוֹר, כִּי אֵין מַפְגִּיעַ.

טוֹרַדְנוּ, וַנְּהִי כַטָמֵא, וּכְבֶגֶד עִידִּים.
טוֹרַפְנוּ וַנִּבֶל, וְרוֹדְפִים קַלִּים חַדִּים.
יוֹצְאֵי חוּצוֹת - מְלֵיאֵי גַעַר כִּמְנֻודִּים,
יוֹשְׁבֵי חֹשֶׁךְ צָגִים כְּרֵיקִים כַּדִּים.

כְּלִימּוֹת וָרוֹק פָּנֵינוּ מְצַפִּים צָפֹה,
כִּי גָדַל הַכְּאֵב פֹּה וָפֹה
לַיּוֹצֵא - בְּעָתָה, וְלַבָּא אֵין רָפוֹא,
לָחוֹן וּלְרַחֵם תִּקְוַת אַיֵּה אֵיפוֹא?

מַה כֹּחִי לְיַחֵל - וּמַתֶּשֶׁת גְּבַרְתָּנִית?
מוֹסִיף, מוֹשֵׁךְ קֵץ מְעַנִּית גֵּיוְתָנִית.
נֵטֶל נַגְּחָנִית, נַשְּׁכָנִית, עֶשֶׂר שׁוֹלְטָנִית,
נָגוֹעַ יְיַגַּע רוֹדְיָנִית, חִנָּם, חׇולְטָנִית.

סִימְלוֹנֵי רָשָׁע לְקַצֵּץ נָקָם תִּלְבּוֹשׁ,
סֵבֶר קֹוֶיךָ וְסִכּוּיָים תַּעֲמִיד מִלֵּבוֹשׁ.
עֶגְלַת לִימּוּדָךְ - מַחַץ וָשֶׁבֶר תַּחֲבוֹשׁ,
עֲבוֹד עֲבוֹדָתְךָ שְׁכֶם אֶחָד תִּכְבּוֹשׁ.

פָּקַרְנוּ בְּאֵימוּן כְּבֵן הַפְכָן וְסָרְבָן,
פִּידְיוֹן לְחִיזּוּק - יִיסּוּרִין חִילַּפְתָּ קָרְבָּן.
צָרוֹת מְצֵירוֹת, נָחָשׁ, עַקְרָב וְרַבְרְבָן,
צִימָּאוֹן וְחוֹסֶר כָּל הַמְשֻׁונֶּה - דָּרְבָן.

קוֹבְלִים הֶסְתֵּר פָּנֶיךָ עַם טוֹרְחָן,
קִינְיָין הֶרֶג-רָב מָכוּר וּמוּרְחָן.
רִיגַּלְנוּ כְּעֶבֶד רַב-פֶּשַׁע וְסוֹרְחָן –
רַבֵּה לְהֵטִיב, רַב-חֶסֶד וְסוֹלְחָן.

שִׁיקַּרְנוּ בְךָ, סַטְנוּ מִמְּךָ לָסוּר,
שְׁכַחְנוּךָ בְּרוֹב כֹּל, בְּלִי חִיסּוּר.
תָּעִינוּ בְּיֵצֶר, תּוֹעַ רוֹעַ, סוּר,
תַּרְנוּ אַחַר הַזּוּג לַעֲבֵירָה סַרְסוּר.

שְׁרִירוּת לֵב יַישֵּׁר, פְּשׁוֹט עַקְמוּמִית,
לֹא יֻרְגַּז עוֹד בְּהִירְהוּר שַׁעֲמוּמִית.
מַהֵר מִלְּעַכֵּב טִימּוּי תְּפִיסַת סְמָמִית –
הֶרֶב כַּבֵּס כְּתָמֵינוּ וְלַבֵּן אַדְמוּמִית.

הֲגוּנִים לְךָ לְעָם אוֹתָנוּ נַוֵּה,
קָטֹן וְגָדוֹל דַּעַת דְּרָכֶיךָ שַׁוֵּה,
יְשׁוּעוֹת יַעֲקֹב חֵלֶק יִשְׂרָאֵל צַוֵּה
הֲלֹא אַתָּה אֱלֹהֵינוּ - וְלֵךְ נְקַוֶּה!״

״אֲנִי יוֹם אִירָא - אֵלֶיךָ אֶקְרָא,
בַּל יַעַשְׁקוּנִי זֵדִים, עוֹזְבֵי יְקָרָה.
גְּמוּל לְהָשִׁיב לָהֶם - שֵׁב לְבַקְּרָה,
דִּין רָשָׁע וְעַוָּל מְלוֹאָם יִקְרָא.

הַחוֹשְׁבִים לְהַשְׁכִּיחַ שֵׁם קוֹדֶשׁ הַנִּכְבָּד
וּלְהַרְגִּיל שֵׁם טֻומְאָה, הַנִּקְלֶה וְנֶאֱבָד,
זֶה דַּרְכָּם, טוּבֵי עַם אִיבֵּד –
חֲשׂוךְ הַנִּשְׁאָר - וְכׇותְשֵׁם בְּבֵית הַבָּד.

טְרוֹף טֶרֶף לִמְּדוּ לְאָדָם, כַּמְדֻובָּר,
יַדּוּ גוֹרָל - כְּעַל הֶפְקֵר בַּמִּדְבָּר,
כִּמְעַט כָּלוּ, לֹא יְעַדְּרוּ דָּבָר –
לוּלֵי רַחֲמֶיךָ אֲדוֹן-חֵן, בְּחֶסֶד דַּבָּר.

מִי יְהַרְהֵר אַחַר מִידּוֹתֶיךָ לְדַיְּינָה?
נׇוכְחֲךָ לְהָשִׁיב - מִי חֲלָצָיו לְזַיְּינָה?
סַרְנוּ וְסָעַרְנוּ, וְנִשְׁמְטָה חֶרֶב הַיּוֹנָה:
עִמְּךָ הַדִּין, וְיָדְךָ עַל הָעֶלְיוֹנָה.

פֶּן יֹאמַר: 'יָכוֹלְתִּי' בַּעַל הַמּוֹט,
צָרֵי יְהוּדָה יָגִילוּ כִּי אֶמּוֹט,
קְרָא לִמְכוּרֵי חִנָּם שְׁנַת שָׁמוֹט,
רָשָׁע הַפּוֹשֵׁט - יִסּוֹג אָחוֹר וְיִקְמוֹט.

שְׁטָרְךָ קוֹדֵם - וְלוֹ נוֹשִׁים מְבַקְּשִׁים.
שִׁעְבּוּדְךָ לְהַרְחִיק רַבּוּ אֲדוֹנִים קָשִׁים;
תְּחוּמֵי סִמָּנֵי מְצָרֶיךָ לַעֲקוֹר מְבַקְשִׁים,
תְּבוּאָה קִידַּשְׁתָּ - בְּלוּעָה, מְלִילוֹת וְקַשִּׁים.

שֶׁפֶט רָעָה תָּבֹא אֲלֵיהֶם לְהַאֲשִׁימֵם,
לָמוֹ תְּעוֹלֵל, וְהִתְעוֹלֵל, בְּכֹבֶד-יָד יְשִׁימֵּם,
מְנָת כּוֹסָם - פַּחִית וּמַפָּח הַגְשִׁימֵם,
הָשֵׁב שִׁבְעָתַיִם נְקַם בְּרִית, וְהַשְׁמֵם!

הָעִיר קִרְיַת טִירַת נָוְךָ נַחֲלֵינוּ,
יָם גְּאוּלִים דֶּרֶךְ לַעֲבוֹר קְהָלֵינוּ,
בְּךָ תּוֹחַלְתֵּינוּ, וְאַתָּה רַב, הַמְחוֹלְלֵינוּ –
נַחֲמֵינוּ נָא, יְהִי חַסְדְּךָ עָלֵינוּ!״

״[זְכוֹר בְּרִית אַבְרָהָם וַעֲקֵידַת יִצְחָק
וְהָשֵׁב שְׁבוּת אָהֳלֵי יַעֲקֹב, וְהוֹשִׁיעֵנוּ
לְמַעַן שְׁמֶךָ.]

אַשְׁמָתֵינוּ כִּי רַבָּה,
בֵּית מִקְדָּשֵׁינוּ לְחָרְבָּה;
בָּגַדְנוּ בִּיתֵירָה חִיבָּה –
וַתְּהִי לְהֶפֶךְ וְסִיבָּה.
[וְשׁוּב בְּרַחֲמִים עַל שְׁאֵירִית יִשְׂרָאֵל וכו']

גָּלָה כְּבוֹדֵינוּ וְנֻוטַּל,
וְרַב בּוּז הֻוטַל,
דָּמָם שִׁיר וּבֻוטַּל
מִבֵּית עֶלְיוֹן הַמְנֻוטָּל.

הֹוָה עַל הֹוָה,
עַוָּה עַל עַוָּה,
וַעֲנֻוגָּה וְרַכָּה וְנָאוָה –
כָּל הַיּוֹם דָּוָה.

זָנַח יִשְׂרָאֵל לֶקַח
וְרֻודַּף מִבֵּית הָרֶקַח,
חֻבַּל כְּעֶבֶד מִקָּח
הַנִּלְקָח בְּאֶבֶן הַלֶּקַח.

טִיפְּשׁוּת אָוֶן אֵירַחְנוּ,
וּמֵעֲבֵירָה לֹא בָרַחְנוּ,
יוֹם נִטְעֵינוּ סָרַחְנוּ,
סֻוגְסַגְנוּ וְלַעֲנָה הִפְרַחְנוּ.

כִּיסַּתְנוּ בוֹשֶׁת וְנִכְלַמְנוּ,
וְלִפְנֵי גוֹזְזֵינוּ נֶאֱלַמְנוּ,
לְפִי דַרְכֵינוּ נֶהֱלַמְנוּ –
כַּאֲשֶׁר עָשִׂינוּ, שֻׁולַּמְנוּ.

מוּעָקָה בְּמָתְנֵינוּ שַׂמְתָּה,
וְיוֹם רְוָויֵינוּ אָסַמְתָּה,
נֶחֱרָפִים –וּפִינוּ חָסַמְתָּה,
כָּלִינוּ - וְלֹא סִיַּימְתָּה.

סַרְנוּ, סַגְנוּ וְנֶאֱלַחְנוּ,
לָבַז מִלְּפָנֶיךָ זֻונַּחְנוּ,
עֲבוֹדָתְךָ לֹא פָלַחְנוּ,
שִׁילַּחְתָּ חִיצֶּיךָ וּפֻולָּחְנוּ.

פְּעֻולַּת רִאשׁוֹנִים חֲסִידֶיךָ
זְכוֹר, וְיִעוּד יְסוֹדֶיךָ,
צַוֵּה יְשׁוּעוֹת עֲבָדֶיךָ,
גַּלֵּה כָּמוּס סוֹדֶיךָ.

קָשַׁרְנוּ בְּאַחְוָה, וְתָפֵל,
וְלִמְאֹד שַׁחְנוּ שָׁפֵל;
רָם, הֲקִימֵנוּ מִנֶּפֶל,
וְהוֹצִיאֵנוּ מאור לאופל.

שְׁעֵה שִׂיחַ הָעֲטוּפִים,
שִׁיבּוֹלֶת מַיִם שְׁטוּפִים,
תּוֹךְ וְחָמָס חֲטוּפִים,
שִׁירַיִים, כַּגֶּפֶן קְטוּפִים.

שַׁתֵּף אֵשׁ וְלֶהָבָה
לְהַשְׁבִּית נוֹגֵשׂ וּמַדְהֵבָה,
מִצָּר עֶזְרָתֵינוּ הָבָה,
הָקִיצָה, וְעוֹרֵר אַהֲבָה.

חַי לֹא יִשָׁן,
זְכוֹר עֵדוּת שׁוֹשָׁן,
קוּמָה כַּחוֹק הַיָּשָׁן –
וְיָפוּצוּ אֹיְבֶיךָ כֶּעָשָׁן!
[וְשׁוּב בְּרַחֲמֶיךָ וְכוּ'.]״

״אָתָאנוּ לְךָ בַּקֵּשׁ רַחֲמֶיךָ,
כִּי עֲוֹנֵינוּ עָבְרוּ שְׁמֶ<י>ךָ,
לְנֹקְשֵׁי דֶלֶת - בַּשֵׂר תַּנְחוּמֶיךָ.
יְיָ - עֲשֵׂה לְמַעַן שְׁמֶךָ.

תָּפֵל טַחְנוּ וְאַשְׁמָה,
תּוֹרָה זָנַחְנוּ וּמְזִימָּה,
שִׁחַתְנוּ בְּחַטָּאת עֲצוּמָה,
שָׁוְא דִּבַּרְנוּ וְתָרְמָה.

רוּחַ חוֹפֵשׂ וְקֶרֶב,
רִחַקְנוּךָ - אַתָּה תְקָרֵב.
קִשְׁיוֹן עֹרֶף וָסֶרֶב
קַשֵּׁט, כַּבֵּס הֶרֶב.

צִיצִת תִּפְאֶרֶת נוֹבֵל,
צַעֲקִי מֵעִינּוּי כֶּבֶל!
פְּתַח שַׁעַר שובל,
פֶּלֶל קַבֵּל מִקּוֹבֵל!

עַלִּיזָה - צָהֲלָה וְשָׂמֵחָה,
עֲלוּבָה - עֲזוּבָה וּזְנוּחָה;
סְגֻולָּה, גֶּפֶן פּוֹרְחָה,
סֹעֲרָה, כְּמֵת נִשְׁכְּחָה.

נָכוּל, נִיוּל, נִשְׂקַד,
נַחֵץ, יִסָּתֵר וְנִפְקַד.
מִפְעַל פִּרְחוֹ עָקַד
מַהֵר לְפָנֶיךָ יְפֻוקַּד,

לְהַדְמִים מִתְלַהְלֵהַּ נִרְגָּן,
לַמַּשְׁחִית בְּלִי לְהִתְמַגָּן,
כְּיוֹשֶׁר קָרְבָּן מְהֻוגָּן
כּוֹנֵן לְסַהַר אַגָּן.

יָחִיד, מִשְׁפָּטֵנוּ יָהֵל,
יִשָּׂא, עָלוֹת יְנַהֵל,
טִהוּר לִסְגֻולָּה הַצְהֵל –
טִכּוּס יוֹשֵׁב אוֹהֶל.

חַטָּאוֹת אִם נוֹסָפִים,
חֲזֵה חוֹסֶיךָ טְרוּפִים,
זְבוּלֶיךָ וְדָתְךָ נִשְׂרָפִים –
זִקוֹת אֵשׁ וּרְשָׁפִים.

וַעֲלֵי כָל אֵלֶּה,
וִעַדְנוּ, פָּנֶיךָ לְחַלֵּה,
הַפְלֵא חֶסֶד לְנִקְלֶה,
הֲמוֹן רַחֲמִים הִמָּלֵא.

דַּרְכֵי צִיּוֹן הָאֲבֵלוֹת –
דְּרוּשִׁים יַעֲלוּ בִמְסִלּוֹת.
גָּדְלְךָ, מַפְלִיא גְדוֹלוֹת,
גְּאוּלִים יַזְכִּירוּ תְהִלּוֹת.

בְּזַעַם תָּדוּשׁ אִיִּים,
בֵּיתְךָ שָׂמוּ לְעִיִּים,
אֶבְיוֹנִים, דַּלִּים וַעֲנִיִּים,
אֲמֻוצִים יַעַבְרוּ כִתִּיִּים.

שׁוּבָה וְהִנָּחֵם לַעֲבָדֶיךָ,
לְמַעַן יֵחָלְצוּן יְדִידֶיךָ.
מָלְאָה הָאָרֶץ כְּבוֹדֶךָ,
הַשְׁמִיעֵנִי בַבֹּקֶר חַסְדֶּךָ!״

״חָנֵּנוּ יְיָ חָנֵּנוּ, עֲנֵנוּ יְיָ עֲנֵנוּ,
עָזְרֵנוּ אֱלֹהֵי יִשְׁעֵנוּ, כִּי עָלֶיךָ נִשְׁעָנּוּ!

אֵלֶּה בְּשָׁלִישֵׁמוֹ / וְאֵלֶּה בְּפָרָשֵׁימוֹ / בַּקְּרָב,
וַאֲנַחְנוּ בְּשִׁמְךָ יְכוֹלֶת לְהִתְעָרָב.
שׁוֹכֵן שְׁחָקִים - לְיִשְׁעֵינוּ תִקְרָב!

אֵלֶּה בְּלִגְיוֹנָם / וְאֵלֶּה בִּגְאוֹנָם / מִתְפָּאֲרִים,
וַאֲנַחְנוּ בְּרַב לְהוֹשִׁיעַ מִתְגַּבְּרִים
לְעוֹזֵר לְאֵין-כֹּחַ מַכְבִּירִים.

אֵלֶּה בְּמָגִינָּתָם / וְאֵלֶּה בְּצִינָּתָם / מִסְתַּחֲרִים,
וַאֲנַחְנוּ בַּאֲמִיתְּךָ מַחַס תָּרִים
מֵרוּחַ עָרִיצִים הֱיוֹת נִסְתָּרִים.

אֵלֶּה בַּהֲגָפַת תְּרִיסִים / וְאֵלֶּה בְּשִׁפְעַת קַלְגַּסִּים, /
וַאֲנַחְנוּ בְּמָגֵן חוֹסִים וּמִיָּגוֹן מִתְגַּיְּיסִים,
שׁוֹבֵר מַטֵּה גְּאוֹן גַּסִּים.

אֵלֶּה בְּהַצָּלָה וָשָׁעַט / וְאֵלֶּה בְּנִיצּוּחַ וָמָעַט, /
וַאֲנַחְנוּ בְּאוֹר כַּשַּׂלְמָה עָט,
מָרוֹם שָׂרָם דִּיכּוּי בָּעָט.

אֵלֶּה בְּקַרְנֵי נְגָחִים / וְאֵלֶּה בְּקוֹל צְוָוחִים, /
וַאֲנַחְנוּ בְּקוֹלוֹ כְמֵי שְׁבָחִים,
מְפָרֵק וּמְפוֹצֵץ סַלְעֵי צְחִיחִים.

אֵלֶּה בְּטִיכּוּס עַרְכָּן / וְאֵלֶּה בְּטַכְסִיסֵי צׇורְכָן, /
וַאֲנַחְנוּ בְּאֵין כְּעֶרְכּוֹ אַרְכֹן וְאַרְכָן,
מֵקִים מֻורְכָּן לַבָּמוֹת דּוֹרְכָן.

אֵלֶּה בְּנִיצְחָנֵימוֹ / וְאֵלֶּה בְּרוֹצְחָנֵימוֹ / הַבְטָחוֹת,
וַאֲנַחְנוּ בְּנֵצַח יִשְׂרָאֵל כַּפֵּינוּ שְׁטוּחוֹת,
מִלַּעֲרוֹץ וּמִלִּירָא וּמִלֵּירַךְ טוּחוֹת.״

״לְךָ יְיָ הַצְּדָקָה תִּלְבּוֹשֶׁת
כִּי אֱמֶת עָשִׂיתָ - וְלָנוּ הַבּשֶׁת,
וַאֲנַחְנוּ הִרְשַׁעְנוּ, וּבְפִינוּ אֲרֶשֶׁת:
נוֹאַלְנוּ, חָטָאנוּ, נָא לָנוּ אַל תָּשֵׁת!

תָּעִינוּ אַחַר יֵצֶר,
תִּעְתַּעְנוּ דִּין יוֹצֵר,
שָׁלוּחַ לֹא עוֹצֵר,
שַׁחַר וָעֶרֶב נוֹצֵר.

רַצְנוּ רַע הַרְגֵּל,
רָשַׁעְנוּ צֶדֶק הַעְגֵּל,
קָשַׁרְנוּ אָהַב הַדְגֵּל,
קִילְקַלְנוּ קֶסֶט גַּלְגֵּל.

צֵרַפְנוּ מָקוֹר רָטוֹב
צִמָּאוֹן, לֹא קוֹרְטוֹב,
פַּצְנוּ רַע לַטּוֹב
פָּרַקְנוּ עוֹל מִלַּחֲטוֹב.

עַל עֵסֶק כִּעוּרִים
עָזַבְנוּ אַלּוּף נְעוּרִים.
סוֹעִים בְּכֵן, וּסְעוּרִים,
סִגּוּף עוֹנִי וְצִעוּרִים.

נַחְנוּ בְמֶרֶד וּבְמַעַל
נוֹאַשְׁנוּ יֶשַׁע מִמַּעַל,
מְרִיֵינוּ מָאַס וְגָעַל,
מִידַּת רַחֲמִים נָעַל.

לֹא בְּצֶדֶק וּבְיוֹשֶׁר
לָנוּ נֶגְדְּךָ פַּשֵּׁר
כֹּחַ וְיִתְרוֹן הַכְשֵׁר –
כַּבִּיר - תֵּן לְפַשֵּׁר!

יָם שָׂם לְגֶשֶׁר,
מִקְוֵה יִשְׂרָאֵל כּוֹשֶׁר,
טְהוֹרִים תִּזְרוֹק וּתְחַשֵּׁר,
טֻומְאָה תַשְׁלִיךְ וּתְנַשֵּׁר.

חַבֵּא חֵטְא וּגְרָע,
חֲזוֹר אָחוֹר מֵרָע,
זְכִיּוֹת כַּף הַכְרַע,
זְרֵה חוֹבוֹת מִלְּהִפָּרַע.

וְאַתָּה הוּא פוֹנֶה,
וְלֹא בוֹזֶה, וְעוֹנֶה,
הָעָם עָנִי וּמְעֻונֶּה
הַיּוֹם לְטוֹבָה תִפְנֶה.

דַּלַת רֹאשׁ, דַּלָּתֶיךָ,
דּוֹפֶקֶת הַיּוֹם דְּלָתֶיךָ,
גַּלֵּה לָהּ גְּדֻולָּתֶיךָ,
גּוֹנְנָהּ, כִּי הִיא גְדֵילָתֶךָ.

דְּרוֹשׁ הַיּוֹם הִנָּם,
דֵּי דִגְלֵי מַחֲנָם.
גִּישׁוֹת עֵרֶךְ חִינּוּנָם –
גְּמוֹל חֶסֶד חִנָּם.

בְּרִיבּוּעַ תָּשִׁיב, כַּמָצאוּי
בְּמִּקְרָא מְשֻׁולָּשׁ, כַּפִּיצּוּי
אַחַר גְּמַר מִיצּוּי:
אֱכוֹל דָּצוּי וְרָצוּי.

בַּלַּע בֶּגֶד בּוֹגְדִים,
בּוֹא בְּחִימּוּץ בְּגָדִים
אֱסֹף אֲבוּדִים אֲגוּדִים,
אַחַר אֲחֵיהֶם אִיזְגַּדִּים.

שַׁדַּי שׁוֹכֵן שְׁחָקִים,
שׁוּר שִׁינַּיִם שְׁחוּקִים.
לְךָ לְבַד לְהָקִים
לְהוּקִים, בְּקוּקִים, נְקוּקִים.

מְרָדֵימוֹ, מְרָיֵימוֹ מְרוּקִים,
מַהֵר, מְחֵה מְחוּקִים,
הַבְּרִית וְהַחֶסֶד הָקִים,
הִיזָּכֵר הָעֵדוֹת וְהַחֻוקִּים.״

״תָּבוֹא לְפָנֶיךָ שַׁוְעַת חִינּוּן,
תְּהִי נָא קַשֶּׁבֶת תַּחֲנוּן,
שִׁמְעָה צֶדֶק, וְהַקְשִׁיבָה רִינּוּן,
שָׁר מֵישָׁרִים, מַעְלִים מֵרִינּוּן.

רֹאשׁ לְהָרִים נִכְלַמְנוּ, בּוֹשְׁנוּ,
רֵיחַ נִרְדֵּינוּ כִּי הִבְאַשְׁנוּ,
קִילְקַלְנוּ יְשָׁרִים וְתוֹרוֹת שִׁבַּשְׁנוּ,
קַרְקַע פָּנֵינוּ בְּכֵן כָּבַשְׁנוּ.

צַר וּמָצוֹק מִכָּל צָד,
צֹאן נִדַּחַת מֵאֵין מְצָד:
פָּנָה לְיָמִין - וַיִּגְזוֹר מַעֲצָד,
פַּחַד מִשְּׂמֹאל, וְצַיָּיד הַצָּד.

עֵינֶיךָ רְאוֹת תִּהְיֶינָה פְקוּחוֹת
עֳונִי וְעִינּוּי צָרוֹת הַמְּתוּחוֹת.
סֶפֶד לְרִינָּה, לְרִיצּוּי - תּוֹכֵחוֹת,
סַבּוֹת וְהָפוֹךְ בִּדְרָכֶיךָ הַנְּכוֹחוֹת.

נֻותַּנְנוּ בַּעֲוֹנֵינוּ לִשְׁבִי וּלְבִיזָּה,
נַחְנוּ, מְלָכֵינוּ וְשָׂרֵינוּ לְבוּזָה.
מֵרוּם נִכְבָּדוֹת וְאַהֲבָה עַזָּה –
מִיגַּרְתָּ לָאָרֶץ לְשַׁמָּה וּלְעִיזָה.

לֹא חִילִּינוּ לְהַקְדִּים תְּחִינָּה
לְהַשְׂכִּיל בַּאֲמִיתְּךָ מֵעֲלוֹת צַחֲנָה.
כָּלִינוּ כִּסְדוֹם בִּשְׁפַל הַטַּחֲנָה
כִּמְעַט רֶגַע - לוּלֵי תְּחִינָּה.

יֶתֶר הַפְּלֵיטָה לְהַשְׁאִיר חַסְתָּה,
יָתֵד וְגָדֵר תַּתָּה וְכִנַּסְתָּה:
טִלְטַלְתָּנוּ כְּנֶגֶד שָׁלֹשׁ מָאַסְתָּה
טִירַת כֶּסֶף בִּגְלָלֵינוּ רָמַסְתָּה.

חָבוֹל חָבַלְנוּ מַעַל לִמְעוֹל,
חֻובַּלְנוּ מֵעוֹל אֶל עוֹל,
זְכוֹר, צִוִּיתָ בְּלִי לִגְעוֹל –
זְרוּיִם לְקַבֵּץ, וּבָם לִבְעוֹל.

וְאַתָּה - אַחֲרֵי כָּל הַבָּא –
וַודַּאי וְצַדִּיק, וְלָנוּ הַדִּיבָּה,
הַיּוֹם, כְּמֵאָז, בְּלִי סִיבָּה,
הִנְנוּ לְפָנֶיךָ בְּאַשְׁמָה רַבָּה.

דַּלַּת עָם לְקֶלֶס וְחֵירוּף,
דְּחוּפִים, סְפִוּחים, נְתוּנִים לְטֵירוּף,
גָּלוּת וְשִׁיעְבּוּד, נִיסָּיוֹן וְצֵירוּף,
גַּלְגֵּל בְּחֶסֶד לִסְלִיחָה וְתֵירוּף.

בְּרַחֲמֶיךָ עוֹד, בְּרַבּוֹת עִתִּים,
בְּךָ נוֹשָׁעִים קַיֵּים: וְהוֹשַׁעְתִּים!
אֵלֶּה מֵרָחוֹק יָבוֹאוּ כִתִּים,
אֵלֶּה מִצָּפוֹן, צִים וְכִתִּים.

שֶׁלְּךָ הֵם, עֲבָדֶיךָ וְעַמֶּךָ,
לַבֵּב כִּימֵי קֶדֶם נְעִימֶיךָ!
מָשְׁכֵינוּ אַחֲרֶיךָ, שִׁימֵינִי בְּרִשּׁוּמֶיךָ –
הַכֹּל חֲפֵצִים לְיִרְאָה אֶת שְׁמֶךָ!

הַקָּטֹן לָאֶלֶף גַּדֵּל, רַחוּמֵינוּ,
וְהַצָּעִיר לְגוֹי לְהַעֲצִים בִּתְחוּמֵינוּ!
יַחַד, בְּכָל צִדְקוֹתֶיךָ, לְרַחֲמֵינוּ,
יָשׇׁוב-נָא אַפְּךָ, וּתְנַחֲמֵינוּ!״

״תּוֹחֶלֶת יִשְׂרָאֵל, חֶסֶד לֹא נֶעֱזָב,
תִּקְוָתָם-נֶצַח, אֵמוּן וְלֹא אַכְזָב,
שׁוֹמְרֵי חֲזוֹן-הַמּוֹעֵד רֹאוֹת כְּמַרְזָב,
שָׁחוּ, כָּלוּ, וְעֵמֶק הַבָּכָא זָב.

רוּם רַעֲנָן מַטָּה מַּטָּה, שָׁפֵל,
רֹאשׁ - לְזָנָב, וְעִקָּר - לַעֲרַאי וְטָפֵל,
קֵץ חֹשֶׁךְ, תַּכְלִית לְיוֹשְׁבֵי מַאֲפֵל,
קַוֵּה לְאוֹר - וְשִׁית עָנָן וַעֲרָפֶל.

צְבִי קֹדֶשׁ חִלַּלְתָּ 'רֵאשִׁית' לְהַרְשִׁים –
צָץ הַמַּטֶּה, פָּרַח הַזָּדוֹן לַיּוֹרְשִׁים,
פִּנִּיתָ סְלָעִים וּטְרָשִׁים לְהַשְׁבִּיחַ שָׁרָשִׁים –
פְּרַצְתָּהּ, מְשׂוֹשׂ פְּרָאִים וַחֲזִירֵי חֳורָשִׁים.

עָרִיץ וְכָל הַבָּא - בּוֹזֵז וְשׁוֹלֵל,
עוֹבֵר וָשָׁב, וְזִיז שָׂדַי יְעוֹלֵל,
סָבָאִים שְׂבֵעִים. וּשְׁאָר - דָּשִׁים לְמוֹלֵל,
סֻוכַּת יֶתֶר הַבָּז לְזַלְזֵל, הִתְעוֹלֵל.

נֶטַע כַּנָּתְךָ לְמִגְזַר בַּרְזֶל חֲרוּצִים,
נָטַשְׁתָּ וְעָזָבְתָּ - וּבָאוּ בָּהּ פָּרִיצִים.
מְצֻויֶּינֶת חִיפּוּי כֶּסֶף, בִּירַקְרַק חֲרוּצִים,
מִדְרַס גַּז וָנֵץ, וְאֶפְרוֹחֶיהָ קְרוּצִים.

לְכָל הַבָּא עָלֵינוּ - מְעַט קָצַפְתָּ,
לָרָעָה עָזְרוּ גּוֹי עַז חִצַּפְתָּ,
כַּף עַיִן בְּעַיִן מִפְּרוֹעַ אָסַפְתָּ,
כְּלַפֵּי חֶסֶד כָּרוֹעַ גִּבּוּי חָסַפְתָּ.

יָד לְקַבֵּל פָּשַׁטְתָּ וְנָתַנְנוּ סוֹרֶרֶת,
יָד אָזְלַת מִקּוֹל כְּקוֹל הַנְּסוֹרֶת,
טוֹר חֶסֶד וּבְרִית לְאָבוֹת מַסּוֹרֶת,
טוּב טַעַם פְּתַח, לְהַתִּיר אֻוסֶּרֶת.

חֶשְׁבּוֹן אַחַר חֶשְׁבּוֹן עַמְּךָ יכתור
חוֹכֶה וְסוֹכֶה וְדָוֶה - וְחוֹזֵר לִכְתּוֹר,
זַח אוֹר מַשְׂכִּיל יְכוֹלֶתּ לַחְתּוֹר
זְמַן קוֹל הַתּוֹר מָתַי לִפְתּוֹר.

וּבְכָל זֹאת - בָּאֱמוּנָה חָלִילָה לְשַׁקֵּר
וְזָר-לֹא-יוֹעִיל קְלוֹנוֹ לְיַיקֵּר!
הֶעָקוֹב רַב, וְהַסַּתָּת תַּשׁ לְעַקֵּר,
הַדָּבָר עָלֶיךָ, כְּלָיוֹת חוֹקֵר וּמְבַקֵּר!

דְּרוֹשׁ וּפְקוֹד דַּלַת שִׁפְלוּת שְׁחוֹחַ,
דַּל כֶּרֶם וּמְעוֹרָב וְצָחֵי-צְחוֹחַ,
גְּדוֹר, עַזֵּק, עֲדוֹר, מִקִּימּוֹשׁ וָחוֹחַ,
גֶּפֶן מַטָּע לְהַכְשִׁיר עִנְבֵי נִיחוֹחַ.

בָּנוּ שִׁמְךָ, עֶלְיוֹן, לֹא תִכְלָא,
בּוֹא בְּמִדַּת חֶסֶד וְלֹא בְכָלָה,
אַשְׁמָתֵנוּ גָדְלָה - עַד תּוֹמֶן וְעֻוכְלָא,
אַתָּה יְיָ רַחֲמֶיךָ לֹא תִכְלָא.

שֶׁבַח הִילּוּל מִידּוֹתֶיךָ אֵין לְשַׁנּוֹת
לְבֵית יִשְׂרָאֵל רַב טוּב חַנּוּת,
מְקוֹם מָנוֹחַ שִׂים בְּרֹאשׁ פִּנּוֹת,
הָכֵן, בְּלִי רוֹגֶז וְתוֹסֶפֶת עַנּוֹת.

הִימָּצֵא חִישׁ לָנוּ כִּי נוֹחַלְנוּ,
קָרְבֵנוּ, הַט שְׂשׂוֹן עֵדוֹתֶיךָ לְנַחֲלֵינוּ,
טוֹב וְסַלָּח - לְךָ לְבַד הִתְחוֹלַלְנוּ,
נַחֲמֵינוּ נָא, יְהִי חַסְדְּךָ עָלֵינוּ!״

״תְּעָלַת צְרִי, תֶּרֶף וְחוֹבֶשׁ לְמָזוֹר,
תַּחְבּוּלוֹת מִלְחָמוֹת וְרַב-יוֹעֵץ לַעֲזוֹר!
שׁוֹטֵט אֵין לְבַד בְּךָ לַחֲזוֹר,
שַׁעֵן שַׁעֲשׁוּעַ-תַּנְחוּם דְּבֵיקֶיךָ כָּאֵזוֹר!

רָעָתֵינוּ רַבָּה, דִּיבַּת עָם מִסָּבִיב מָגוֹר,
רֶטֶט וְרִיפְיוֹן וְאֵין גִּיבּוֹר לַחֲגוֹר;
קוֹשֶׁט נוֹשֵׂא אֲלֻּמוֹת יַדְמִים קָטֵיגוֹר,
קִיחַת מֶשֶׁךְ הַזֶּרַע לְהַקְנוֹת סָנֵיגוֹר.

צִיחְצוּחַ תַּצְלִיחַ לִיכְלוּךְ גִּניַּת אֱגוֹזֶיךָ,
צְרוּפֵי אֵימֶיךָ, קְצוּפֵי הֶרְגֶּשׁ רָוגְזֶיךָ,
פְּרוֹט חוֹבָם דַּבְּרָנִים - וּכְרָחֵל נִגְזָזֶיךָ –
פִּתְחוֹן זְכִיּוֹתָם חׇון-נָא, מִגַּזֶּיךָ!

עָבְרוּ בָּאִים בִּזְרוֹעַ וְנַחַת דָּשֶׁן,
עַרְעָר הָעוֹלֶלֶת, הָאוֹכֶלֶת שִׁילּוּחַ הַשֵּׁן.
סַע חָזוֹן וְרָזוֹן, וְזַח הַחוֹשֶׁן,
שִׂיחַ תְּפִילַּת הָעֲטוּפִים תֵּפֶן מִלְּעַשֵּׁן.

נוֹצֵחַ לוֹ נוֹחַ - וּמִידָּתְךָ תְּנַכֵּר:
נוֹתֵן כֹּחַ לְהִינָּצֵחַ, שָׂמֵחַ וּמִשְׂתַּכֵּר.
מִשְׁפְּטֵי צִדְקֶךָ לֹא בְּשֶׁטֶף וְהַכֵּר,
מְדַבֵּר בִּצְדָקָה, נָס לְעֶזְרָה מִלְּהִתְנַכֵּר.

לֹא לְעוֹלָם תִּיטּוֹר, טוּבְךָ, כַּבִּיר,
לֹא כַחֲטָאֵינוּ תִּגְמוֹל וּמִידָּה תַּחְבִּיר.
כִּי כִגְבוֹהַּ שָׁמַיִם חַסְדְּךָ תַגְבִּיר,
כִּרְחוֹק מִזְרָח מִמַּעֲרָב פְּשָׁעֵינוּ תַעֲבִיר.

יוֹצְרֵינוּ, בְּיוֹשֶׁר שְׁבִילֶךָ - קַשֵּׁט בָּרוּר,
יַחַד לֵב עָקוֹב אַחֲרֶיךָ גָרוּר,
טְעוֹן עֻולְּךָ כּוֹף, כְּבוֹשׁ סָרוּר,
טִיף מֵי טוֹהַר, לְהַמְתִּיק מֵירוּר.

חָצִיר וְקַשׁ יָבֵשׁ, וְנִדָּף עָלֶה
חֲשׁוּבִים הִנֵּנוּ, כְּצִיץ הַשָּׂדֶה עוֹלֶה.
זָכוֹר כִּי אֲנַחְנוּ עָפָר וּבָלֶה,
זָכוֹר כִּי חַיֵּינוּ צֵל, וְכָלֶה.

וּמִי יַעֲמוֹד חֵטְא אִם תִּשְׁמוֹר?
וּמִי יָקוּם דִּין אִם תִּגְמוֹר?
הַסְּלִיחָה עִמְּךָ - סָלַחְתִּי לֵאמֹר,
הָרַחֲמִים גַּם לְךָ, מִדָּתְךָ, לְכְמוֹר.

דִּיכְדּוּךְ דַּלּוּתֵינוּ אַל נָא תַכְלִים,
דַּעַת נְתִיב דְּרָכֶיךָ חֶפְצֵינוּ תַשְׁלִים.
גָּדוֹל וְקָטוֹן רוּחַ שֵׂכֶל הַחֲלִים,
גִּיבּוֹרֵי כֹחַ, חָזָק וְאַמִּיץ, הַאֲלִים.

בְּצִילְּךָ שָׁבִים שֶׁבֶת - קַבֵּל נְדָבָה,
בְּבֵיתְךָ יַפְרִיחוּ וְלֹא יוֹסִיפוּ דַּאֲבָה,
אוֹבֵד וְנִידָּח תַּשְׁבִּית נוֹגֵשׂ וּמַדְהֵבָה,
אָז יַעֲלוּ וְיֵרָאוּ בְּרוּחַ נְדִיבָה.

שִׁילּוּם פָּרִים שְׂפָתֵינוּ תִּיכּוֹן אֱמֶת,
לֶכֶת דְּרָכֶיךָ תֹּם וָיוֹשֶׁר הַעֲמֵת,
מֵלִיץ צֶדֶק עָלַי קַבֵּל, וּמְלׇושְׁנִי צַמֵּת,
הֶחָפֵץ בַּחַיִּים וְלֹא בְּמוֹת הַמֵּת.

הֲקִימֵנוּ בְּאוֹר פָּנֶיךָ, חֶשְׁבּוֹן יִתְמַצֶּה,
קִיּוּם מֵרֶדֶּת שַׁחַת כּוֹפֶר יִמָּצֵא,
טֶרֶם נִקְרָא, עוֹד דִּיבּוּר יֵצֵא –
נְדָבוֹת פִּינוּ יְיָ נָא רְצֵה!״

״תַּעֲרוֹג אֵלֶיךָ כְּאַיָּל עַל אֲפִיקִים
שָׁחָה לְקֵץ רְוָיָה וְאֵין מפיקים,
רָחָב וְעָמוֹק – וְלִדְלוֹת אֵין מַסְפִּיקִים
קֹוִוים קְדוּרִים וַאֲמָנָה לֹא מְפִיקִים.

צְמֵאֶיךָ צוֹק לָמוֹ יִיחוּל לְבָבָם,
פָּנֶיךָ לֵרָאוֹת בֵּית מְעוֹן חֲבִיבָם,
עֲבוֹר בַּסָּךְ וַעֲמוֹד שֶׁבִיבֶיךָ סְבִיבָם –
סוּר טָמֵא וְעָרֵל, בַּעֲלֵי דְבָבָם.

'נַכְחִידֵם, נִירְשָׁה, לָּנוּ', נָמוּ, חָרָשׁוּ,
מִירֻושָּׁתְךָ אֲשֶׁר הוֹרַשְׁתָּ - עַמְּךָ גּוֹרָשׁוּ,
לְהַכְחִיד יוֹם יוֹם עוֹד יִדְרוֹשׁוּ,
כְּנוֹס מַס וְחָמָס, עֲנִיֶּיךָ יְרוֹשֵׁשׁוּ.

יָבוּזוּ, יָבֹזּוּ יְדִידֶיךָ, יְכַנּוּם כְּלָבִים,
טִפֵּשׁ, כּוּמָר וְנָבָל, תַּעַר הַגַּלָּבִים,
חֶרְפָּתָם שׁוֹמְעִים, עֲלוּבִים וְלֹא עֹלְבִים,
זָר לֹא נִשְׁלָבִים וְעָלֶיךָ נִצְלָבִים.

וַאֲנַחְנוּ עַמְּךָ, נֶגְדְּךָ זֶה עִינְיָנֵינוּ;
הַעַל אֵלֶּה תִּתְאַפַּק, תֶּחֱשֶׁה וּתְעַנֵּינוּ?
דִּכִּיתַנִי בִּמְקוֹם תַּנִּים, וְעָלֶיךָ עֵינֵינוּ,
גַּם גֵּיא צַלְמָוֶת שִׁיחַרְנוּךְ מָגִינֵּינוּ.

בָּנוּ שִׁמְךָ הַגָּדוֹל אַל יִתְחַלָּל,
בְּךָ כָּל זֶרַע יַעֲקֹב יִתְהַלָּל,
אִם עֲוֹנֵינוּ רַבּוּ וּמָלְאוּ חָלָל –
'אָנֹכִי מוֹחֶה פְשָׁעֶיךָ' פִּיךָ מִילַּל.

שְׁגִיאוֹת שְׁאוֹנָם - זֵדִים וְגַם אוֹנֶס –
לִפְנִים מִשּׁוּרַת הַדִּין לָנוּ הִכָּנֵס,
מִפְּנֵי קוֹשֶׁט סֶלָה יֵבוּשׁ הָאוֹנֵס,
הַנֵּס יְנוֹסֵס, נִדָּחַיי לִי יְכַנֵס.

הָעֲתָרַי בַּת פּוּצַי קָרְבָּן תֵּישַׁע,
קְרָב יַחְדָּיו עֲשִׂירִית עִם תֵּשַׁע,
טַבַּע חֵטְא, וְלֹא יִמָּצֵא רֶשַׁע –
נוֹשֵׂא עָוֹן וְעוֹבֵר עַל פֶּשַׁע!״

״תָּשׁוּב תְּרַחֲמֵינוּ וְשׁוּב שְׁבִיתֵנוּ כִּנְאוּמֶיךָ,
שׁוּב כְּקֶדֶם, חַדֵּשׁ חִיבַּת לְאֻומֶּיךָ.
רַחֵק רוֹגֶז וְשַׁכֵּךְ זַעְמֶיךָ,
שׁוּב מֵחֲרוֹן אַפֶּךָ וְהִנָּחֵם עַל הָרָעָה לְעַמֶּךָ.

קַבֵּץ וּבַקֵּשׁ אוֹבְדוֹת וְחַזֵּק נַחֲלָתֶךָ,
צֹאן הַהֲרֵיגָה וְהַגְּזוּזָה בִּרְחֵלוֹתֶיךָ.
פֶּשַׁע אִם רָב - תַּרְבֶּה מְחִילָתֶךָ,
שׁוּב לְמַעַן עֲבָדֶיךָ שִׁבְטֵי נַחֲלָתֶךָ!

עֲוֹנוֹת אֲבוֹתֵינוּ הִטְעִינָנוּ וּבוֹסֶר הִטְעִימָנוּ,
סַף רַעַל הִשְׁקָנוּ, הִלְעִינָנוּ וְהִזְעִימָנוּ,
נָא כְּאָז בִּשְׁכּוֹן יִחוּדָךְ הִנְעִימָנוּ –
שׁוּבֵנוּ אֱלֹהֵי יִשְׁעֵנוּ וְהָפֵר כַּעַסְךָ עִמָּנוּ.

מָאוֹס לֹא מָאַסְתָּ חוֹזֶיךָ, מְעִידֶיךָ.
לוֹקִים כְּסוֹרֵחַ - וּלְהִתְמָרֵחַ סִבְרֵינוּ עָדֶיךָ;
כַּעַס לְרַצּוֹת כְּמִידַּת יוֹשֶׁר מַעְבָּדֶיךָ –
שׁוּבָה יְיָ עַד מָתָי וְהִנָּחֵם עַל עֲבָדֶיךָ.

יוֹם יוֹם נְצֵפֶּה גִּילּוּי סוֹדֶיךָ
טָמוּם וְעָמוּם - וְכָלֶה בְּשַׂר חֲסִידֶיךָ,
חֶרְפָּה שָׁבְרָה לִבֵּנוּ, מְקַוִּים לַחֲסָדֶיךָ –
שׁוּבָהּ יְיָ חָלְצָה נַפְשֵׁנוּ, וְהוֹשִׁיעֵנוּ לְמַעַן חַסְדֶּיךָ.

זֻונַּחְנוּ וְנֶחְשַׁבְנוּ עִם יוֹרְדֵי רֶגֶב
וַיְהִי לְאֵבֶל כִּינּוֹרִי וְשִׁיר עֶגֶב.
הַקּוֹדְרִים - בְּרִינָּה, בְּתוֹדָה יִתְהַלֵּךְ, בְּשֶׂגֶב,
שׁוּבָה יְיָ אֶת שְׁבִיתֵנוּ כַּאֲפִיקִים בַּנֶּגֶב.

דִּמְעַת לֶחֶם חוֹסֶיךָ צוּרִי אֵל,
'גּוֹאֵל אַיֵּה' בְּחָרֵף צָר יוֹצְרֵי אֵל,
בְּחָזוֹן אָז כְּדִבָּרְתָּ לְיִשְׁרֵי אֵל –
שׁוּבָה יְיָ רִבְבוֹת אַלְפֵי יִשְׂרָאֵל.

שִׁמְעָה יְיָ, הַטֵּה אָזְנְךָ, וְהַקְשִׁיבָה,
